024package org.kohsuke.stapler;
025
026import net.sf.json.JSON;
027import net.sf.json.JSONArray;
028import net.sf.json.JSONNull;
029import net.sf.json.JSONObject;
030import net.sf.json.util.JSONUtils;
031import org.kohsuke.stapler.export.Flavor;
032
033import javax.servlet.ServletException;
034import java.io.IOException;
035import java.io.PrintWriter;
036import java.util.Collection;
037
038/**
039 * Pluggable interface that takes the return value from request handling
040 * methods and convert that to HTTP responses.
041 *
042 * @author Kohsuke Kawaguchi
043 */
044public abstract class HttpResponseRenderer {
045    /**
046     *
047     * @param node
048     *      Object that handled the request.
049     * @param response
050     *      The return value or the exception from the method.
051     * @return
052     *      true if the response object was understood and rendered by this method.
053     *      false otherwise, in which case the next {@link HttpResponseRenderer}
054     *      will be consulted.
055     */
056    public abstract boolean generateResponse(StaplerRequest req, StaplerResponse rsp, Object node, Object response) throws IOException, ServletException;
057
058    /**
059     * Default {@link HttpResponseRenderer}.
060     */
061    public static class Default extends HttpResponseRenderer {
062        @Override
063        public boolean generateResponse(StaplerRequest req, StaplerResponse rsp, Object node, Object response) throws IOException, ServletException {
064            return handleHttpResponse(req, rsp, node, response)
065                || handleJSON(rsp, response)
066                || handleJavaScriptProxyMethodCall(req,rsp,response)
067                || handlePrimitive(rsp, response);
068        }
069
070        protected boolean handleJavaScriptProxyMethodCall(StaplerRequest req, StaplerResponse rsp, Object response) throws IOException {
071            if (req.isJavaScriptProxyCall()) {
072                rsp.setContentType(Flavor.JSON.contentType);
073                PrintWriter w = rsp.getWriter();
074
075                // handle other primitive types as JSON response
076                if (response instanceof String) {
077                    w.print(quote((String) response));
078                } else
079                if (response instanceof Number || response instanceof Boolean) {
080                    w.print(response);
081                } else
082                if (response instanceof Collection || (response!=null && response.getClass().isArray())) {
083                    JSONArray.fromObject(response, rsp.getJsonConfig()).write(w);
084                } else
085                if (response==null) {
086                    JSONNull.getInstance().write(w);
087                } else {
088                    // last fall back
089                    JSONObject.fromObject(response, rsp.getJsonConfig()).write(w);
090                }
091                return true;
092            }
093            return false;
094        }
095
096        protected boolean handlePrimitive(StaplerResponse rsp, Object response) throws IOException {
097            if (response instanceof String || response instanceof Integer) {
098                rsp.setContentType("text/plain;charset=UTF-8");
099                rsp.getWriter().print(response);
100                return true;
101            }
102            return false;
103        }
104
105        protected boolean handleHttpResponse(StaplerRequest req, StaplerResponse rsp, Object node, Object response) throws IOException, ServletException {
106            if (response instanceof HttpResponse) {
107                // let the result render the response
108                HttpResponse r = (HttpResponse) response;
109                try {
110                    r.generateResponse(req,rsp,node);
111                } catch (IOException e) {
112                    if (!handleHttpResponse(req,rsp,node,e))
113                        throw e;
114                } catch (RuntimeException e) {
115                    if (!handleHttpResponse(req,rsp,node,e))
116                        throw e;
117                } catch (ServletException e) {
118                    if (!handleHttpResponse(req,rsp,node,e))
119                        throw e;
120                }
121                return true;
122            }
123            return false;
124        }
125
126        protected boolean handleJSON(StaplerResponse rsp, Object response) throws IOException {
127            if (response instanceof JSON) {
128                rsp.setContentType(Flavor.JSON.contentType);
129                ((JSON)response).write(rsp.getWriter());
130                return true;
131            }
132            return false;
133        }
134    }
135
136    static String quote(String text) {
137        return JSONUtils.quote(text);
138    }
139
140}
