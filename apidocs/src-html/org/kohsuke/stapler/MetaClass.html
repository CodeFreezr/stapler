027import org.apache.commons.io.IOUtils;
028import org.kohsuke.stapler.bind.JavaScriptMethod;
029import org.kohsuke.stapler.lang.Klass;

031import javax.servlet.ServletException;
032import static javax.servlet.http.HttpServletResponse.SC_NOT_FOUND;
033import java.io.IOException;
034import java.lang.reflect.Field;
035import java.lang.reflect.InvocationTargetException;
036import java.lang.reflect.Type;
037import java.util.ArrayList;
038import java.util.List;

041/**
042 * Created one instance each for a {@link Klass},
043 * that retains some useful cache about a class and its views.
044 *
045 * @author Kohsuke Kawaguchi
046 * @see WebApp#getMetaClass(Klass)
047 */
048public class MetaClass extends TearOffSupport {
049    /**
050     * This meta class wraps this class
051     * 
052     * @deprecated as of 1.177
053     *      Use {@link #klass}. If you really want the Java class representation, use {@code klass.toJavaClass()}.
054     */
055    public final Class clazz;
056    
057    public final Klass<?> klass;
058
059    /**
060     * {@link MetaClassLoader} that wraps {@code clazz.getClassLoader()}.
061     * Null if the class is loaded by the bootstrap classloader.
062     */
063    public final MetaClassLoader classLoader;
064
065    public final List<Dispatcher> dispatchers = new ArrayList<Dispatcher>();
066
067    /**
068     * Base metaclass.
069     * Note that <tt>baseClass.clazz==clazz.getSuperClass()</tt>
070     */
071    public final MetaClass baseClass;
072
073    /**
074     * {@link WebApp} that owns this meta class.
075     */
076    public final WebApp webApp;
077
078    /*package*/ MetaClass(WebApp webApp, Klass<?> klass) {
079        this.clazz = klass.toJavaClass();
080        this.klass = klass;
081        this.webApp = webApp;
082        this.baseClass = webApp.getMetaClass(klass.getSuperClass());
083        this.classLoader = MetaClassLoader.get(clazz.getClassLoader());
084        buildDispatchers();
085    }
086
087    /**
088     * Build {@link #dispatchers}.
089     *
090     * <p>
091     * This is the meat of URL dispatching. It looks at the class
092     * via reflection and figures out what URLs are handled by who. 
093     */
094    /*package*/ void buildDispatchers() {
095        this.dispatchers.clear();
096        ClassDescriptor node = new ClassDescriptor(clazz,null/*TODO:support wrappers*/);
097
098        // check action <obj>.do<token>(...)
099        for( final Function f : node.methods.prefix("do") ) {
100            WebMethod a = f.getAnnotation(WebMethod.class);
101            
102            String[] names;
103            if(a!=null && a.name().length>0)   names=a.name();
104            else    names=new String[]{camelize(f.getName().substring(2))}; // 'doFoo' -> 'foo'
105
106            for (String name : names) {
107                dispatchers.add(new NameBasedDispatcher(name,0) {
108                    public boolean doDispatch(RequestImpl req, ResponseImpl rsp, Object node) throws IllegalAccessException, InvocationTargetException, ServletException, IOException {
109                        if(traceable())
110                            trace(req,rsp,"-> <%s>.%s(...)",node,f.getName());
111                        return f.bindAndInvokeAndServeResponse(node, req, rsp);
112                    }
113                    public String toString() {
114                        return f.getQualifiedName()+"(...) for url=/"+name+"/...";
115                    }
116                });
117            }
118        }
119
120        // JavaScript proxy method invocations for <obj>js<token>
121        // reacts only to a specific content type
122        for( final Function f : node.methods.prefix("js") ) {
123            String name = camelize(f.getName().substring(2)); // jsXyz -> xyz
124
125            dispatchers.add(new JavaScriptProxyMethodDispatcher(name, f));
126        }
127
128        // JavaScript proxy method with @JavaScriptMethod
129        // reacts only to a specific content type
130        for( final Function f : node.methods.annotated(JavaScriptMethod.class) ) {
131            JavaScriptMethod a = f.getAnnotation(JavaScriptMethod.class);
132
133            String[] names;
134            if(a!=null && a.name().length>0)   names=a.name();
135            else    names=new String[]{f.getName()};
136
137            for (String name : names)
138                dispatchers.add(new JavaScriptProxyMethodDispatcher(name,f));
139        }
140
141        for (Facet f : webApp.facets)
142            f.buildViewDispatchers(this, dispatchers);
143
144        // check action <obj>.doIndex(...)
145        for( final Function f : node.methods.name("doIndex") ) {
146
147            dispatchers.add(new Dispatcher() {
148                public boolean dispatch(RequestImpl req, ResponseImpl rsp, Object node) throws IllegalAccessException, InvocationTargetException, ServletException, IOException {
149                    if(req.tokens.hasMore())
150                        return false;   // applicable only when there's no more token
151
152                    if(traceable())
153                        trace(req,rsp,"-> <%s>.doIndex(...)",node);
154
155                    return f.bindAndInvokeAndServeResponse(node,req,rsp);
156                }
157                public String toString() {
158                    return f.getQualifiedName()+"(StaplerRequest,StaplerResponse) for url=/";
159                }
160            });
161        }
162
163        // check public properties of the form NODE.TOKEN
164        for (final Field f : node.fields) {
165            dispatchers.add(new NameBasedDispatcher(f.getName()) {
166                final String role = getProtectedRole(f);
167                public boolean doDispatch(RequestImpl req, ResponseImpl rsp, Object node) throws IOException, ServletException, IllegalAccessException {
168                    if(role!=null && !req.isUserInRole(role))
169                        throw new IllegalAccessException("Needs to be in role "+role);
170
171                    if(traceable())
172                        traceEval(req,rsp,node,f.getName());
173                    req.getStapler().invoke(req, rsp, f.get(node));
174                    return true;
175                }
176                public String toString() {
177                    return String.format("%1$s.%2$s for url=/%2$s/...",f.getDeclaringClass().getName(),f.getName());
178                }
179            });
180        }
181
182        FunctionList getMethods = node.methods.prefix("get");
183
184        // check public selector methods of the form NODE.getTOKEN()
185        for( final Function f : getMethods.signature() ) {
186            if(f.getName().length()<=3)
187                continue;
188
189            WebMethod a = f.getAnnotation(WebMethod.class);
190
191            String[] names;
192            if(a!=null && a.name().length>0)   names=a.name();
193            else    names=new String[]{camelize(f.getName().substring(3))}; // 'getFoo' -> 'foo'
194
195
196            for (String name : names) {
197                dispatchers.add(new NameBasedDispatcher(name) {
198                    public boolean doDispatch(RequestImpl req, ResponseImpl rsp, Object node) throws IOException, ServletException, IllegalAccessException, InvocationTargetException {
199                        if(traceable())
200                            traceEval(req,rsp,node,f.getName()+"()");
201                        req.getStapler().invoke(req,rsp, f.invoke(req, rsp, node));
202                        return true;
203                    }
204                    public String toString() {
205                        return String.format("%1$s() for url=/%2$s/...",f.getQualifiedName(),name);
206                    }
207                });
208            }
209        }
210
211        // check public selector methods of the form static NODE.getTOKEN(StaplerRequest)
212        for( final Function f : getMethods.signature(StaplerRequest.class) ) {
213            if(f.getName().length()<=3)
214                continue;
215            String name = camelize(f.getName().substring(3)); // 'getFoo' -> 'foo'
216            dispatchers.add(new NameBasedDispatcher(name) {
217                public boolean doDispatch(RequestImpl req, ResponseImpl rsp, Object node) throws IOException, ServletException, IllegalAccessException, InvocationTargetException {
218                    if(traceable())
219                        traceEval(req,rsp,node,f.getName()+"(...)");
220                    req.getStapler().invoke(req,rsp, f.invoke(req, rsp, node, req));
221                    return true;
222                }
223                public String toString() {
224                    return String.format("%1$s(StaplerRequest) for url=/%2$s/...",f.getQualifiedName(),name);
225                }
226            });
227        }
228
229        // check public selector methods <obj>.get<Token>(String)
230        for( final Function f : getMethods.signature(String.class) ) {
231            if(f.getName().length()<=3)
232                continue;
233            String name = camelize(f.getName().substring(3)); // 'getFoo' -> 'foo'
234            dispatchers.add(new NameBasedDispatcher(name,1) {
235                public boolean doDispatch(RequestImpl req, ResponseImpl rsp, Object node) throws IOException, ServletException, IllegalAccessException, InvocationTargetException {
236                    String token = req.tokens.next();
237                    if(traceable())
238                        traceEval(req,rsp,node,f.getName()+"(\""+token+"\")");
239                    req.getStapler().invoke(req,rsp, f.invoke(req, rsp, node,token));
240                    return true;
241                }
242                public String toString() {
243                    return String.format("%1$s(String) for url=/%2$s/TOKEN/...",f.getQualifiedName(),name);
244                }
245            });
246        }
247
248        // check public selector methods <obj>.get<Token>(int)
249        for( final Function f : getMethods.signature(int.class) ) {
250            if(f.getName().length()<=3)
251                continue;
252            String name = camelize(f.getName().substring(3)); // 'getFoo' -> 'foo'
253            dispatchers.add(new NameBasedDispatcher(name,1) {
254                public boolean doDispatch(RequestImpl req, ResponseImpl rsp, Object node) throws IOException, ServletException, IllegalAccessException, InvocationTargetException {
255                    int idx = req.tokens.nextAsInt();
256                    if(traceable())
257                        traceEval(req,rsp,node,f.getName()+"("+idx+")");
258                    req.getStapler().invoke(req,rsp, f.invoke(req, rsp, node,idx));
259                    return true;
260                }
261                public String toString() {
262                    return String.format("%1$s(int) for url=/%2$s/N/...",f.getQualifiedName(),name);
263                }
264            });
265        }
266
267        if(node.clazz.isArray()) {
268            dispatchers.add(new Dispatcher() {
269                public boolean dispatch(RequestImpl req, ResponseImpl rsp, Object node) throws IOException, ServletException {
270                    if(!req.tokens.hasMore())
271                        return false;
272                    try {
273                        int index = req.tokens.nextAsInt();
274                        if(traceable())
275                            traceEval(req,rsp,node,"((Object[])",")["+index+"]");
276                        req.getStapler().invoke(req,rsp, ((Object[]) node)[index]);
277                        return true;
278                    } catch (NumberFormatException e) {
279                        return false; // try next
280                    }
281                }
282                public String toString() {
283                    return "Array look-up for url=/N/...";
284                }
285            });
286        }
287
288        if(List.class.isAssignableFrom(node.clazz)) {
289            dispatchers.add(new Dispatcher() {
290                public boolean dispatch(RequestImpl req, ResponseImpl rsp, Object node) throws IOException, ServletException {
291                    if(!req.tokens.hasMore())
292                        return false;
293                    try {
294                        int index = req.tokens.nextAsInt();
295                        if(traceable())
296                            traceEval(req,rsp,node,"((List)",").get("+index+")");
297                        List list = (List) node;
298                        if (0<=index && index<list.size())
299                            req.getStapler().invoke(req,rsp, list.get(index));
300                        else {
301                            if(traceable())
302                                trace(req,rsp,"-> IndexOutOfRange [0,%d)",list.size());
303                            rsp.sendError(SC_NOT_FOUND);
304                        }
305
306                        return true;
307                    } catch (NumberFormatException e) {
308                        return false; // try next
309                    }
310                }
311                public String toString() {
312                    return "List.get(int) look-up for url=/N/...";
313                }
314            });
315        }
316
317        if(Map.class.isAssignableFrom(node.clazz)) {
318            dispatchers.add(new Dispatcher() {
319                public boolean dispatch(RequestImpl req, ResponseImpl rsp, Object node) throws IOException, ServletException {
320                    if(!req.tokens.hasMore())
321                        return false;
322                    try {
323                        String key = req.tokens.peek();
324                        if(traceable())
325                            traceEval(req,rsp,"((Map)",").get(\""+key+"\")");
326
327                        Object item = ((Map)node).get(key);
328                        if(item!=null) {
329                            req.tokens.next();
330                            req.getStapler().invoke(req,rsp,item);
331                            return true;
332                        } else {
333                            // otherwise just fall through
334                            if(traceable())
335                                trace(req,rsp,"Map.get(\""+key+"\")==null. Back tracking.");
336                            return false;
337                        }
338                    } catch (NumberFormatException e) {
339                        return false; // try next
340                    }
341                }
342                public String toString() {
343                    return "Map.get(String) look-up for url=/TOKEN/...";
344                }
345            });
346        }
347
348        // TODO: check if we can route to static resources
349        // which directory shall we look up a resource from?
350
351        for (Facet f : webApp.facets)
352            f.buildFallbackDispatchers(this, dispatchers);
353
354        // check action <obj>.doDynamic(...)
355        for( final Function f : node.methods.name("doDynamic") ) {
356
357            dispatchers.add(new Dispatcher() {
358                public boolean dispatch(RequestImpl req, ResponseImpl rsp, Object node) throws IllegalAccessException, InvocationTargetException, ServletException, IOException {
359                    if(traceable())
360                        trace(req,rsp,"-> <%s>.doDynamic(...)",node);
361                    return f.bindAndInvokeAndServeResponse(node,req,rsp);
362                }
363                public String toString() {
364                    return String.format("%s(StaplerRequest,StaplerResponse) for any URL",f.getQualifiedName());
365                }
366            });
367        }
368
369        // check public selector methods <obj>.getDynamic(<token>,...)
370        for( final Function f : getMethods.signatureStartsWith(String.class).name("getDynamic")) {
371            dispatchers.add(new Dispatcher() {
372                public boolean dispatch(RequestImpl req, ResponseImpl rsp, Object node) throws IllegalAccessException, InvocationTargetException, IOException, ServletException {
373                    if(!req.tokens.hasMore())
374                        return false;
375                    String token = req.tokens.next();
376                    if(traceable())
377                        traceEval(req,rsp,node,"getDynamic(\""+token+"\",...)");
378
379                    Object target = f.bindAndInvoke(node, req,rsp, token);
380                    if(target!=null) {
381                        req.getStapler().invoke(req,rsp, target);
382                        return true;
383                    } else {
384                        if(traceable())
385                            // indent:    "-> evaluate(
386                            trace(req,rsp,"            %s.getDynamic(\"%s\",...)==null. Back tracking.",node,token);
387                        req.tokens.prev(); // cancel the next effect
388                        return false;
389                    }
390                }
391                public String toString() {
392                    return String.format("%s(String,StaplerRequest,StaplerResponse) for url=/TOKEN/...",f.getQualifiedName());
393                }
394            });
395        }
396    }

398    private String getProtectedRole(Field f) {
399        try {
400            LimitedTo a = f.getAnnotation(LimitedTo.class);
401            return (a!=null)?a.value():null;
402        } catch (LinkageError e) {
403            return null;    // running in JDK 1.4
404        }
405    }
406
407    private static String camelize(String name) {
408        return Character.toLowerCase(name.charAt(0))+name.substring(1);
409    }
410
411    private static class JavaScriptProxyMethodDispatcher extends NameBasedDispatcher {
412        private final Function f;
413
414        public JavaScriptProxyMethodDispatcher(String name, Function f) {
415            super(name, 0);
416            this.f = f;
417        }
418
419        public boolean doDispatch(RequestImpl req, ResponseImpl rsp, Object node) throws IllegalAccessException, InvocationTargetException, ServletException, IOException {
420            if (!req.isJavaScriptProxyCall())
421                return false;
422
423            req.stapler.getWebApp().getCrumbIssuer().validateCrumb(req,req.getHeader("Crumb"));
424
425            if(traceable())
426                trace(req,rsp,"-> <%s>.%s(...)",node, f.getName());
427
428            JSONArray jsargs = JSONArray.fromObject(IOUtils.toString(req.getReader()));
429            Object[] args = new Object[jsargs.size()];
430            Class[] types = f.getParameterTypes();
431            Type[] genericTypes = f.getParameterTypes();
432
433            for (int i=0; i<args.length; i++)
434                args[i] = req.bindJSON(genericTypes[i],types[i],jsargs.get(i));
435
436            return f.bindAndInvokeAndServeResponse(node,req,rsp,args);
437        }
438
439        public String toString() {
440            return f.getQualifiedName()+"(...) for url=/"+name+"/...";
441        }
442    }
