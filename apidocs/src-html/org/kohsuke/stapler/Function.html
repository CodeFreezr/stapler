097    /**
098     * Calls {@link #bindAndInvoke(Object, StaplerRequest, StaplerResponse, Object...)} and then
099     * optionally serve the response object.
100     *
101     * @return
102     *      true if the request was dispatched and processed. false if the dispatch was cancelled
103     *      and the search for the next request handler should continue. An exception is thrown
104     *      if the request was dispatched but the processing failed.
105     */
106    boolean bindAndInvokeAndServeResponse(Object node, RequestImpl req, ResponseImpl rsp, Object... headArgs) throws IllegalAccessException, InvocationTargetException, ServletException, IOException {
107        try {
108            Object r = bindAndInvoke(node, req, rsp, headArgs);
109            if (getReturnType() != void.class)
110                renderResponse(req, rsp, node, r);
111            return true;
112        } catch (CancelRequestHandlingException _) {
113            return false;
114        } catch (InvocationTargetException e) {
115            // exception as an HttpResponse
116            Throwable te = e.getTargetException();
117            if (te instanceof CancelRequestHandlingException)
118                return false;
119            if (renderResponse(req,rsp,node,te))
120                return true;    // exception rendered the response
121            throw e;    // unprocessed exception
122        }
123    }
124
125    static boolean renderResponse(RequestImpl req, ResponseImpl rsp, Object node, Object ret) throws IOException, ServletException {
126        for (HttpResponseRenderer r : req.stapler.getWebApp().getResponseRenderers())
127            if (r.generateResponse(req,rsp,node,ret))
128                return true;
129        return false;
130    }

132    /**
133     * Use the given arguments as the first N arguments,
134     * then figure out the rest of the arguments by looking at parameter annotations,
135     * then finally call {@link #invoke}.
136     */
137    Object bindAndInvoke(Object o, StaplerRequest req, StaplerResponse rsp, Object... headArgs) throws IllegalAccessException, InvocationTargetException, ServletException {
138        Class[] types = getParameterTypes();
139        Annotation[][] annotations = getParameterAnnotations();
140        String[] parameterNames = getParameterNames();
141
142        Object[] arguments = new Object[types.length];
143
144        // fill in the first N arguments
145        System.arraycopy(headArgs,0,arguments,0,headArgs.length);
146
147        try {
148            // find the rest of the arguments. either known types, or with annotations
149            for( int i=headArgs.length; i<types.length; i++ ) {
150                Class t = types[i];
151                if(t==StaplerRequest.class || t==HttpServletRequest.class) {
152                    arguments[i] = req;
153                    continue;
154                }
155                if(t==StaplerResponse.class || t==HttpServletResponse.class) {
156                    arguments[i] = rsp;
157                    continue;
158                }
159
160                // if the databinding method is provided, call that
161                Function binder = PARSE_METHODS.getUnchecked(t);
162                if (binder!=RETURN_NULL) {
163                    arguments[i] = binder.bindAndInvoke(null,req,rsp);
164                    continue;
165                }
166                
167                arguments[i] = AnnotationHandler.handle(req,annotations[i],
168                    i<parameterNames.length ? parameterNames[i] : null,
169                    t);
170            }
171        } catch (IllegalArgumentException e) {
172            throw new IllegalArgumentException("Failed to invoke "+getDisplayName(),e);
173        }
174
175        return invoke(req, rsp, o,arguments);
176    }

178    /**
179     * Computing map that discovers the static 'fromStapler' method from a class.
180     * The discovered method will be returned as a Function so that the invocation can do parameter injections.
181     */
182    private static final LoadingCache<Class,Function> PARSE_METHODS;
183    private static final Function RETURN_NULL;
184
185    static {
186        try {
187            RETURN_NULL = new StaticFunction(Function.class.getMethod("returnNull"));
188        } catch (NoSuchMethodException e) {
189            throw new AssertionError(e);    // impossible
190        }
191
192        PARSE_METHODS = CacheBuilder.newBuilder().weakKeys().build(new CacheLoader<Class,Function>() {
193            public Function load(Class from) {
194                // MethdFunction for invoking a static method as a static method
195                FunctionList methods = new ClassDescriptor(from).methods.name("fromStapler");
196                switch (methods.size()) {
197                case 0: return RETURN_NULL;
198                default:
199                    throw new IllegalArgumentException("Too many 'fromStapler' methods on "+from);
200                case 1:
201                    Method m = ((MethodFunction)methods.get(0)).m;
202                    return new MethodFunction(m) {
203                        @Override
204                        public Class[] getParameterTypes() {
205                            return m.getParameterTypes();
206                        }
207
208                        @Override
209                        public Type[] getGenericParameterTypes() {
210                            return m.getGenericParameterTypes();
211                        }
212
213                        @Override
214                        public Annotation[][] getParameterAnnotations() {
215                            return m.getParameterAnnotations();
216                        }
217
218                        @Override
219                        public Object invoke(StaplerRequest req, StaplerResponse rsp, Object o, Object... args) throws IllegalAccessException, InvocationTargetException {
220                            return m.invoke(null,args);
221                        }
222                    };
223                }
224            }
225        });
226    }

238    final Function wrapByInterceptors(AnnotatedElement m) {
239        try {
240            Function f = this;
241            for (Annotation a : m.getAnnotations()) {
242                final InterceptorAnnotation ia = a.annotationType().getAnnotation(InterceptorAnnotation.class);
243                if (ia!=null) {
244                    try {
245                        Interceptor i = ia.value().newInstance();
246                        switch (ia.stage()) {
247                        case SELECTION:
248                            f = new SelectionInterceptedFunction(f,i);
249                            break;
250                        case PREINVOKE:
251                            f = new PreInvokeInterceptedFunction(f,i);
252                            break;
253                        }
254                    } catch (InstantiationException e) {
255                        throw (Error)new InstantiationError("Failed to instantiate interceptor for "+f.getDisplayName()).initCause(e);
256                    } catch (IllegalAccessException e) {
257                        throw (Error)new IllegalAccessError("Failed to instantiate interceptor for "+f.getDisplayName()).initCause(e);
258                    }
259                }
260            }
261
262            return f;
263        } catch (LinkageError e) {
264            // running in JDK 1.4
265            return this;
266        }
267    }

273    private abstract static class MethodFunction extends Function {
274        protected final Method m;
275        private volatile String[] names;
276
277        public MethodFunction(Method m) {
278            this.m = m;
279        }
280
281        public final String getName() {
282            return m.getName();
283        }
284
285        public final String getDisplayName() {
286            return m.toGenericString();
287        }
288
289        @Override
290        public String getQualifiedName() {
291            return m.getDeclaringClass().getName()+'.'+getName();
292        }
293
294        public <A extends Annotation> A getAnnotation(Class<A> annotation) {
295            return m.getAnnotation(annotation);
296        }
297
298        public Annotation[] getAnnotations() {
299            return m.getAnnotations();
300        }
301
302        public final String[] getParameterNames() {
303            if(names==null)
304                names = ClassDescriptor.loadParameterNames(m);
305            return names;
306        }
307
308        @Override
309        public Class getReturnType() {
310            return m.getReturnType();
311        }
312    }

339    /**
340     * Instance method that overrides other instance methods where we join annotations.
341     */
342    static final class OverridingInstanceFunction extends InstanceFunction {
343        // the last one takes precedence
344        private final List<Method> methods;
345
346        public OverridingInstanceFunction(List<Method> m) {
347            super(m.get(0));
348            methods = m;
349        }
350
351        @Override
352        public <A extends Annotation> A getAnnotation(Class<A> annotation) {
353            for (Method m : methods) {
354                A a = m.getAnnotation(annotation);
355                if (a!=null)    return a;
356            }
357            return null;
358        }
359
360        @Override
361        public Annotation[] getAnnotations() {
362            Annotation[] x = new Annotation[0];
363            for (Method m : methods) {
364                x = union(x, m.getAnnotations());
365            }
366            return x;
367        }
368
369        public Annotation[][] getParameterAnnotations() {
370            Annotation[][] all = null;
371            for (Method m : methods) {
372                Annotation[][] next = m.getParameterAnnotations();
373                if (all==null) {
374                    all = next;
375                } else {
376                    for (int i = 0; i < next.length; i++) {
377                        all[i] = union(all[i], next[i]);
378                    }
379                }
380            }
381            return all;
382        }
383    }

385    /**
386     * Static methods on the wrapper type.
387     */
388    static final class StaticFunction extends MethodFunction {
389        public StaticFunction(Method m) {
390            super(m);
391        }
392
393        public Class[] getParameterTypes() {
394            Class[] p = m.getParameterTypes();
395            Class[] r = new Class[p.length-1];
396            System.arraycopy(p,1,r,0,r.length);
397            return r;
398        }
399
400        @Override
401        public Type[] getGenericParameterTypes() {
402            Type[] p = m.getGenericParameterTypes();
403            Type[] r = new Type[p.length-1];
404            System.arraycopy(p,1,r,0,r.length);
405            return r;
406        }
407
408        public Annotation[][] getParameterAnnotations() {
409            Annotation[][] a = m.getParameterAnnotations();
410            Annotation[][] r = new Annotation[a.length-1][];
411            System.arraycopy(a,1,r,0,r.length);
412            return r;
413        }
414
415        public Object invoke(StaplerRequest req, StaplerResponse rsp, Object o, Object... args) throws IllegalAccessException, InvocationTargetException {
416            Object[] r = new Object[args.length+1];
417            r[0] = o;
418            System.arraycopy(args,0,r,1,args.length);
419            return m.invoke(null,r);
420        }
421    }
