024package org.kohsuke.stapler;
025
026import java.net.URL;
027import java.util.Collection;
028
029/**
030 * Partial default implementation of tear-off class, for convenience of derived classes.
031 *
032 * @param <CLT>
033 *      ClassLoader tear-off.
034 * @author Kohsuke Kawaguchi
035 */
036public abstract class AbstractTearOff<CLT,S,E extends Exception> extends CachingScriptLoader<S,E> {
037    protected final MetaClass owner;
038    protected final CLT classLoader;
039
040    protected AbstractTearOff(MetaClass owner, Class<CLT> cltClass) {
041        this.owner = owner;
042        if(owner.classLoader!=null)
043            classLoader = owner.classLoader.loadTearOff(cltClass);
044        else
045            classLoader = null;
046    }
047
048    protected final WebApp getWebApp() {
049        return owner.webApp;
050    }
051
052    /**
053     * The file extension of this kind of scripts, such as ".jelly"
054     */
055    protected abstract String getDefaultScriptExtension();
056
057    /**
058     * Checks if the file name is allowed as a script of this type.
059     *
060     * This is necessary to have multiple facets co-exist peacefully
061     * without them trying to load each other's scripts.
062     */
063    protected boolean hasAllowedExtension(String name) {
064        return name.endsWith(getDefaultScriptExtension());
065    }
066
067    /**
068     * Loads the script just from the target class without considering inherited scripts
069     * from its base types.
070     */
071    public S resolveScript(String name) throws E {
072        if (name.lastIndexOf('.')<=name.lastIndexOf('/'))   // no file extension provided
073            name += getDefaultScriptExtension();
074        if (!hasAllowedExtension(name))
075            // for multiple Facets to co-exist peacefully, we need to be able to determine
076            // which Facet is responsible for a given view just from the file name
077            return null;
078
079        URL res = getResource(name);
080        if(res==null) {
081            // look for 'defaults' file
082            int dot = name.lastIndexOf('.');
083            // foo/bar.groovy -> foo/bar.default.groovy
084            // but don't do foo.bar/test -> foo.default.bar/test
085            // as of 2010/9, this behaviour is considered deprecated, but left here for backward compatibility.
086            // we need a better way to refer to the resource of the same name in the base type.
087            if(name.lastIndexOf('/')<dot)
088                res = getResource(name.substring(0, dot) + ".default" + name.substring(dot));
089        }
090        if(res!=null)
091            return parseScript(res);
092
093        return null;
094    }
095
096    protected final S loadScript(String name) throws E {
097        S s = resolveScript(name);
098        if (s!=null)    return s;
099
100        // not found on this class, delegate to the parent
101        if(owner.baseClass!=null)
102            return ((AbstractTearOff<CLT,S,E>)owner.baseClass.loadTearOff(getClass())).findScript(name);
103
104        return null;
105    }
106
107    /**
108     * Compiles a script into the compiled form.
109     */
110    protected abstract S parseScript(URL res) throws E;
111
112    protected URL getResource(String name) {
113        return owner.klass.getResource(name);
114    }
115}
