024package org.kohsuke.stapler;
025
026import java.lang.annotation.Annotation;
027import java.util.ArrayList;
028import java.util.Arrays;
029import java.util.HashMap;
030import java.util.List;
031import java.util.Map;
032
033/**
034 * @author Kohsuke Kawaguchi
035 */
036public class ReflectionUtils {
037    /**
038     * Given the primitive type, returns the VM default value for that type in a boxed form.
039     * For reference types, return null.
040     */
041    public static Object getVmDefaultValueFor(Class<?> type) {
042        return defaultPrimitiveValue.get(type);
043    }
044
045    private static final Map<Class,Object> defaultPrimitiveValue = new HashMap<Class, Object>();
046    static {
047        defaultPrimitiveValue.put(boolean.class,false);
048        defaultPrimitiveValue.put(int.class,0);
049        defaultPrimitiveValue.put(long.class,0L);
050    }
051
052    /**
053     * Merge  two sets of annotations. If both contains the same annotation, the definition
054     * in 'b' overrides the definition in 'a' and shows up in the result
055     */
056    public static Annotation[] union(Annotation[] a, Annotation[] b) {
057        // fast path
058        if (a.length==0)    return b;
059        if (b.length==0)    return a;
060
061        // slow path
062        List<Annotation> combined = new ArrayList<Annotation>(a.length+b.length);
063        combined.addAll(Arrays.asList(a));
064
065        OUTER:
066        for (Annotation x : b) {
067            for (int i=0; i<a.length; i++) {
068                if (x.annotationType()==combined.get(i).annotationType()) {
069                    combined.set(i,x);  // override
070                    continue OUTER;
071                }
072            }
073            // not overlapping. append
074            combined.add(x);
075        }
076
077        return combined.toArray(new Annotation[combined.size()]);
078    }
079
080}
