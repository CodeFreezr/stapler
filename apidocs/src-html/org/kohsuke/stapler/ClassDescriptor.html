050/**
051 * Reflection information of a {@link Class}.
052 *
053 * @author Kohsuke Kawaguchi
054 */
055public final class ClassDescriptor {
056    public final Class clazz;
057    public final FunctionList methods;
058    public final Field[] fields;
059
060    /**
061     * @param clazz
062     *      The class to build a descriptor around.
063     * @param wrappers
064     *      Optional wrapper duck-typing classes.
065     *      Static methods on this class that has the first parameter
066     *      as 'clazz' will be handled as if it's instance methods on
067     *      'clazz'. Useful for adding view/controller methods on
068     *      model classes.
069     */
070    public ClassDescriptor(Class clazz, Class... wrappers) {
071        this.clazz = clazz;
072        this.fields = clazz.getFields();
073
074        // instance methods
075        List<Function> functions = new ArrayList<Function>();
076        for (Method m : clazz.getMethods()) {
077            functions.add(new Function.InstanceFunction(m).wrapByInterceptors(m));
078        }
079        if(wrappers!=null) {
080            for (Class w : wrappers) {
081                for (Method m : w.getMethods()) {
082                    if(!Modifier.isStatic(m.getModifiers()))
083                        continue;
084                    Class<?>[] p = m.getParameterTypes();
085                    if(p.length==0)
086                        continue;
087                    if(p[0].isAssignableFrom(clazz))
088                        continue;
089                    functions.add(new Function.StaticFunction(m).wrapByInterceptors(m));
090                }
091            }
092        }
093        this.methods = new FunctionList(functions);
094    }
095
096    /**
097     * Loads the list of parameter names of the given method, by using a stapler-specific way of getting it.
098     *
099     * <p>
100     * This is not the best place to expose this, but for now this would do.
101     */
102    public static String[] loadParameterNames(Method m) {
103        CapturedParameterNames cpn = m.getAnnotation(CapturedParameterNames.class);
104        if(cpn!=null)   return cpn.value();
105
106        // debug information, if present, is more trustworthy
107        try {
108            String[] n = ASM.loadParametersFromAsm(m);
109            if (n!=null)    return n;
110        } catch (LinkageError e) {
111            LOGGER.log(FINE, "Incompatible ASM", e);
112        } catch (IOException e) {
113            LOGGER.log(WARNING, "Failed to load a class file", e);
114        }
115
116        // otherwise check the .stapler file
117        Class<?> c = m.getDeclaringClass();
118        URL url = c.getClassLoader().getResource(
119                    c.getName().replace('.', '/').replace('$','/') + '/' + m.getName() + ".stapler");
120        if(url!=null) {
121            try {
122                return IOUtils.toString(url.openStream()).split(",");
123            } catch (IOException e) {
124                LOGGER.log(WARNING, "Failed to load "+url,e);
125                return EMPTY_ARRAY;
126            }
127        }
128
129        // couldn't find it
130        return EMPTY_ARRAY;
131    }

133    /**
134     * Loads the list of parameter names of the given method, by using a stapler-specific way of getting it.
135     *
136     * <p>
137     * This is not the best place to expose this, but for now this would do.
138     */
139    public static String[] loadParameterNames(Constructor<?> m) {
140        CapturedParameterNames cpn = m.getAnnotation(CapturedParameterNames.class);
141        if(cpn!=null)   return cpn.value();
142
143        // debug information, if present, is more trustworthy
144        try {
145            String[] n = ASM.loadParametersFromAsm(m);
146            if (n!=null)    return n;
147        } catch (LinkageError e) {
148            LOGGER.log(FINE, "Incompatible ASM", e);
149        } catch (IOException e) {
150            LOGGER.log(WARNING, "Failed to load a class file", e);
151        }
152
153        // couldn't find it
154        return EMPTY_ARRAY;
155    }

157    /**
158     * Determines the constructor parameter names.
159     *
160     * <p>
161     * First, try to load names from the debug information. Otherwise
162     * if there's the .stapler file, load it as a property file and determines the constructor parameter names.
163     * Otherwise, look for {@link CapturedParameterNames} annotation.
164     */
165    public String[] loadConstructorParamNames() {
166        Constructor<?>[] ctrs = clazz.getConstructors();
167        // which constructor was data bound?
168        Constructor<?> dbc = null;
169        for (Constructor<?> c : ctrs) {
170            if (c.getAnnotation(DataBoundConstructor.class) != null) {
171                dbc = c;
172                break;
173            }
174        }
175
176        if (dbc==null)
177            throw new NoStaplerConstructorException("There's no @DataBoundConstructor on any constructor of " + clazz);
178
179        String[] names = ClassDescriptor.loadParameterNames(dbc);
180        if (names.length==dbc.getParameterTypes().length)
181            return names;
182
183        String resourceName = clazz.getName().replace('.', '/').replace('$','/') + ".stapler";
184        ClassLoader cl = clazz.getClassLoader();
185        if(cl==null)
186            throw new NoStaplerConstructorException(clazz+" is a built-in type");
187        InputStream s = cl.getResourceAsStream(resourceName);
188        if (s != null) {// load the property file and figure out parameter names
189            try {
190                Properties p = new Properties();
191                p.load(s);
192                s.close();
193
194                String v = p.getProperty("constructor");
195                if (v.length() == 0) return new String[0];
196                return v.split(",");
197            } catch (IOException e) {
198                throw new IllegalArgumentException("Unable to load " + resourceName, e);
199            }
200        }
201
202        // no debug info and no stapler file
203        throw new NoStaplerConstructorException(
204                "Unable to find " + resourceName + ". " +
205                        "Run 'mvn clean compile' once to run the annotation processor.");
206    }

209    /**
210     * Isolate the ASM dependency to its own class, as otherwise this seems to cause linkage error on the whole {@link ClassDescriptor}.
211     */
212    private static class ASM {
213        /**
214         * Try to load parameter names from the debug info by using ASM.
215         */
216        private static String[] loadParametersFromAsm(final Method m) throws IOException {
217            final String[] paramNames = new String[m.getParameterTypes().length];
218            if (paramNames.length==0) return paramNames;
219            Class<?> c = m.getDeclaringClass();
220            URL clazz = c.getClassLoader().getResource(c.getName().replace('.', '/') + ".class");
221            if (clazz==null)    return null;
222
223            final TreeMap<Integer,String> localVars = new TreeMap<Integer,String>();
224            ClassReader r = new ClassReader(clazz.openStream());
225            r.accept(new ClassVisitor(ASM5) {
226                final String md = Type.getMethodDescriptor(m);
227                // First localVariable is "this" for non-static method
228                final int limit = (m.getModifiers() & Modifier.STATIC) != 0 ? 0 : 1;
229                @Override public MethodVisitor visitMethod(int access, String methodName, String desc, String signature, String[] exceptions) {
230                    if (methodName.equals(m.getName()) && desc.equals(md))
231                        return new MethodVisitor(ASM5) {
232                            @Override public void visitLocalVariable(String name, String desc, String signature, Label start, Label end, int index) {
233                                if (index >= limit)
234                                    localVars.put(index, name);
235                            }
236                        };
237                    else
238                        return null; // ignore this method
239                }
240            }, 0);
241
242            // Indexes may not be sequential, but first set of local variables are method params
243            int i = 0;
244            for (String s : localVars.values()) {
245                paramNames[i] = s;
246                if (++i == paramNames.length) return paramNames;
247            }
248            return null; // Not enough data found to fill array
249        }
250
251        /**
252         * Try to load parameter names from the debug info by using ASM.
253         */
254        private static String[] loadParametersFromAsm(final Constructor m) throws IOException {
255            final String[] paramNames = new String[m.getParameterTypes().length];
256            if (paramNames.length==0) return paramNames;
257            Class<?> c = m.getDeclaringClass();
258            URL clazz = c.getClassLoader().getResource(c.getName().replace('.', '/') + ".class");
259            if (clazz==null)    return null;
260
261            final TreeMap<Integer,String> localVars = new TreeMap<Integer,String>();
262            InputStream is = clazz.openStream();
263            try {
264                ClassReader r = new ClassReader(is);
265                r.accept(new ClassVisitor(ASM5) {
266                    final String md = getConstructorDescriptor(m);
267                    public MethodVisitor visitMethod(int access, String methodName, String desc, String signature, String[] exceptions) {
268                        if (methodName.equals("<init>") && desc.equals(md))
269                            return new MethodVisitor(ASM5) {
270                                @Override public void visitLocalVariable(String name, String desc, String signature, Label start, Label end, int index) {
271                                    if (index>0)   // 0 is 'this'
272                                        localVars.put(index, name);
273                                }
274                            };
275                        else
276                            return null; // ignore this method
277                    }
278                }, 0);
279            } finally {
280                is.close();
281            }
282
283            // Indexes may not be sequential, but first set of local variables are method params
284            int i = 0;
285            for (String s : localVars.values()) {
286                paramNames[i] = s;
287                if (++i == paramNames.length) return paramNames;
288            }
289            return null; // Not enough data found to fill array
290        }
291
292        private static String getConstructorDescriptor(Constructor c) {
293            StringBuilder buf = new StringBuilder("(");
294            for (Class p : c.getParameterTypes())
295                buf.append(Type.getDescriptor(p));
296            return buf.append(")V").toString();
297        }
298    }
