001package org.kohsuke.stapler.lang;
002
003import org.kohsuke.stapler.Facet;
004
005/**
006 * Objects can implement this interface to designate its own {@link Klass}.
007 *
008 * This allows specific classes or instances to take over the routing rules without
009 * going through the trouble of creating a new {@link Facet}
010 *
011 * @author Kohsuke Kawaguchi
012 */
013public interface KInstance<C> {
014    /**
015     * @return
016     *      null if there's no designated {@link Klass} for this instance, in which
017     *      case Stapler treats this instance normally as if it didn't implement
018     *      {@link KInstance} to begin with (for example, by calling {@link Object#getClass()}
019     */
020    Klass<C> getKlass();
021}
