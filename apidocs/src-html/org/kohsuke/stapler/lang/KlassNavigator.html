001package org.kohsuke.stapler.lang;
002
003import org.kohsuke.stapler.ClassDescriptor;
004import org.kohsuke.stapler.Function;
005import org.kohsuke.stapler.MetaClassLoader;
006
007import java.lang.reflect.Array;
008import java.lang.reflect.Field;
009import java.lang.reflect.Method;
010import java.net.URL;
011import java.util.AbstractList;
012import java.util.ArrayList;
013import java.util.List;
014import java.util.Map;
015
016/**
017 * Strategy pattern to provide navigation across class-like objects in other languages of JVM.
018 *
019 * <p>
020 * Implementations should be stateless and typically a singleton.
021 *
022 * @param <C>
023 *     Variable that represents the type of {@code Class} like object in this language.
024 *
025 * @author Kohsuke Kawaguchi
026 */
027public abstract class KlassNavigator<C> {
028    /**
029     * Loads the resources associated with this class.
030     *
031     * <p>
032     * In stapler, the convention is that the "associated" resources live in the directory named after
033     * the fully qualified class name (as opposed to the behavior of {@link Class#getResource(String)},
034     * that looks for resources in the same package as the class.)
035     *
036     * <p>
037     * But other languages can choose their own conventions if it makes more sense to do so.
038     * For example, stapler-jruby uses camelized class name.
039     *
040     * <p>
041     * Implementation must consult {@link MetaClassLoader#debugLoader} if it's available. Implementation
042     * must not look for resources in the base type. That operation is performed by the caller when
043     * needed.
044     *
045     * @return
046     *      non-null if the resource is found. Otherwise null.
047     */
048    public abstract URL getResource(C clazz, String resourceName);
049
050    /**
051     * Lists up all the ancestor classes, from specific to general, without any duplicate.
052     *
053     * This is used to look up a resource.
054     */
055    public abstract Iterable<Klass<?>> getAncestors(C clazz);
056
057    /**
058     * Gets the super class.
059     *
060     * @return
061     *      Can be null.
062     */
063    public abstract Klass<?> getSuperClass(C clazz);
064
065    /**
066     * For backward compatibility, map the given class to the closest Java equivalent.
067     * In the worst case, this is Object.class
068     */
069    public abstract Class toJavaClass(C clazz);
070
071    /**
072     * List methods of this class, regardless of access modifier.
073     *
074     * This list excludes methods from super classes.
075     * @since 1.220
076     */
077    public abstract List<MethodRef> getDeclaredMethods(C clazz);
078
079    /**
080     * List fields of this class.
081     *
082     * This list excludes fields from super classes.
083     */
084    public abstract List<FieldRef> getDeclaredFields(C clazz);
085
086    /**
087     * Reports all the methods that can be used for routing requests on this class.
088     */
089    public abstract List<Function> getFunctions(C clazz);
090
091    /**
092     * If the given type is an array that supports index retrieval.
093     * @see #getArrayElement(Object, int)
094     */
095    public boolean isArray(C clazz) {
096        Class j = toJavaClass(clazz);
097        return j.isArray() || List.class.isAssignableFrom(j);
098    }
099
100    /**
101     * Given an instance for which the type reported {@code isArray()==true}, obtains the element
102     * of the specified index.
103     * @see #isArray(Object)
104     */
105    public Object getArrayElement(Object o, int index) throws IndexOutOfBoundsException {
106        if (o instanceof List)
107            return ((List)o).get(index);
108        return Array.get(o,index);
109    }
110
111    /**
112     * If the given type is a map/associative array type that supports lookup by a string key
113     */
114    public boolean isMap(C clazz) {
115        return Map.class.isAssignableFrom(toJavaClass(clazz));
116    }
117
118    /**
119     * Given an instance for which the type reported {@code isMap()==true}, obtains the element
120     * of the specified index.
121     */
122    public Object getMapElement(Object o, String key) {
123        return ((Map)o).get(key);
124    }
125
126    public static final KlassNavigator<Class> JAVA = new KlassNavigator<Class>() {
127        @Override
128        public URL getResource(Class clazz, String resourceName) {
129            ClassLoader cl = clazz.getClassLoader();
130            if (cl==null)   return null;
131
132            String fullName;
133            if (resourceName.startsWith("/"))
134                fullName = resourceName.substring(1);
135            else
136                fullName = clazz.getName().replace('.','/').replace('$','/')+'/'+resourceName;
137
138            if (MetaClassLoader.debugLoader!=null) {
139                URL res = MetaClassLoader.debugLoader.loader.getResource(fullName);
140                if (res!=null)  return res;
141            }
142            return cl.getResource(fullName);
143        }
144
145        @Override
146        public Klass<Class> getSuperClass(Class clazz) {
147            return Klass.java(clazz.getSuperclass());
148        }
149
150        @Override
151        public Iterable<Klass<?>> getAncestors(Class clazz) {
152            // TODO: shall we support interfaces?
153            List<Klass<?>> r = new ArrayList<Klass<?>>();
154            for (; clazz!=null; clazz=clazz.getSuperclass()) {
155                r.add(Klass.java(clazz));
156            }
157            return r;
158        }
159
160        @Override
161        public Class toJavaClass(Class clazz) {
162            return clazz;
163        }
164
165        @Override
166        public List<MethodRef> getDeclaredMethods(Class clazz) {
167            final Method[] methods = clazz.getDeclaredMethods();
168            return new AbstractList<MethodRef>() {
169                @Override
170                public MethodRef get(int index) {
171                    return MethodRef.wrap(methods[index]);
172                }
173
174                @Override
175                public int size() {
176                    return methods.length;
177                }
178            };
179        }
180
181        @Override
182        public List<FieldRef> getDeclaredFields(Class clazz) {
183            final Field[] fields = clazz.getDeclaredFields();
184            return new AbstractList<FieldRef>() {
185                @Override
186                public FieldRef get(int index) {
187                    return FieldRef.wrap(fields[index]);
188                }
189
190                @Override
191                public int size() {
192                    return fields.length;
193                }
194            };
195        }
196
197        @Override
198        public List<Function> getFunctions(Class clazz) {
199            // Historically ClassDescriptor used to own this non-trivial logic of computing
200            // valid functions for the class, so we'll keep it there.
201            return new ClassDescriptor(clazz).methods;
202        }
203    };
204}
