001package org.kohsuke.stapler.lang;
002
003import java.lang.annotation.Annotation;
004import java.lang.reflect.Field;
005
006/**
007 * Fields of {@link Klass}.
008 *
009 * @see Klass#getDeclaredFields()
010 * @author Kohsuke Kawaguchi
011 */
012public abstract class FieldRef extends AnnotatedRef {
013    /**
014     * Name of the method.
015     *
016     * @see Field#getName()
017     */
018    public abstract String getName();
019
020    /**
021     * Obtains the value of the field of the instance.
022     */
023    public abstract Object get(Object instance) throws IllegalAccessException;
024
025    /**
026     * Gets a fully qualified name of this field that includes the declaring type.
027     */
028    public abstract String getQualifiedName();
029
030    public static FieldRef wrap(final Field f) {
031        f.setAccessible(true);
032
033        return new FieldRef() {
034            @Override
035            public <T extends Annotation> T getAnnotation(Class<T> type) {
036                return f.getAnnotation(type);
037            }
038
039            @Override
040            public String getName() {
041                return f.getName();
042            }
043
044            @Override
045            public Object get(Object instance) throws IllegalAccessException {
046                return f.get(instance);
047            }
048
049            @Override
050            public String getQualifiedName() {
051                return f.getDeclaringClass().getName()+"."+getName();
052            }
053        };
054    }
055}
