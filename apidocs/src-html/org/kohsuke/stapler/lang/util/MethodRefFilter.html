001package org.kohsuke.stapler.lang.util;
002
003import org.kohsuke.stapler.lang.MethodRef;
004
005import java.lang.annotation.Annotation;
006import java.lang.reflect.InvocationTargetException;
007import java.lang.reflect.Method;
008
009/**
010 * {@link MethodRef} filter as a convenience class.
011 *
012 * @author Kohsuke Kawaguchi
013 */
014public abstract class MethodRefFilter extends MethodRef {
015    protected abstract MethodRef getBase();
016
017    @Override
018    public boolean isRoutable() {
019        return getBase().isRoutable();
020    }
021
022    @Override
023    public String getName() {
024        return getBase().getName();
025    }
026
027    @Override
028    public Object invoke(Object _this, Object... args) throws InvocationTargetException, IllegalAccessException {
029        return getBase().invoke(_this, args);
030    }
031
032    public static MethodRef wrap(Method m) {
033        return MethodRef.wrap(m);
034    }
035
036    @Override
037    public <T extends Annotation> T getAnnotation(Class<T> type) {
038        return getBase().getAnnotation(type);
039    }
040
041    @Override
042    public boolean hasAnnotation(Class<? extends Annotation> type) {
043        return getBase().hasAnnotation(type);
044    }
045}
