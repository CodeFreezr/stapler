001package org.kohsuke.stapler;
002
003import net.sf.json.JSONObject;
004
005/**
006 * For data-bound class (that has a constructor marked with {@link DataBoundConstructor}, the
007 * {@link #bindResolve(StaplerRequest, JSONObject)} allows an instance to replace the object
008 * bound from submitted JSON object.
009 *
010 * <p>
011 * This method is automatically invoked by Stapler during databinding method like
012 * {@link StaplerRequest#bindJSON(Class, JSONObject)}.
013 *
014 * <p>
015 * This method definition is inspired by Java serialization's {@code readResolve()} method.
016 *
017 * @author Kohsuke Kawaguchi
018 * @see <a href="https://issues.jenkins-ci.org/browse/JENKINS-20262">JENKINS-201262</a>
019 */
020public interface DataBoundResolvable {
021    /**
022     * Called after the object is instantiated to allow the object to nominate its replacement.
023     *
024     * @param request
025     *      Request object that's currently performing databinding. Passed in as a contextual
026     *      parameter.
027     * @param src
028     *      JSON object that originally constructed the 'this' instance on which this method
029     *      is being invoked.
030     * @return
031     *      Can be any value, including null. Typically, this method would have to return an
032     *      instance of a type compatible to the caller's expectation.
033     */
034    Object bindResolve(StaplerRequest request, JSONObject src);
035}
