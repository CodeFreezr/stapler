001package org.kohsuke.stapler;
002
003import net.sf.json.JSONObject;
004
005import javax.annotation.PostConstruct;
006import java.beans.Introspector;
007import java.lang.annotation.Documented;
008import java.lang.annotation.Retention;
009import java.lang.annotation.Target;
010
011import static java.lang.annotation.ElementType.*;
012import static java.lang.annotation.RetentionPolicy.*;
013
014/**
015 * Designates a setter method or a field used to databind JSON values into objects in methods like
016 * {@link StaplerRequest#bindJSON(Class, JSONObject)} and
017 * {@link StaplerRequest#bindParameters(Class, String)}.
018 *
019 * <p>
020 * Stapler will first invoke {@link DataBoundConstructor}-annotated constructor, and if there's any
021 * remaining properties in JSON, it'll try to find a matching {@link DataBoundSetter}-annotated setter
022 * method or a field.
023 *
024 * <p>
025 * The setter method is discovered through {@link Introspector}, so setter method name must match
026 * the property name (such as <tt>setFoo</tt> for the <tt>foo</tt> property), and it needs to be public.
027 *
028 * <p>
029 * The field is discovered through simple reflection, so its name must match the property name, but
030 * its access modifier can be anything.
031 *
032 * <p>
033 * To create a method to be called after all the setter injections are complete, annotate a method
034 * with {@link PostConstruct}.
035 *
036 * @author Kohsuke Kawaguchi
037 */
038@Retention(RUNTIME)
039@Target({METHOD,FIELD})
040@Documented
041public @interface DataBoundSetter {
042}
