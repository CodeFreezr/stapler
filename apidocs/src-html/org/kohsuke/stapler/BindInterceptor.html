001package org.kohsuke.stapler;
002
003import net.sf.json.JSONObject;
004
005import java.lang.reflect.Type;
006
007/**
008 * Intercepts (and receives callbacks) about the JSON â†’ object binding process.
009 *
010 * @author Kohsuke Kawaguchi
011 * @see StaplerRequest#setBindInterceptor(BindInterceptor)
012 * @see WebApp#bindInterceptors
013 */
014public class BindInterceptor {
015    /**
016     * Called for each object conversion, after the expected type is determined.
017     *
018     * @param targetType
019     *      Type that the converted object must be assignable to.
020     * @param targetTypeErasure
021     *      Erasure of the {@code targetType} parameter.
022     * @param jsonSource
023     *      JSON object to be mapped to Java object.
024     * @return
025     *      {@link #DEFAULT} to indicate that the default conversion process should proceed.
026     *      Any other values (including null) will override the process.
027     */
028    public Object onConvert(Type targetType, Class targetTypeErasure, Object jsonSource) {
029        return DEFAULT;
030    }
031
032    /**
033     * Called for each object conversion, after the actual subtype to instantiate is determined.
034     *
035     * @param actualType
036     *      The type to instantiate
037     * @param json
038     *      JSON object to be mapped to Java object.
039     * @return
040     *      {@link #DEFAULT} to indicate that the default conversion process should proceed.
041     *      Any other values (including null) will override the process.
042     */
043    public Object instantiate(Class actualType, JSONObject json) {
044        return DEFAULT;
045    }
046
047    /**
048     * Indicates that the conversion should proceed as it normally does,
049     * and that the listener isn't replacing the process.
050     */
051    public static final Object DEFAULT = new Object();
052
053    /**
054     * Default {@link BindInterceptor} that does nothing.
055     */
056    public static final BindInterceptor NOOP = new BindInterceptor();
057}
