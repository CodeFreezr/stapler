024package org.kohsuke.stapler;
025
026import org.kohsuke.stapler.interceptor.Interceptor;
027import org.kohsuke.stapler.interceptor.InterceptorAnnotation;
028
029import javax.servlet.ServletException;
030
031import static java.lang.annotation.ElementType.FIELD;
032import static java.lang.annotation.ElementType.METHOD;
033import java.lang.annotation.Retention;
034import static java.lang.annotation.RetentionPolicy.RUNTIME;
035import java.lang.annotation.Target;
036import java.lang.reflect.InvocationTargetException;
037
038/**
039 * Declares that methods are only available for requests that
040 * have the specified role(s).
041 *
042 * <p>
043 * This annotation should be placed on methods that need to be
044 * secured (iow protected from anonymous users.)
045 *
046 * @author Kohsuke Kawaguchi
047 */
048@Retention(RUNTIME)
049@Target({METHOD,FIELD})
050@InterceptorAnnotation(LimitedTo.Processor.class)
051public @interface LimitedTo {
052    /**
053     * The name of role.
054     * the method can be invoked only if the user belongs
055     * to this role.
056     */
057    String value();
058    
059    public static class Processor extends Interceptor {
060        private String role;
061        @Override
062        public void setTarget(Function target) {
063            role = target.getAnnotation(LimitedTo.class).value();
064            super.setTarget(target);
065        }
066
067        @Override
068        public Object invoke(StaplerRequest request, StaplerResponse response, Object instance, Object[] arguments)
069                throws IllegalAccessException, InvocationTargetException, ServletException {
070            if(request.isUserInRole(role))
071                return target.invoke(request, response, instance, arguments);
072            else
073                throw new IllegalAccessException("Needs to be in role "+role);
074        }
075    }
076}
