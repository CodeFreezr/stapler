046/**
047 * Defines additional parameters/operations made available by Stapler.
048 *
049 * @see Stapler#getCurrentRequest()
050 * @author Kohsuke Kawaguchi
051 */
052public interface StaplerRequest extends HttpServletRequest {
053    /**
054     * Gets the {@link Stapler} instance that this belongs to.
055     */
056    Stapler getStapler();
057
058    /**
059     * Short for {@code getStapler().getWebApp()}
060     */
061    WebApp getWebApp();
062
063    /**
064     * Returns the additional URL portion that wasn't used by the stapler,
065     * excluding the query string.
066     *
067     * <p>
068     * For example, if the requested URL is "foo/bar/zot/abc?def=ghi" and
069     * "foo/bar" portion matched <tt>bar.jsp</tt>, this method returns
070     * "/zot/abc".
071     *
072     * <p>
073     * If this method is invoked from getters or {@link StaplerProxy#getTarget()}
074     * during the object traversal, this method returns the path portion
075     * that is not yet processed.
076     *
077     * @return
078     *      can be empty string, but never null.
079     */
080    String getRestOfPath();
081
082    /**
083     * Returns the same thing as {@link #getRestOfPath()} but in the pre-decoded form,
084     * so all "%HH"s as present in the request URL is intact.
085     */
086    String getOriginalRestOfPath();
087
088    /**
089     * Returns the {@link ServletContext} object given to the stapler
090     * dispatcher servlet.
091     */
092    ServletContext getServletContext();
093
094    /**
095     * {@link #getRequestURI()} plus additional query string part, if it exists.
096     */
097    String getRequestURIWithQueryString();
098
099    /**
100     * {@link #getRequestURL()} plus additional query string part, if it exists.
101     */
102    StringBuffer getRequestURLWithQueryString();
103
104    /**
105     * Gets the {@link RequestDispatcher} that represents a specific view
106     * for the given object.
107     *
108     * This support both JSP and Jelly.
109     *
110     * @param viewName
111     *      If this name is relative name like "foo.jsp" or "bar/zot.jelly",
112     *      then the corresponding "side file" is searched by this name.
113     *      <p>
114     *      For Jelly, this also accepts absolute path name that starts
115     *      with '/', such as "/foo/bar/zot.jelly". In this case,
116     *      <tt>it.getClass().getClassLoader()</tt> is searched for this script. 
117     *
118     * @return null
119     *      if neither JSP nor Jelly is not found by the given name.
120     */
121    RequestDispatcher getView(Object it,String viewName) throws IOException;
122
123    /**
124     * Convenience method to call {@link #getView(Klass, String)} with {@link Class}.
125     */
126    RequestDispatcher getView(Class clazz,String viewName) throws IOException;
127
128    /**
129     * Gets the {@link RequestDispatcher} that represents a specific view
130     * for the given class.
131     *
132     * <p>
133     * Unlike {@link #getView(Object, String)}, calling this request dispatcher
134     * doesn't set the "it" variable, so
135     * {@code getView(it.getClass(),viewName)} and {@code getView(it,viewName)}
136     * aren't the same thing.
137     */
138    RequestDispatcher getView(Klass<?> clazz, String viewName) throws IOException;
139
140    /**
141     * Gets the part of the request URL from protocol up to the context path.
142     * So typically it's something like <tt>http://foobar:8080/something</tt>
143     */
144    String getRootPath();
145
146    /**
147     * Gets the referer header (like "http://foobar.com/zot") or null.
148     *
149     * This is just a convenience method.
150     */
151    String getReferer();
152
153    /**
154     * Returns a list of ancestor objects that lead to the "it" object.
155     * The returned list contains {@link Ancestor} objects sorted in the
156     * order from root to the "it" object.
157     *
158     * <p>
159     * For example, if the URL was "foo/bar/zot" and the "it" object
160     * was determined as <code>root.getFoo().getBar("zot")</code>,
161     * then this list will contain the following 3 objects in this order:
162     * <ol>
163     *  <li>the root object
164     *  <li>root.getFoo() object
165     *  <li>root.getFoo().getBar("zot") object (the "it" object)
166     * </ol>
167     * <p>
168     * 
169     *
170     * @return
171     *      list of {@link Ancestor}s. Can be empty, but always non-null.
172     */
173    List<Ancestor> getAncestors();
174
175    /**
176     * Finds the nearest ancestor that has the object of the given type, or null if not found.
177     */
178    Ancestor findAncestor(Class type);
179
180    /**
181     * Short for {@code findAncestor(type).getObject()}, with proper handling for null de-reference.
182     * This version is also type safe.
183     */
184    <T> T findAncestorObject(Class<T> type);
185
186    /**
187     * Finds the nearest ancestor whose {@link Ancestor#getObject()} matches the given object.
188     */
189    Ancestor findAncestor(Object o);
190
191    /**
192     * Short for {@code getParameter(name)!=null}
193     */
194    boolean hasParameter(String name);
195
196    /**
197     * Gets the {@link HttpServletRequest#getRequestURI() request URI}
198     * of the original request, so that you can access the value even from
199     * JSP.
200     */
201    String getOriginalRequestURI();
202
203    /**
204     * Checks "If-Modified-Since" header and returns false
205     * if the resource needs to be served.
206     *
207     * <p>
208     * This method can behave in three ways.
209     *
210     * <ol>
211     *  <li>If <tt>timestampOfResource</tt> is 0 or negative,
212     *      this method just returns false.
213     *
214     *  <li>If "If-Modified-Since" header is sent and if it's bigger than
215     *      <tt>timestampOfResource</tt>, then this method sets
216     *      {@link HttpServletResponse#SC_NOT_MODIFIED} as the response code
217     *      and returns true.
218     *
219     *  <li>Otherwise, "Last-Modified" header is added with <tt>timestampOfResource</tt> value,
220     *      and this method returns false.
221     * </ol>
222     *
223     * <p>
224     * This method sends out the "Expires" header to force browser
225     * to re-validate all the time.
226     *
227     * @param timestampOfResource
228     *      The time stamp of the resource.
229     * @param rsp
230     *      This object is updated accordingly to simplify processing.
231     *
232     * @return
233     *      false to indicate that the caller has to serve the actual resource.
234     *      true to indicate that the caller should just quit processing right there
235     *      (and send back {@link HttpServletResponse#SC_NOT_MODIFIED}.
236     */
237    boolean checkIfModified(long timestampOfResource, StaplerResponse rsp);
238
239    /**
240     * @see #checkIfModified(long, StaplerResponse)
241     */
242    boolean checkIfModified(Date timestampOfResource, StaplerResponse rsp);
243
244    /**
245     * @see #checkIfModified(long, StaplerResponse)
246     */
247    boolean checkIfModified(Calendar timestampOfResource, StaplerResponse rsp);
248
249    /**
250     * @param expiration
251     *      The number of milliseconds until the resource will "expire".
252     *      Until it expires the browser will be allowed to cache it
253     *      and serve it without checking back with the server.
254     *      After it expires, the client will send conditional GET to
255     *      check if the resource is actually modified or not.
256     *      If 0, it will immediately expire.
257     *
258     * @see #checkIfModified(long, StaplerResponse)
259     */
260    boolean checkIfModified(long timestampOfResource, StaplerResponse rsp, long expiration);
261
262    /**
263     * Binds form parameters to a bean by using introspection.
264     *
265     * For example, if there's a parameter called 'foo' that has value 'abc',
266     * then <tt>bean.setFoo('abc')</tt> will be invoked. This will be repeated
267     * for all parameters. Parameters that do not have corresponding setters will
268     * be simply ignored.
269     *
270     * <p>
271     * Values are converted into the right type. See {@link ConvertUtils#convert(String, Class)}.
272     *
273     * @see BeanUtils#setProperty(Object, String, Object)
274     *
275     * @param bean
276     *      The object which will be filled out.
277     */
278    void bindParameters( Object bean );
279
280    /**
281     * Binds form parameters to a bean by using introspection.
282     *
283     * This method works like {@link #bindParameters(Object)}, but it performs a
284     * pre-processing on property names. Namely, only property names that start
285     * with the given prefix will be used for binding, and only the portion of the
286     * property name after the prefix is used.
287     *
288     * So for example, if the prefix is "foo.", then property name "foo.bar" with value
289     * "zot" will invoke <tt>bean.setBar("zot")</tt>.
290     *
291     *
292     * @deprecated
293     *      Instead of using prefix to group object among form parameter names,
294     *      use structured form submission and {@link #bindJSON(Class, JSONObject)}.
295     */
296    void bindParameters( Object bean, String prefix );
297
298    /**
299     * Binds collection form parameters to beans by using introspection or
300     * constructor parameters injection.
301     *
302     * <p>
303     * This method works like {@link #bindParameters(Object,String)} and
304     * {@link #bindParameters(Class, String)}, but it assumes
305     * that form parameters have multiple-values, and use individual values to
306     * fill in multiple beans.
307     *
308     * <p>
309     * For example, if <tt>getParameterValues("foo")=={"abc","def"}</tt>
310     * and <tt>getParameterValues("bar")=={"5","3"}</tt>, then this method will
311     * return two objects (the first with "abc" and "5", the second with
312     * "def" and "3".)
313     *
314     * @param type
315     *      Type of the bean to be created. This class must have the default no-arg
316     *      constructor.
317     *
318     * @param prefix
319     *      See {@link #bindParameters(Object, String)} for details.
320     *
321     * @return
322     *      Can be empty but never null.
323     *
324     *
325     * @deprecated
326     *      Instead of using prefix to group object among form parameter names,
327     *      use structured form submission and {@link #bindJSON(Class, JSONObject)}.
328     */
329    <T>
330    List<T> bindParametersToList( Class<T> type, String prefix );
331
332    /**
333     * Instantiates a new object by injecting constructor parameters from the form parameters.
334     *
335     * <p>
336     * The given class must have a constructor annotated with '@stapler-constructor',
337     * and must be processed by the maven-stapler-plugin, so that the parameter names
338     * of the constructor is available at runtime.
339     *
340     * <p>
341     * The prefix is used to control the form parameter name. For example,
342     * if the prefix is "foo." and if the constructor is define as
343     * <code>Foo(String a, String b)</code>, then the constructor will be invoked
344     * as <code>new Foo(getParameter("foo.a"),getParameter("foo.b"))</code>.
345     *
346     * @deprecated
347     *      Instead of using prefix to group object among form parameter names,
348     *      use structured form submission and {@link #bindJSON(Class, JSONObject)}.
349     */
350    <T>
351    T bindParameters( Class<T> type, String prefix );
352
353    /**
354     * Works like {@link #bindParameters(Class, String)} but uses n-th value
355     * of all the parameters.
356     *
357     * <p>
358     * This is useful for creating multiple instances from repeated form fields.
359     *
360     *
361     * @deprecated
362     *      Instead of using prefix to group object among form parameter names,
363     *      use structured form submission and {@link #bindJSON(Class, JSONObject)}.
364     */
365    <T>
366    T bindParameters( Class<T> type, String prefix, int index );
367
368    /**
369     * Data-bind from a {@link JSONObject} to the given target type,
370     * by using introspection or constructor parameters injection.
371     *
372     * <p>
373     * For example, if you have a constructor that looks like the following:
374     *
375     * <pre>
376     * class Foo {
377     *   &#64;{@link DataBoundConstructor}
378     *   public Foo(Integer x, String y, boolean z, Bar bar) { ... }
379     * }
380     *
381     * class Bar {
382     *   &#64;{@link DataBoundConstructor}
383     *   public Bar(int x, int y) {}
384     * }
385     * </pre>
386     *
387     * ... and if JSONObject looks like
388     *
389     * <pre>{ y:"text", z:true, bar:{x:1,y:2}}</pre>
390     *
391     * then, this method returns
392     *
393     * <pre>new Foo(null,"text",true,new Bar(1,2))</pre>
394     *
395     * <h3>Sub-typing</h3>
396     * <p>
397     * In the above example, a new instance of <tt>Bar</tt> was created,
398     * but you can also create a subtype of Bar by having the 'stapler-class' property in
399     * JSON like this:
400     *
401     * <pre>
402     * class BarEx extends Bar {
403     *   &#64;{@link DataBoundConstructor}
404     *   public BarEx(int a, int b, int c) {}
405     * }
406     *
407     * { y:"text", z:true, bar: { stapler-class:"p.k.g.BarEx", a:1, b:2, c:3 } }
408     * </pre>
409     *
410     * <p>
411     * The type that shows up in the constructor (<tt>Bar</tt> in this case)
412     * can be an interface or an abstract class.
413     */
414    <T>
415    T bindJSON(Class<T> type, JSONObject src);
416
417    /**
418     * Data-bind from one of the JSON object types ({@link JSONObject}, {@link JSONArray},
419     * {@link String}, {@link Integer}, and so on) to the expected type given as an argument.
420     *
421     * @param genericType
422     *      The generic type of the 'erasure' parameter.
423     * @param erasure
424     *      The expected type to convert the JSON argument to.
425     * @param json
426     *      One of the JSON value type.
427     */
428    <T> T bindJSON(Type genericType, Class<T> erasure, Object json);
429
430    /**
431     * Data-binds from {@link JSONObject} to the given object.
432     *
433     * <p>
434     * This method is bit like {@link #bindJSON(Class, JSONObject)}, except that this method
435     * populates an existing object, instead of creating a new instance.
436     *
437     * <p>
438     * This method is also bit like {@link #bindParameters(Object, String)}, in that it
439     * populates an existing object from a form submission, except that this method
440     * obtains data from {@link JSONObject} thus more structured, whereas {@link #bindParameters(Object, String)}
441     * uses the map structure of the form submission. 
442     */
443    void bindJSON( Object bean, JSONObject src );
444
445    /**
446     * Data-bind from either {@link JSONObject} or {@link JSONArray} to a list,
447     * by using {@link #bindJSON(Class, JSONObject)} as the lower-level mechanism.
448     *
449     * <p>
450     * If the source is {@link JSONObject}, the returned list will contain
451     * a single item. If it is {@link JSONArray}, each item will be bound.
452     * If it is null, then the list will be empty.
453     */
454    <T>
455    List<T> bindJSONToList(Class<T> type, Object src);
