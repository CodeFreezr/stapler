047/**
048 * Defines additional parameters/operations made available by Stapler.
049 *
050 * @see Stapler#getCurrentRequest()
051 * @author Kohsuke Kawaguchi
052 */
053public interface StaplerRequest extends HttpServletRequest {
054    /**
055     * Gets the {@link Stapler} instance that this belongs to.
056     */
057    Stapler getStapler();
058
059    /**
060     * Short for {@code getStapler().getWebApp()}
061     */
062    WebApp getWebApp();
063
064    /**
065     * Returns the additional URL portion that wasn't used by the stapler,
066     * excluding the query string.
067     *
068     * <p>
069     * For example, if the requested URL is "foo/bar/zot/abc?def=ghi" and
070     * "foo/bar" portion matched <tt>bar.jsp</tt>, this method returns
071     * "/zot/abc".
072     *
073     * <p>
074     * If this method is invoked from getters or {@link StaplerProxy#getTarget()}
075     * during the object traversal, this method returns the path portion
076     * that is not yet processed.
077     *
078     * @return
079     *      can be empty string, but never null.
080     */
081    String getRestOfPath();
082
083    /**
084     * Returns the same thing as {@link #getRestOfPath()} but in the pre-decoded form,
085     * so all "%HH"s as present in the request URL is intact.
086     */
087    String getOriginalRestOfPath();
088
089    /**
090     * Returns the {@link ServletContext} object given to the stapler
091     * dispatcher servlet.
092     */
093    ServletContext getServletContext();

095    /**
096     * {@link #getRequestURI()} plus additional query string part, if it exists.
097     */
098    String getRequestURIWithQueryString();
099
100    /**
101     * {@link #getRequestURL()} plus additional query string part, if it exists.
102     */
103    StringBuffer getRequestURLWithQueryString();

105    /**
106     * Gets the {@link RequestDispatcher} that represents a specific view
107     * for the given object.
108     *
109     * This support both JSP and Jelly.
110     *
111     * @param viewName
112     *      If this name is relative name like "foo.jsp" or "bar/zot.jelly",
113     *      then the corresponding "side file" is searched by this name.
114     *      <p>
115     *      For Jelly, this also accepts absolute path name that starts
116     *      with '/', such as "/foo/bar/zot.jelly". In this case,
117     *      <tt>it.getClass().getClassLoader()</tt> is searched for this script. 
118     *
119     * @return null
120     *      if neither JSP nor Jelly is not found by the given name.
121     */
122    RequestDispatcher getView(Object it,String viewName) throws IOException;
123
124    /**
125     * Convenience method to call {@link #getView(Klass, String)} with {@link Class}.
126     */
127    RequestDispatcher getView(Class clazz,String viewName) throws IOException;

129    /**
130     * Gets the {@link RequestDispatcher} that represents a specific view
131     * for the given class.
132     *
133     * <p>
134     * Unlike {@link #getView(Object, String)}, calling this request dispatcher
135     * doesn't set the "it" variable, so
136     * {@code getView(it.getClass(),viewName)} and {@code getView(it,viewName)}
137     * aren't the same thing.
138     */
139    RequestDispatcher getView(Klass<?> clazz, String viewName) throws IOException;

141    /**
142     * Gets the part of the request URL from protocol up to the context path.
143     * So typically it's something like <tt>http://foobar:8080/something</tt>
144     */

147    /**
148     * Gets the referer header (like "http://foobar.com/zot") or null.

154    /**
155     * Returns a list of ancestor objects that lead to the "it" object.
156     * The returned list contains {@link Ancestor} objects sorted in the
157     * order from root to the "it" object.
158     *
159     * <p>
160     * For example, if the URL was "foo/bar/zot" and the "it" object
161     * was determined as <code>root.getFoo().getBar("zot")</code>,
162     * then this list will contain the following 3 objects in this order:
163     * <ol>
164     *  <li>the root object
165     *  <li>root.getFoo() object
166     *  <li>root.getFoo().getBar("zot") object (the "it" object)
167     * </ol>
168     * <p>
169     * 
170     *
171     * @return
172     *      list of {@link Ancestor}s. Can be empty, but always non-null.
173     */
174    List<Ancestor> getAncestors();
175
176    /**
177     * Finds the nearest ancestor that has the object of the given type, or null if not found.
178     */
179    Ancestor findAncestor(Class type);
180
181    /**
182     * Short for {@code findAncestor(type).getObject()}, with proper handling for null de-reference.
183     * This version is also type safe.
184     */
185    <T> T findAncestorObject(Class<T> type);
186
187    /**
188     * Finds the nearest ancestor whose {@link Ancestor#getObject()} matches the given object.
189     */
190    Ancestor findAncestor(Object o);

193     * Short for {@code getParameter(name)!=null}
194     */
195    boolean hasParameter(String name);

197    /**
198     * Gets the {@link HttpServletRequest#getRequestURI() request URI}
199     * of the original request, so that you can access the value even from
200     * JSP.
201     */
202    String getOriginalRequestURI();

204    /**
205     * Checks "If-Modified-Since" header and returns false
206     * if the resource needs to be served.
207     *
208     * <p>
209     * This method can behave in three ways.
210     *
211     * <ol>
212     *  <li>If <tt>timestampOfResource</tt> is 0 or negative,
213     *      this method just returns false.
214     *
215     *  <li>If "If-Modified-Since" header is sent and if it's bigger than
216     *      <tt>timestampOfResource</tt>, then this method sets
217     *      {@link HttpServletResponse#SC_NOT_MODIFIED} as the response code
218     *      and returns true.
219     *
220     *  <li>Otherwise, "Last-Modified" header is added with <tt>timestampOfResource</tt> value,
221     *      and this method returns false.
222     * </ol>
223     *
224     * <p>
225     * This method sends out the "Expires" header to force browser
226     * to re-validate all the time.
227     *
228     * @param timestampOfResource
229     *      The time stamp of the resource.
230     * @param rsp
231     *      This object is updated accordingly to simplify processing.
232     *
233     * @return
234     *      false to indicate that the caller has to serve the actual resource.
235     *      true to indicate that the caller should just quit processing right there
236     *      (and send back {@link HttpServletResponse#SC_NOT_MODIFIED}.
237     */
238    boolean checkIfModified(long timestampOfResource, StaplerResponse rsp);
239
240    /**
241     * @see #checkIfModified(long, StaplerResponse)
242     */
243    boolean checkIfModified(Date timestampOfResource, StaplerResponse rsp);
244
245    /**
246     * @see #checkIfModified(long, StaplerResponse)
247     */
248    boolean checkIfModified(Calendar timestampOfResource, StaplerResponse rsp);
249
250    /**
251     * @param expiration
252     *      The number of milliseconds until the resource will "expire".
253     *      Until it expires the browser will be allowed to cache it
254     *      and serve it without checking back with the server.
255     *      After it expires, the client will send conditional GET to
256     *      check if the resource is actually modified or not.
257     *      If 0, it will immediately expire.
258     *
259     * @see #checkIfModified(long, StaplerResponse)
260     */
261    boolean checkIfModified(long timestampOfResource, StaplerResponse rsp, long expiration);

263    /**
264     * Binds form parameters to a bean by using introspection.
265     *
266     * For example, if there's a parameter called 'foo' that has value 'abc',
267     * then <tt>bean.setFoo('abc')</tt> will be invoked. This will be repeated
268     * for all parameters. Parameters that do not have corresponding setters will
269     * be simply ignored.
270     *
271     * <p>
272     * Values are converted into the right type. See {@link ConvertUtils#convert(String, Class)}.
273     *
274     * @see BeanUtils#setProperty(Object, String, Object)
275     *
276     * @param bean
277     *      The object which will be filled out.
278     */
279    void bindParameters( Object bean );
280
281    /**
282     * Binds form parameters to a bean by using introspection.
283     *
284     * This method works like {@link #bindParameters(Object)}, but it performs a
285     * pre-processing on property names. Namely, only property names that start
286     * with the given prefix will be used for binding, and only the portion of the
287     * property name after the prefix is used.
288     *
289     * So for example, if the prefix is "foo.", then property name "foo.bar" with value
290     * "zot" will invoke <tt>bean.setBar("zot")</tt>.
291     *
292     *
293     * @deprecated
294     *      Instead of using prefix to group object among form parameter names,
295     *      use structured form submission and {@link #bindJSON(Class, JSONObject)}.
296     */
297    void bindParameters( Object bean, String prefix );
298
299    /**
300     * Binds collection form parameters to beans by using introspection or
301     * constructor parameters injection.
302     *
303     * <p>
304     * This method works like {@link #bindParameters(Object,String)} and
305     * {@link #bindParameters(Class, String)}, but it assumes
306     * that form parameters have multiple-values, and use individual values to
307     * fill in multiple beans.
308     *
309     * <p>
310     * For example, if <tt>getParameterValues("foo")=={"abc","def"}</tt>
311     * and <tt>getParameterValues("bar")=={"5","3"}</tt>, then this method will
312     * return two objects (the first with "abc" and "5", the second with
313     * "def" and "3".)
314     *
315     * @param type
316     *      Type of the bean to be created. This class must have the default no-arg
317     *      constructor.
318     *
319     * @param prefix
320     *      See {@link #bindParameters(Object, String)} for details.
321     *
322     * @return
323     *      Can be empty but never null.
324     *
325     *
326     * @deprecated
327     *      Instead of using prefix to group object among form parameter names,
328     *      use structured form submission and {@link #bindJSON(Class, JSONObject)}.
329     */
330    <T>
331    List<T> bindParametersToList( Class<T> type, String prefix );
332
333    /**
334     * Instantiates a new object by injecting constructor parameters from the form parameters.
335     *
336     * <p>
337     * The given class must have a constructor annotated with '&#64;stapler-constructor',
338     * and must be processed by the maven-stapler-plugin, so that the parameter names
339     * of the constructor is available at runtime.
340     *
341     * <p>
342     * The prefix is used to control the form parameter name. For example,
343     * if the prefix is "foo." and if the constructor is define as
344     * <code>Foo(String a, String b)</code>, then the constructor will be invoked
345     * as <code>new Foo(getParameter("foo.a"),getParameter("foo.b"))</code>.
346     *
347     * @deprecated
348     *      Instead of using prefix to group object among form parameter names,
349     *      use structured form submission and {@link #bindJSON(Class, JSONObject)}.
350     */
351    <T>
352    T bindParameters( Class<T> type, String prefix );
353
354    /**
355     * Works like {@link #bindParameters(Class, String)} but uses n-th value
356     * of all the parameters.
357     *
358     * <p>
359     * This is useful for creating multiple instances from repeated form fields.
360     *
361     *
362     * @deprecated
363     *      Instead of using prefix to group object among form parameter names,
364     *      use structured form submission and {@link #bindJSON(Class, JSONObject)}.
365     */
366    <T>
367    T bindParameters( Class<T> type, String prefix, int index );

369    /**
370     * Data-bind from a {@link JSONObject} to the given target type,
371     * by using introspection or constructor parameters injection.
372     *
373     * <p>
374     * For example, if you have a constructor that looks like the following:
375     *
376     * <pre>
377     * class Foo {
378     *   &#64;{@link DataBoundConstructor}
379     *   public Foo(Integer x, String y, boolean z, Bar bar) { ... }
380     * }
381     *
382     * class Bar {
383     *   &#64;{@link DataBoundConstructor}
384     *   public Bar(int x, int y) {}
385     * }
386     * </pre>
387     *
388     * ... and if JSONObject looks like
389     *
390     * <pre>{ y:"text", z:true, bar:{x:1,y:2}}</pre>
391     *
392     * then, this method returns
393     *
394     * <pre>new Foo(null,"text",true,new Bar(1,2))</pre>
395     *
396     * <h3>Sub-typing</h3>
397     * <p>
398     * In the above example, a new instance of <tt>Bar</tt> was created,
399     * but you can also create a subtype of Bar by having the '$class' property in
400     * JSON like this:
401     *
402     * <pre>
403     * class BarEx extends Bar {
404     *   &#64;{@link DataBoundConstructor}
405     *   public BarEx(int a, int b, int c) {}
406     * }
407     *
408     * { y:"text", z:true, bar: { $class:"p.k.g.BarEx", a:1, b:2, c:3 } }
409     * </pre>
410     *
411     * <p>
412     * The type that shows up in the constructor (<tt>Bar</tt> in this case)
413     * can be an interface or an abstract class.
414     */
415    <T>
416    T bindJSON(Class<T> type, JSONObject src);
417
418    /**
419     * Data-bind from one of the JSON object types ({@link JSONObject}, {@link JSONArray},
420     * {@link String}, {@link Integer}, and so on) to the expected type given as an argument.
421     *
422     * @param genericType
423     *      The generic type of the 'erasure' parameter.
424     * @param erasure
425     *      The expected type to convert the JSON argument to.
426     * @param json
427     *      One of the JSON value type.
428     */
429    <T> T bindJSON(Type genericType, Class<T> erasure, Object json);
430
431    /**
432     * Data-binds from {@link JSONObject} to the given object.
433     *
434     * <p>
435     * This method is bit like {@link #bindJSON(Class, JSONObject)}, except that this method
436     * populates an existing object, instead of creating a new instance.
437     *
438     * <p>
439     * This method is also bit like {@link #bindParameters(Object, String)}, in that it
440     * populates an existing object from a form submission, except that this method
441     * obtains data from {@link JSONObject} thus more structured, whereas {@link #bindParameters(Object, String)}
442     * uses the map structure of the form submission. 
443     */
444    void bindJSON( Object bean, JSONObject src );
445
446    /**
447     * Data-bind from either {@link JSONObject} or {@link JSONArray} to a list,
448     * by using {@link #bindJSON(Class, JSONObject)} as the lower-level mechanism.
449     *
450     * <p>
451     * If the source is {@link JSONObject}, the returned list will contain
452     * a single item. If it is {@link JSONArray}, each item will be bound.
453     * If it is null, then the list will be empty.
454     */
455    <T>
456    List<T> bindJSONToList(Class<T> type, Object src);
