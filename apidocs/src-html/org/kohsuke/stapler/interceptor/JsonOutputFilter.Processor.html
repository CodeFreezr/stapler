023package org.kohsuke.stapler.interceptor;
024
025import net.sf.json.JsonConfig;
026import net.sf.json.util.PropertyFilter;
027import org.kohsuke.stapler.StaplerRequest;
028import org.kohsuke.stapler.StaplerResponse;
029import org.kohsuke.stapler.bind.JavaScriptMethod;
030
031import javax.servlet.ServletException;
032import java.lang.annotation.Retention;
033import java.lang.annotation.Target;
034import java.lang.reflect.InvocationTargetException;
035import java.util.Set;
036
037import static java.lang.annotation.ElementType.FIELD;
038import static java.lang.annotation.ElementType.METHOD;
039import static java.lang.annotation.RetentionPolicy.RUNTIME;
040import java.util.Arrays;
041import java.util.HashSet;
042
043/**
044 * Annotation for filtering the JSON data returned from a {@link JavaScriptMethod} annotated method.
045 * Put this annotation on your js proxied method and provide the properties you want filtered.
046 *
047 * @author Robert Sandell &lt;sandell.robert@gmail.com&gt;
048 */
049@Retention(RUNTIME)
050@Target({METHOD, FIELD})
051@InterceptorAnnotation(JsonOutputFilter.Processor.class)
052public @interface JsonOutputFilter {
053
054    /**
055     * White-list of properties to include in the output.
056     */
057    String[] includes() default {};
058
059    /**
060     * Black-list of properties to exclude from the output.
061     */
062    String[] excludes() default {};
063
064    /**
065     * If transient fields should be ignored. Default true.
066     *
067     * @see JsonConfig#isIgnoreTransientFields()
068     */
069    boolean ignoreTransient() default true;
070
071    /**
072     * If {@link JsonConfig#DEFAULT_EXCLUDES} should be ignored. Default false
073     *
074     * @see JsonConfig#isIgnoreDefaultExcludes()
075     */
076    boolean ignoreDefaultExcludes() default false;
077
078    public static class Processor extends Interceptor {
079        @Override
080        public Object invoke(StaplerRequest request, StaplerResponse response, Object instance, Object[] arguments)
081                throws IllegalAccessException, InvocationTargetException, ServletException {
082            JsonOutputFilter annotation = target.getAnnotation((JsonOutputFilter.class));
083            if (annotation != null) {
084                JsonConfig config = new JsonConfig();
085                config.setJsonPropertyFilter(new FilterPropertyFilter(annotation.includes(), annotation.excludes()));
086                config.setIgnoreTransientFields(annotation.ignoreTransient());
087                config.setIgnoreDefaultExcludes(annotation.ignoreDefaultExcludes());
088                response.setJsonConfig(config);
089            }
090            return target.invoke(request, response, instance, arguments);
091        }
092    }
093
094    /**
095     * Json Property filter for handling the include and exclude.
096     */
097    static class FilterPropertyFilter implements PropertyFilter {
098
099        private Set<String> includes;
100        private Set<String> excludes;
101
102        public FilterPropertyFilter(String[] includes, String[] excludes) {
103            this(new HashSet<String>(Arrays.asList(includes)), new HashSet<String>(Arrays.asList(excludes)));
104        }
105
106        public FilterPropertyFilter(Set<String> includes, Set<String> excludes) {
107            this.includes = includes;
108            this.excludes = excludes;
109        }
110
111        public boolean apply(Object source, String name, Object value) {
112            if (excludes.contains(name)) {
113                return true;
114            } else if (!includes.isEmpty()) {
115                return !includes.contains(name);
116            } else {
117                return false;
118            }
119        }
120    }
121}
