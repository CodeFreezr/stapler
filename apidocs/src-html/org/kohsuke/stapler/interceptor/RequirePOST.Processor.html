001package org.kohsuke.stapler.interceptor;
002
003import java.io.IOException;
004import java.io.PrintWriter;
005import org.kohsuke.stapler.StaplerRequest;
006import org.kohsuke.stapler.StaplerResponse;
007
008import java.lang.annotation.Retention;
009import java.lang.annotation.Target;
010import java.lang.reflect.InvocationTargetException;
011
012import static java.lang.annotation.ElementType.*;
013import static java.lang.annotation.RetentionPolicy.*;
014import javax.servlet.ServletException;
015import javax.servlet.http.HttpServletResponse;
016import org.kohsuke.stapler.HttpResponses;
017
018/**
019 * Requires the request to be a POST.
020 *
021 * @author Kohsuke Kawaguchi
022 * @since 1.180
023 */
024@Retention(RUNTIME)
025@Target({METHOD,FIELD})
026@InterceptorAnnotation(RequirePOST.Processor.class)
027public @interface RequirePOST {
028    public static class Processor extends Interceptor {
029        @Override
030        public Object invoke(StaplerRequest request, StaplerResponse response, Object instance, Object[] arguments)
031                throws IllegalAccessException, InvocationTargetException {
032            if (!request.getMethod().equals("POST")) {
033                throw new InvocationTargetException(new HttpResponses.HttpResponseException() {
034                    public void generateResponse(StaplerRequest req, StaplerResponse rsp, Object node) throws IOException, ServletException {
035                        rsp.setStatus(HttpServletResponse.SC_METHOD_NOT_ALLOWED);
036                        rsp.addHeader("Allow", "POST");
037                        rsp.setContentType("text/html");
038                        PrintWriter w = rsp.getWriter();
039                        w.println("<html><head><title>POST required</title></head><body>");
040                        w.println("POST is required for " + target.getQualifiedName() + "<br>");
041                        w.println("<form method='POST'><input type='submit' value='Try POSTing'></form>");
042                        w.println("</body></html>");
043                    }
044                });
045            }
046            return target.invoke(request, response, instance, arguments);
047        }
048    }
049}
