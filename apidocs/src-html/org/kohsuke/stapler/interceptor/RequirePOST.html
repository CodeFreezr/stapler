001package org.kohsuke.stapler.interceptor;
002
003import java.io.IOException;
004import java.io.PrintWriter;
005import org.kohsuke.stapler.StaplerRequest;
006import org.kohsuke.stapler.StaplerResponse;
007
008import java.lang.annotation.Retention;
009import java.lang.annotation.Target;
010import java.lang.reflect.InvocationTargetException;
011
012import static java.lang.annotation.ElementType.*;
013import static java.lang.annotation.RetentionPolicy.*;
014import javax.servlet.ServletException;
015import javax.servlet.http.HttpServletResponse;
016import org.kohsuke.stapler.HttpResponses;
017import org.kohsuke.stapler.verb.POST;
018
019/**
020 * Requires the request to be a POST.
021 *
022 * <p>
023 * When the current request has a non-matching HTTP method (such as 'GET'), this annotation
024 * will send a failure response instead of searching for other matching web methods.
025 *
026 * @author Kohsuke Kawaguchi
027 * @since 1.180
028 * @see POST
029 */
030@Retention(RUNTIME)
031@Target({METHOD,FIELD})
032@InterceptorAnnotation(RequirePOST.Processor.class)
033public @interface RequirePOST {
034    public static class Processor extends Interceptor {
035        @Override
036        public Object invoke(StaplerRequest request, StaplerResponse response, Object instance, Object[] arguments)
037                throws IllegalAccessException, InvocationTargetException, ServletException {
038            if (!request.getMethod().equals("POST")) {
039                throw new InvocationTargetException(new HttpResponses.HttpResponseException() {
040                    public void generateResponse(StaplerRequest req, StaplerResponse rsp, Object node) throws IOException, ServletException {
041                        rsp.setStatus(HttpServletResponse.SC_METHOD_NOT_ALLOWED);
042                        rsp.addHeader("Allow", "POST");
043                        rsp.setContentType("text/html");
044                        PrintWriter w = rsp.getWriter();
045                        w.println("<html><head><title>POST required</title></head><body>");
046                        w.println("POST is required for " + target.getQualifiedName() + "<br>");
047                        w.println("<form method='POST'><input type='submit' value='Try POSTing'></form>");
048                        w.println("</body></html>");
049                    }
050                });
051            }
052            return target.invoke(request, response, instance, arguments);
053        }
054    }
055}
