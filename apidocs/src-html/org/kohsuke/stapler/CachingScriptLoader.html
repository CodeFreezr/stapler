001package org.kohsuke.stapler;
002
003import com.google.common.cache.CacheBuilder;
004import com.google.common.cache.CacheLoader;
005import com.google.common.cache.LoadingCache;
006
007import java.net.URL;
008
009/**
010 * Convenient base class for caching loaded scripts.
011 * @author Kohsuke Kawaguchi
012 */
013public abstract class CachingScriptLoader<S, E extends Exception> {
014    /**
015     * Compiled scripts of this class.
016     * Access needs to be synchronized.
017     *
018     * <p>
019     * Jelly leaks memory (because Scripts hold on to Tag)
020     * which usually holds on to JellyContext that was last used to run it,
021     * which often holds on to some big/heavy objects.)
022     *
023     * So it's important to allow Scripts to be garbage collected.
024     * This is not an ideal fix, but it works.
025     *
026     * {@link Optional} is used as Google Collection doesn't allow null values in a map.
027     */
028    private final LoadingCache<String,Optional<S>> scripts = CacheBuilder.newBuilder().softValues().build(new CacheLoader<String, Optional<S>>() {
029        public Optional<S> load(String from) {
030            try {
031                return Optional.create(loadScript(from));
032            } catch (RuntimeException e) {
033                throw e;    // pass through
034            } catch (Exception e) {
035                throw new ScriptLoadException(e);
036            }
037        }
038    });
039
040    /**
041     * Locates the view script of the given name.
042     *
043     * @param name
044     *      if this is a relative path, such as "foo.jelly" or "foo/bar.groovy",
045     *      then it is assumed to be relative to this class, so
046     *      "org/acme/MyClass/foo.jelly" or "org/acme/MyClass/foo/bar.groovy"
047     *      will be searched.
048     *      <p>
049     *      If the extension is omitted, the default extension will be appended.
050     *      This is useful for some loaders that support loading multiple file extension types
051     *      (such as Jelly support.)
052     *      <p>
053     *      If this starts with "/", then it is assumed to be absolute,
054     *      and that name is searched from the classloader. This is useful
055     *      to do mix-in.
056     * @return null if none was found.
057     */
058    public S findScript(String name) throws E {
059        if (MetaClass.NO_CACHE) 
060            return loadScript(name);
061        else
062            return scripts.getUnchecked(name).get();
063    }
064
065    /**
066     * Cache-less version of the {@link #findScript(String)} that provides the actual logic.
067     */
068    protected abstract S loadScript(String name) throws E;
069
070    /**
071     * Discards the cached script.
072     */
073    public synchronized void clearScripts() {
074        scripts.invalidateAll();
075    }
076
077    protected abstract URL getResource(String name);
078}
