024package org.kohsuke.stapler.jelly;
025
026import org.apache.commons.jelly.JellyContext;
027import org.apache.commons.jelly.JellyException;
028import org.apache.commons.jelly.Script;
029import org.apache.commons.jelly.Tag;
030import org.apache.commons.jelly.TagLibrary;
031import org.apache.commons.jelly.impl.TagScript;
032import org.kohsuke.stapler.MetaClass;
033import org.kohsuke.stapler.MetaClassLoader;
034import org.xml.sax.Attributes;
035
036import java.net.URL;
037import java.util.Hashtable;
038import java.util.List;
039import java.util.Map;
040
041/**
042 * {@link TagLibrary} that loads tags from tag files in a directory.
043 *
044 * @author Kohsuke Kawaguchi
045 */
046public final class CustomTagLibrary extends TagLibrary {
047
048    /**
049     * Inherits values from this context.
050     * This context would be shared by multiple threads,
051     * but as long as we all just read it, it should be OK.
052     */
053    private final JellyContext master;
054
055    private final ClassLoader classLoader;
056    public final MetaClassLoader metaClassLoader;
057    public final String nsUri;
058    public final String basePath;
059
060    /**
061     * Compiled tag files.
062     */
063    private final Map<String,Script> scripts = new Hashtable<String,Script>();
064
065    private final List<JellyTagFileLoader> loaders;
066
067    public CustomTagLibrary(JellyContext master, ClassLoader classLoader, String nsUri, String basePath) {
068        this.master = master;
069        this.classLoader = classLoader;
070        this.nsUri = nsUri;
071        this.basePath = basePath;
072        this.metaClassLoader = MetaClassLoader.get(classLoader);
073        this.loaders = JellyTagFileLoader.discover(classLoader);
074    }
075
076    public TagScript createTagScript(String name, Attributes attributes) throws JellyException {
077        final Script def = load(name);
078        if(def==null) return null;
079
080        return new CallTagLibScript() {
081            protected Script resolveDefinition(JellyContext context) {
082                return def;
083            }
084        };
085    }
086
087    public Tag createTag(String name, Attributes attributes) throws JellyException {
088        // IIUC, this method is only used by static tag to discover the correct tag at runtime,
089        // and since stapler taglibs are always resolved statically, we shouldn't have to implement this method
090        // at all.
091
092        // by not implementing this method, we can put all the login in the TagScript-subtype, which eliminates
093        // the need of stateful Tag instances and their overheads.
094        return null;
095    }
096
097    /**
098     * Obtains the script for the given tag name. Loads if necessary.
099     *
100     * <p>
101     * Synchronizing this method would have a potential race condition
102     * if two threads try to load two tags that are referencing each other.
103     *
104     * <p>
105     * So we synchronize {@link #scripts}, even though this means
106     * we may end up compiling the same script twice.
107     */
108    private Script load(String name) throws JellyException {
109
110        Script script = scripts.get(name);
111        if(script!=null && !MetaClass.NO_CACHE)
112            return script;
113
114        script=null;
115        if(MetaClassLoader.debugLoader!=null)
116            script = load(name, MetaClassLoader.debugLoader.loader);
117        if(script==null)
118            script = load(name, classLoader);
119        return script;
120    }
121
122    private Script load(String name, ClassLoader classLoader) throws JellyException {
123        Script script;
124        // prefer 'foo.jellytag' but for backward compatibility, support the plain .jelly extension as well.
125        URL res = classLoader.getResource(basePath + '/' + name + ".jellytag");
126        if (res==null)
127            res = classLoader.getResource(basePath + '/' + name + ".jelly");
128        if(res!=null) {
129            script = loadJellyScript(res);
130            scripts.put(name,script);
131            return script;
132        }
133
134        for (JellyTagFileLoader loader : loaders) {
135            Script s = loader.load(this, name, classLoader);
136            if(s!=null) {
137                scripts.put(name,s);
138                return s;
139            }
140        }
141
142        return null;
143    }
144
145    private Script loadJellyScript(URL res) throws JellyException {
146        // compile script
147        JellyContext context = new CustomJellyContext(master);
148        context.setClassLoader(classLoader);
149        return context.compileScript(res);
150    }
151}
