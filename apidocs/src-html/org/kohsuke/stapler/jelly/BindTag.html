024package org.kohsuke.stapler.jelly;
025
026import org.apache.commons.jelly.JellyTagException;
027import org.apache.commons.jelly.XMLOutput;
028import org.jvnet.maven.jellydoc.annotation.NoContent;
029import org.jvnet.maven.jellydoc.annotation.Required;
030import org.kohsuke.stapler.WebApp;
031import org.kohsuke.stapler.bind.Bound;
032import org.kohsuke.stapler.framework.adjunct.AdjunctsInPage;
033import org.xml.sax.SAXException;
034
035/**
036 * Binds a server-side object to client side so that JavaScript can call into server.
037 * This tag evaluates to a &lt;script> tag.
038 * 
039 * @author Kohsuke Kawaguchi
040 */
041@NoContent
042public class BindTag extends AbstractStaplerTag {
043    private String varName;
044    private Object javaObject;
045
046    /**
047     * JavaScript variable name to set the proxy to.
048     * <p>
049     * This name can be arbitrary left hand side expression,
050     * such as "a[0]" or "a.b.c".
051     *
052     * If this value is unspecified, the tag generates a JavaScript expression to create a proxy.
053     */
054    public void setVar(String varName) {
055        this.varName = varName;
056    }
057
058    @Required
059    public void setValue(Object o) {
060        this.javaObject = o;
061    }
062
063    public void doTag(XMLOutput out) throws JellyTagException {
064        // make sure we get the supporting script in place
065        AdjunctsInPage adjuncts = AdjunctsInPage.get();
066        if (!adjuncts.isIncluded("org.kohsuke.stapler.framework.prototype.prototype") &&
067                !adjuncts.isIncluded("org.kohsuke.stapler.jquery")) {
068            //We try to not include prototype if jQuery is already included, but default to prototype
069            AdjunctTag a = new AdjunctTag();
070            a.setContext(getContext());
071            a.setIncludes("org.kohsuke.stapler.framework.prototype.prototype");
072            a.doTag(out);
073        } else if (adjuncts.isIncluded("org.kohsuke.stapler.jquery")) {
074            //Old browsers (like htmlunit or <=IE7) doesn't support JSON.stringify needed to do the bind call via jQuery
075            //So to be on the safe side we include json2.js. See https://github.com/douglascrockford/JSON-js
076            AdjunctTag a = new AdjunctTag();
077            a.setContext(getContext());
078            a.setIncludes("org.kohsuke.stapler.json2");
079            a.doTag(out);
080        }
081        AdjunctTag a = new AdjunctTag();
082        a.setContext(getContext());
083        a.setIncludes("org.kohsuke.stapler.bind");
084        a.doTag(out);
085
086        try {
087            String expr;
088            if (javaObject==null) {
089                expr = "null";
090            } else {
091                Bound h = WebApp.getCurrent().boundObjectTable.bind(javaObject);
092                expr = h.getProxyScript();
093            }
094
095            if (varName==null) {
096                // this mode (of writing just the expression) needs to be used with caution because
097                // the adjunct tag above might produce <script> tag.
098                out.write(expr);
099            } else {
100                out.startElement("script");
101                out.write(varName + "=" + expr + ";");
102                out.endElement("script");
103            }
104        } catch (SAXException e) {
105            throw new JellyTagException(e);
106        }
107    }
108}
