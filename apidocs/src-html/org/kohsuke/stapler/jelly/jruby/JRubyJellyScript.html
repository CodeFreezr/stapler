001package org.kohsuke.stapler.jelly.jruby;
002
003import org.apache.commons.jelly.JellyContext;
004import org.apache.commons.jelly.JellyException;
005import org.apache.commons.jelly.JellyTagException;
006import org.apache.commons.jelly.Script;
007import org.apache.commons.jelly.TagLibrary;
008import org.apache.commons.jelly.XMLOutput;
009import org.apache.commons.jelly.expression.ConstantExpression;
010import org.apache.commons.jelly.impl.TagScript;
011import org.jruby.Ruby;
012import org.jruby.RubyProc;
013import org.jruby.RubySymbol;
014import org.jruby.runtime.builtin.IRubyObject;
015
016import java.util.Map;
017import java.util.Map.Entry;
018
019/**
020 * Exposes ERb/Haml scripts as {@link Script}.
021 *
022 * <p>
023 * This abstract class defines the Java half of it, and it is further subtyped in Ruby for different languages.
024 *
025 * @author Kohsuke Kawaguchi
026 */
027public abstract class JRubyJellyScript implements Script {
028
029    protected JRubyJellyScript() {
030    }
031
032    public Script compile() throws JellyException {
033        return this;
034    }
035
036    // this method is implemented in Ruby
037    public abstract void run(JellyContext context, XMLOutput output) throws JellyTagException;
038
039    /**
040     * Invokes other Jelly tag libaries.
041     */
042    public void invokeTaglib(final IJRubyContext rcon, JellyContext context, XMLOutput output, String uri, String localName, Map<RubySymbol,?> attributes, final RubyProc proc) throws JellyException {
043        TagScript tagScript = createTagScript(context, uri, localName);
044
045        if (attributes!=null) {
046            for (Entry<RubySymbol, ?> e : attributes.entrySet()) {
047                tagScript.addAttribute(e.getKey().asJavaString(), new ConstantExpression(e.getValue()));
048            }
049        }
050        
051        if (proc!=null) {
052            final Ruby runtime = ((IRubyObject)rcon).getRuntime();
053
054            tagScript.setTagBody(new Script() {
055                public Script compile() throws JellyException {
056                    return this;
057                }
058
059                public void run(JellyContext context, XMLOutput output) throws JellyTagException {
060                    JellyContext oc = rcon.getJellyContext();
061                    XMLOutput oo = rcon.getOutput();
062                    try {
063                        rcon.setJellyContext(context);
064                        rcon.setOutput(output);
065                        proc.getBlock().yield(runtime.getCurrentContext(),null);
066                    } finally {
067                        rcon.setJellyContext(oc);
068                        rcon.setOutput(oo);
069                    }
070                }
071            });
072        }
073        tagScript.run(context, output);
074    }
075
076    private TagScript createTagScript(JellyContext context, String uri, String name) throws JellyException {
077        TagLibrary lib = context.getTagLibrary(uri);
078        if (lib==null)
079            throw new JellyException("Undefined tag library namespace URI: "+uri);
080        TagScript tagScript = lib.createTagScript(name, null/*this parameter appears to be unused.*/);
081        if (tagScript!=null) return tagScript;
082        tagScript = lib.createTagScript(name.replace('_','-'), null);
083        if (tagScript!=null) return tagScript;
084        throw new JellyException(String.format("name '%s' not found for '%s'", name, uri));
085    }
086}
