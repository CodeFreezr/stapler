001package org.kohsuke.stapler.jelly.jruby;
002
003import org.jruby.embed.ScriptingContainer;
004
005/**
006 * Ruby template language binding.
007 *
008 * <p>
009 * Implementations of this is discovered via service-loader mechanism.
010 *
011 * @author Kohsuke Kawaguchi
012 */
013public abstract class RubyTemplateLanguage {
014    /**
015     * Defines the file extension, like ".erb", that designates this kind of view type.
016     */
017    protected abstract String getScriptExtension();
018
019    protected abstract Class<? extends AbstractRubyTearOff> getTearOffClass();
020
021    /**
022     * Called to set up this template language binding on the specified scripting container.
023     */
024    protected abstract RubyTemplateContainer createContainer(ScriptingContainer container);
025}
