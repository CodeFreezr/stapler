001package org.kohsuke.stapler.jelly.jruby.haml;
002
003import org.jruby.RubyClass;
004import org.jruby.embed.ScriptingContainer;
005import org.kohsuke.MetaInfServices;
006import org.kohsuke.stapler.jelly.jruby.RubyTemplateContainer;
007import org.kohsuke.stapler.jelly.jruby.RubyTemplateLanguage;
008
009/**
010 * @author Hiroshi Nakamura
011 */
012@MetaInfServices
013public class HamlLanguage extends RubyTemplateLanguage {
014    @Override
015    protected String getScriptExtension() {
016        return ".haml";
017    }
018
019    @Override
020    protected Class<HamlClassTearOff> getTearOffClass() {
021        return HamlClassTearOff.class;
022    }
023
024    @Override
025    protected RubyTemplateContainer createContainer(ScriptingContainer jruby) {
026        jruby.put("gem_path", getClass().getResource("/gem").getPath());
027
028        return new RubyTemplateContainer(
029            (RubyClass)jruby.runScriptlet("ENV['GEM_PATH'] = gem_path\n" +
030                "require 'rubygems'\n" +
031                "require 'org/kohsuke/stapler/jelly/jruby/JRubyJellyScriptImpl'\n"+
032                "require 'org/kohsuke/stapler/jelly/jruby/haml/JRubyJellyHamlScript'\n"+
033                "JRubyJellyScriptImpl::JRubyJellyHamlScript"
034            ),
035            this, jruby);
036    }
037}
