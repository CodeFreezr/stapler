001package org.kohsuke.stapler.jelly.jruby;
002
003import org.jruby.RubyModule;
004import org.jruby.internal.runtime.methods.DynamicMethod;
005import org.jruby.javasupport.Java;
006import org.jruby.runtime.builtin.IRubyObject;
007import org.kohsuke.stapler.lang.MethodRef;
008
009import java.lang.annotation.Annotation;
010import java.lang.reflect.InvocationTargetException;
011
012/**
013 * @author Kohsuke Kawaguchi
014 */
015public class RubyMethodRef extends MethodRef {
016    private final RubyModule klass;
017    private final DynamicMethod method;
018
019
020    public RubyMethodRef(RubyModule klass, DynamicMethod method) {
021        this.klass = klass;
022        this.method = method;
023    }
024
025    @Override
026    public <T extends Annotation> T getAnnotation(Class<T> type) {
027        // TODO: what's the equivalent in JRuby?
028        return null;
029    }
030
031    @Override
032    public Object invoke(Object _this, Object... args) throws InvocationTargetException, IllegalAccessException {
033        IRubyObject[] argList = new IRubyObject[args.length];
034        for (int i=0; i<args.length; i++)
035            argList[i] = Java.getInstance(klass.getRuntime(),args[i]);
036        return method.call(klass.getRuntime().getCurrentContext() , (IRubyObject)_this, klass, method.getName(), argList);
037    }
038}
