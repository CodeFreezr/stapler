001package org.kohsuke.stapler.jelly.jruby;
002
003import org.jruby.Ruby;
004import org.jruby.RubyClass;
005import org.jruby.RubyModule;
006import org.jruby.RubyObject;
007import org.kohsuke.stapler.MetaClassLoader;
008import org.kohsuke.stapler.lang.Klass;
009import org.kohsuke.stapler.lang.KlassNavigator;
010
011import java.net.URL;
012import java.util.ArrayList;
013import java.util.List;
014import java.util.Locale;
015
016/**
017 * @author Kohsuke Kawaguchi
018 */
019public class RubyKlassNavigator extends KlassNavigator<RubyModule> {
020    private final Ruby ruby;
021    /**
022     * ClassLoader to load resources from.
023     */
024    private final ClassLoader classLoader;
025
026    public RubyKlassNavigator(Ruby ruby, ClassLoader cl) {
027        this.ruby = ruby;
028        this.classLoader = cl;
029    }
030    
031
032    @Override
033    public URL getResource(RubyModule clazz, String resourceName) {
034        String fullName;
035        if (resourceName.startsWith("/"))
036            fullName = resourceName.substring(1);
037        else
038            fullName = decamelize(clazz.getName().replace("::","/")+'/'+resourceName);
039
040        if (MetaClassLoader.debugLoader!=null) {
041            URL res = MetaClassLoader.debugLoader.loader.getResource(fullName);
042            if (res!=null)  return res;
043        }
044        return classLoader.getResource(fullName);
045    }
046
047    @Override
048    public Iterable<Klass<?>> getAncestors(RubyModule clazz) {
049        List<Klass<?>> r = new ArrayList<Klass<?>>();
050        for (RubyModule anc : (List<RubyModule>)(List)clazz.getAncestorList()) {
051            r.add(wrap(anc));
052        }
053        return r;
054    }
055
056    @Override
057    public Klass<?> getSuperClass(RubyModule clazz) {
058        // TODO: what happens when a Ruby class extends from Java class?
059        return wrap(clazz.getSuperClass());
060    }
061
062    @Override
063    public Class toJavaClass(RubyModule clazz) {
064        if (clazz instanceof RubyClass) {
065            RubyClass rc = (RubyClass) clazz;
066            Class c = rc.getReifiedClass();
067            if (c!=null)    return c;   // is this right?
068        }
069        return RubyObject.class;
070    }
071    
072    public Klass<RubyModule> wrap(RubyModule m) {
073        return m==null ? null : new Klass<RubyModule>(m,this);
074    }
075
076    /**
077     * Converts "FooBarZot" to "foo_bar_zot"
078     */
079    static String decamelize(String s) {
080        return s.replaceAll("(\\p{javaLetterOrDigit})(\\p{javaUpperCase}\\p{javaLowerCase})","$1_$2")
081                .replaceAll("(\\p{javaLowerCase})(\\p{javaUpperCase})","$1_$2")
082                .toLowerCase(Locale.ENGLISH);
083    }
084}
