001package org.kohsuke.stapler.jelly.jruby;
002
003import org.jruby.Ruby;
004import org.jruby.RubyClass;
005import org.jruby.RubyModule;
006import org.jruby.RubyObject;
007import org.jruby.internal.runtime.methods.DynamicMethod;
008import org.kohsuke.stapler.ClassDescriptor;
009import org.kohsuke.stapler.Function;
010import org.kohsuke.stapler.MetaClassLoader;
011import org.kohsuke.stapler.lang.FieldRef;
012import org.kohsuke.stapler.lang.Klass;
013import org.kohsuke.stapler.lang.KlassNavigator;
014import org.kohsuke.stapler.lang.MethodRef;
015
016import java.net.URL;
017import java.util.ArrayList;
018import java.util.Collections;
019import java.util.List;
020import java.util.Locale;
021
022/**
023 * {@link KlassNavigator} implementation for JRuby.
024 *
025 * @author Kohsuke Kawaguchi
026 */
027public class RubyKlassNavigator extends KlassNavigator<RubyModule> {
028    private final Ruby ruby;
029    /**
030     * ClassLoader to load resources from.
031     */
032    private final ClassLoader classLoader;
033
034    public RubyKlassNavigator(Ruby ruby, ClassLoader cl) {
035        this.ruby = ruby;
036        this.classLoader = cl;
037    }
038    
039
040    @Override
041    public URL getResource(RubyModule clazz, String resourceName) {
042        String fullName;
043        if (resourceName.startsWith("/"))
044            fullName = resourceName.substring(1);
045        else
046            fullName = decamelize(clazz.getName().replace("::","/")+'/'+resourceName);
047
048        if (MetaClassLoader.debugLoader!=null) {
049            URL res = MetaClassLoader.debugLoader.loader.getResource(fullName);
050            if (res!=null)  return res;
051        }
052        return classLoader.getResource(fullName);
053    }
054
055    @Override
056    public Iterable<Klass<?>> getAncestors(RubyModule clazz) {
057        List<Klass<?>> r = new ArrayList<Klass<?>>();
058        for (RubyModule anc : (List<RubyModule>)(List)clazz.getAncestorList()) {
059            r.add(wrap(anc));
060        }
061        return r;
062    }
063
064    @Override
065    public Klass<?> getSuperClass(RubyModule clazz) {
066        // TODO: what happens when a Ruby class extends from Java class?
067        return wrap(clazz.getSuperClass());
068    }
069
070    @Override
071    public List<MethodRef> getDeclaredMethods(RubyModule clazz) {
072        List<MethodRef> r = new ArrayList<MethodRef>();
073        for (DynamicMethod m : clazz.getMethods().values()) {
074            // TODO: not sure if this is entirely correct
075            if (m.getImplementationClass()==clazz)
076                r.add(new RubyMethodRef(clazz,m));
077        }
078        return r;
079    }
080
081    @Override
082    public List<FieldRef> getDeclaredFields(RubyModule clazz) {
083        // IIUC, Ruby doesn't have statically defined instance fields
084        return Collections.emptyList();
085    }
086
087    @Override
088    public List<Function> getFunctions(RubyModule clazz) {
089        // implemented as a fallback to Java through reified class, but maybe there's a better way to do this
090        return new ClassDescriptor(toJavaClass(clazz)).methods;
091    }
092
093    @Override
094    public Class toJavaClass(RubyModule clazz) {
095        if (clazz instanceof RubyClass) {
096            RubyClass rc = (RubyClass) clazz;
097            Class c = rc.getReifiedClass();
098            if (c!=null)    return c;   // is this right?
099        }
100        return RubyObject.class;
101    }
102    
103    public Klass<RubyModule> wrap(RubyModule m) {
104        return m==null ? null : new Klass<RubyModule>(m,this);
105    }
106
107    /**
108     * Converts "FooBarZot" to "foo_bar_zot"
109     */
110    static String decamelize(String s) {
111        return s.replaceAll("(\\p{javaLetterOrDigit})(\\p{javaUpperCase}\\p{javaLowerCase})","$1_$2")
112                .replaceAll("(\\p{javaLowerCase})(\\p{javaUpperCase})","$1_$2")
113                .toLowerCase(Locale.ENGLISH);
114    }
115}
