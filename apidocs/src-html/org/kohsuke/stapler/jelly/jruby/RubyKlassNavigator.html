001package org.kohsuke.stapler.jelly.jruby;
002
003import org.jruby.Ruby;
004import org.jruby.RubyClass;
005import org.jruby.RubyModule;
006import org.jruby.RubyObject;
007import org.jruby.internal.runtime.methods.DynamicMethod;
008import org.kohsuke.stapler.MetaClassLoader;
009import org.kohsuke.stapler.lang.Klass;
010import org.kohsuke.stapler.lang.KlassNavigator;
011import org.kohsuke.stapler.lang.MethodRef;
012
013import java.net.URL;
014import java.util.ArrayList;
015import java.util.List;
016import java.util.Locale;
017
018/**
019 * @author Kohsuke Kawaguchi
020 */
021public class RubyKlassNavigator extends KlassNavigator<RubyModule> {
022    private final Ruby ruby;
023    /**
024     * ClassLoader to load resources from.
025     */
026    private final ClassLoader classLoader;
027
028    public RubyKlassNavigator(Ruby ruby, ClassLoader cl) {
029        this.ruby = ruby;
030        this.classLoader = cl;
031    }
032    
033
034    @Override
035    public URL getResource(RubyModule clazz, String resourceName) {
036        String fullName;
037        if (resourceName.startsWith("/"))
038            fullName = resourceName.substring(1);
039        else
040            fullName = decamelize(clazz.getName().replace("::","/")+'/'+resourceName);
041
042        if (MetaClassLoader.debugLoader!=null) {
043            URL res = MetaClassLoader.debugLoader.loader.getResource(fullName);
044            if (res!=null)  return res;
045        }
046        return classLoader.getResource(fullName);
047    }
048
049    @Override
050    public Iterable<Klass<?>> getAncestors(RubyModule clazz) {
051        List<Klass<?>> r = new ArrayList<Klass<?>>();
052        for (RubyModule anc : (List<RubyModule>)(List)clazz.getAncestorList()) {
053            r.add(wrap(anc));
054        }
055        return r;
056    }
057
058    @Override
059    public Klass<?> getSuperClass(RubyModule clazz) {
060        // TODO: what happens when a Ruby class extends from Java class?
061        return wrap(clazz.getSuperClass());
062    }
063
064    @Override
065    public List<MethodRef> getDeclaredMethods(RubyModule clazz) {
066        List<MethodRef> r = new ArrayList<MethodRef>();
067        for (DynamicMethod m : clazz.getMethods().values()) {
068            // TODO: not sure if this is entirely correct
069            if (m.getImplementationClass()==clazz)
070                r.add(new RubyMethodRef(clazz,m));
071        }
072        return r;
073    }
074
075    @Override
076    public Class toJavaClass(RubyModule clazz) {
077        if (clazz instanceof RubyClass) {
078            RubyClass rc = (RubyClass) clazz;
079            Class c = rc.getReifiedClass();
080            if (c!=null)    return c;   // is this right?
081        }
082        return RubyObject.class;
083    }
084    
085    public Klass<RubyModule> wrap(RubyModule m) {
086        return m==null ? null : new Klass<RubyModule>(m,this);
087    }
088
089    /**
090     * Converts "FooBarZot" to "foo_bar_zot"
091     */
092    static String decamelize(String s) {
093        return s.replaceAll("(\\p{javaLetterOrDigit})(\\p{javaUpperCase}\\p{javaLowerCase})","$1_$2")
094                .replaceAll("(\\p{javaLowerCase})(\\p{javaUpperCase})","$1_$2")
095                .toLowerCase(Locale.ENGLISH);
096    }
097}
