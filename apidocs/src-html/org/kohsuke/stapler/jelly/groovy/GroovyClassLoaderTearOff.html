024package org.kohsuke.stapler.jelly.groovy;
025
026import groovy.lang.GroovyClassLoader;
027import groovy.lang.GroovyCodeSource;
028import org.apache.commons.jelly.XMLOutput;
029import org.codehaus.groovy.control.CompilerConfiguration;
030import org.kohsuke.stapler.MetaClass;
031import org.kohsuke.stapler.MetaClassLoader;
032
033import java.io.IOException;
034import java.net.URL;
035
036/**
037 * @author Kohsuke Kawaguchi
038 */
039public class GroovyClassLoaderTearOff {
040    private final MetaClassLoader owner;
041
042    private final GroovyClassLoader gcl;
043
044    private final SimpleTemplateParser parser = new SimpleTemplateParser() {
045        /**
046         * Sends the output via {@link XMLOutput#write(String)}
047         */
048        @Override
049        protected String printCommand() {
050            return "output.write";
051        }
052    };
053
054    public GroovyClassLoaderTearOff(MetaClassLoader owner) {
055        this.owner = owner;
056        gcl = createGroovyClassLoader();
057    }
058
059    private GroovyClassLoader createGroovyClassLoader() {
060        CompilerConfiguration cc = new CompilerConfiguration();
061        // use GroovyClosureScript class as the base class of the compiled script,
062        // so that we can set a delegate.
063        cc.setScriptBaseClass(StaplerClosureScript.class.getName());
064
065        // enable re-compilation support
066        cc.setRecompileGroovySource(MetaClass.NO_CACHE);
067        return new GroovyClassLoader(owner.loader,cc) {
068            /**
069             * Groovy calls this method to locate .groovy script files,
070             * so during the development it's important to check the
071             * resource path before target/classes.
072             */
073            @Override
074            public URL getResource(String name) {
075                // allow the resource path to take precedence when loading script
076                if(MetaClassLoader.debugLoader!=null) {
077                    URL res = MetaClassLoader.debugLoader.loader.getResource(name);
078                    if(res!=null)
079                        return res;
080                }
081                return super.getResource(name);
082            }
083        };
084    }
085
086    public GroovierJellyScript parse(URL script) throws IOException {
087        // we do the caching on our own, so don't let GroovyClassLoader cache this. Or else
088        // dynamic reloading won't work
089        GroovyCodeSource gcs = new GroovyCodeSource(script);
090        gcs.setCachable(false);
091
092        return new GroovierJellyScript(gcl.parseClass(gcs),script);
093    }
094
095    public GroovierJellyScript parseGSP(URL res) throws IOException, ClassNotFoundException {
096        GroovyCodeSource gcs = new GroovyCodeSource(parser.parse(res), res.toExternalForm(), res.toExternalForm());
097        gcs.setCachable(false);
098
099        return new GroovierJellyScript(gcl.parseClass(gcs),res);
100    }
101}
