024package org.kohsuke.stapler.jelly.groovy;
025
026import groovy.lang.GroovyObject;
027import groovy.lang.GroovyObjectSupport;
028import groovy.xml.QName;
029import org.apache.commons.jelly.XMLOutput;
030import org.codehaus.groovy.runtime.InvokerHelper;
031import org.xml.sax.SAXException;
032
033import java.lang.reflect.InvocationHandler;
034import java.lang.reflect.InvocationTargetException;
035import java.lang.reflect.Method;
036import java.lang.reflect.Proxy;
037
038/**
039 * @author Kohsuke Kawaguchi
040 */
041public class Namespace extends GroovyObjectSupport {
042    private final JellyBuilder builder;
043    private final String nsUri;
044    private final String prefix;
045
046    // note that the mapping from nsUri to TagLibrary
047    // may change depending on the scope, so we can't cache TagLibrary
048
049    Namespace(JellyBuilder builder, String nsUri, String prefix) {
050        this.builder = builder;
051        this.nsUri = nsUri;
052        this.prefix = prefix==null ? "" : prefix;
053    }
054
055    public Object invokeMethod(String localName, Object args) {
056        builder.doInvokeMethod(new QName(nsUri,localName,prefix),args);
057        return null;
058    }
059
060    public void startPrefixMapping(XMLOutput output) throws SAXException {
061        output.startPrefixMapping(prefix,nsUri);
062    }
063
064    public void endPrefixMapping(XMLOutput output) throws SAXException {
065        output.endPrefixMapping(prefix);
066    }
067
068    /**
069     * Creates a type-safe invoker for calling taglibs.
070     */
071    public <T extends TypedTagLibrary> T createInvoker(Class<T> type) {
072        ProxyImpl handler = new ProxyImpl();
073        Object proxy = Proxy.newProxyInstance(type.getClassLoader(), new Class[]{type}, handler);
074        handler.setMetaClass(InvokerHelper.getMetaClass(proxy.getClass()));
075        return type.cast(proxy);
076    }
077
078    private class ProxyImpl extends GroovyObjectSupport implements InvocationHandler {
079        public Object invoke(Object proxy, Method method, Object[] args) throws Throwable {
080            Class<?> decl = method.getDeclaringClass();
081            if (decl==Object.class || decl==GroovyObject.class)
082                try {
083                    return method.invoke(this,args);
084                } catch (InvocationTargetException e) {
085                    throw e.getCause();
086                }
087
088            TagFile a = method.getAnnotation(TagFile.class);
089            String tagName = a!=null ? a.value() : method.getName();
090
091            // invoke methods
092            builder.doInvokeMethod(new QName(nsUri,tagName,prefix),args);
093            return null;
094        }
095
096        /**
097         * Also accepts arbitrary type-unsafe invocations.
098         */
099        @Override
100        public Object invokeMethod(String name, Object args) {
101            return Namespace.this.invokeMethod(name,args);
102        }
103    }
104}
