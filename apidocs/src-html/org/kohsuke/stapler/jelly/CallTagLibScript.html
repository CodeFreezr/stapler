024package org.kohsuke.stapler.jelly;
025
026import org.apache.commons.jelly.JellyContext;
027import org.apache.commons.jelly.JellyException;
028import org.apache.commons.jelly.JellyTagException;
029import org.apache.commons.jelly.Script;
030import org.apache.commons.jelly.XMLOutput;
031import org.apache.commons.jelly.expression.Expression;
032import org.apache.commons.jelly.impl.DynamicTag;
033import org.apache.commons.jelly.impl.ExpressionAttribute;
034import org.apache.commons.jelly.impl.TagScript;
035import org.xml.sax.SAXException;
036
037import java.util.HashMap;
038import java.util.Map;
039
040/**
041 * {@link TagScript} that invokes a {@link Script} as a tag.
042 *
043 * @author Kohsuke Kawaguchi
044 */
045public abstract class CallTagLibScript extends TagScript {
046    /**
047     * Resolves to the definition of the script to call to.
048     */
049    protected abstract Script resolveDefinition(JellyContext context) throws JellyTagException;
050
051    @Override
052    public void run(final JellyContext context, XMLOutput output) throws JellyTagException {
053        // evaluated values of the attributes
054        Map args = new HashMap(attributes.size());
055
056        for (Map.Entry<String, ExpressionAttribute> e : attributes.entrySet()) {
057            Expression expression = e.getValue().exp;
058            args.put(e.getKey(),expression.evaluate(context));
059        }
060
061        // create new context based on current attributes
062        JellyContext newJellyContext = context.newJellyContext(args);
063        newJellyContext.setVariable( "attrs", args );
064
065        // <d:invokeBody> uses this to discover what to invoke
066        newJellyContext.setVariable("org.apache.commons.jelly.body", new Script() {
067            public Script compile() throws JellyException {
068                return this;
069            }
070
071            /**
072             * When &lt;d:invokeBody/> is used to call back into the calling script,
073             * the Jelly name resolution rule is in such that the body is evaluated with
074             * the variable scope of the &lt;d:invokeBody/> caller. This is very different
075             * from a typical closure name resolution mechanism, where the body is evaluated
076             * with the variable scope of where the body was created.
077             *
078             * <p>
079             * More concretely, in Jelly, this often shows up as a problem as inability to
080             * access the "attrs" variable from inside a body, because every {@link DynamicTag}
081             * invocation sets this variable in a new scope.
082             *
083             * <p>
084             * To counter this effect, this class temporarily restores the original "attrs"
085             * when the body is evaluated. This makes the name resolution of 'attrs' work
086             * like what programmers normally expect.
087             *
088             * <p>
089             * The same problem also shows up as a lack of local variables &mdash; when a tag
090             * calls into the body via &lt;d:invokeBody/>, the invoked body will see all the
091             * variables that are defined in the caller, which is again not what a normal programming language
092             * does. But unfortunately, changing this is too pervasive.
093             */
094            public void run(JellyContext nestedContext, XMLOutput output) throws JellyTagException {
095                Map m = nestedContext.getVariables();
096                Object oldAttrs = m.put("attrs",context.getVariable("attrs"));
097                try {
098                    getTagBody().run(nestedContext,output);
099                } finally {
100                    m.put("attrs",oldAttrs);
101                }
102            }
103        });
104        newJellyContext.setVariable("org.apache.commons.jelly.body.scope", context);
105        final Script def = resolveDefinition(newJellyContext);
106
107        if(JellyFacet.TRACE) {
108            try {
109                String source = getSource();
110                String msg = "<" + source+">";
111                output.comment(msg.toCharArray(),0,msg.length());
112                def.run(newJellyContext, output);
113                msg = "</" + source+">";
114                output.comment(msg.toCharArray(),0,msg.length());
115            } catch (SAXException e) {
116                throw new JellyTagException(e);
117            }
118        } else {
119            def.run(newJellyContext, output);
120        }
121    }
122
123    protected String getSource() {
124        return "{jelly:"+getNsUri()+"}:"+getLocalName();
125    }
126}
