024package org.kohsuke.stapler;
025
026import org.kohsuke.stapler.Header.HandlerImpl;
027
028import javax.servlet.ServletException;
029
030import static java.lang.annotation.ElementType.PARAMETER;
031import java.lang.annotation.Retention;
032import static java.lang.annotation.RetentionPolicy.RUNTIME;
033import java.lang.annotation.Target;
034import java.lang.annotation.Documented;
035
036/**
037 * Indicates that this parameter is bound from HTTP header.
038 *
039 * @author Kohsuke Kawaguchi
040 */
041@Retention(RUNTIME)
042@Target(PARAMETER)
043@Documented
044@InjectedParameter(HandlerImpl.class)
045public @interface Header {
046    /**
047     * HTTP header name.
048     */
049    String value();
050
051    /**
052     * If true, request without this header will be rejected.
053     */
054    boolean required() default false;
055
056    class HandlerImpl extends AnnotationHandler<Header> {
057        public Object parse(StaplerRequest request, Header a, Class type, String parameterName) throws ServletException {
058            String name = a.value();
059            if(name.length()==0)    name=parameterName;
060            if(name==null)
061                throw new IllegalArgumentException("Parameter name unavailable neither in the code nor in annotation");
062
063            String value = request.getHeader(name);
064            if(a.required() && value==null)
065                throw new ServletException("Required HTTP header "+name+" is missing");
066
067            return convert(type,value);
068        }
069    }
070}
