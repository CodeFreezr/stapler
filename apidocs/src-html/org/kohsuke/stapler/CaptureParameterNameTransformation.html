024package org.kohsuke.stapler;
025
026import org.codehaus.groovy.ast.ASTNode;
027import org.codehaus.groovy.ast.AnnotatedNode;
028import org.codehaus.groovy.ast.AnnotationNode;
029import org.codehaus.groovy.ast.ClassNode;
030import org.codehaus.groovy.ast.MethodNode;
031import org.codehaus.groovy.ast.Parameter;
032import org.codehaus.groovy.ast.expr.ConstantExpression;
033import org.codehaus.groovy.ast.expr.ListExpression;
034import org.codehaus.groovy.control.SourceUnit;
035import org.codehaus.groovy.transform.ASTTransformation;
036import org.codehaus.groovy.transform.GroovyASTTransformation;
037import org.kohsuke.MetaInfServices;
038
039import java.util.Collections;
040import java.util.List;
041import java.util.Set;
042
043/**
044 * Groovy AST transformation that capture necessary parameter names.
045 *
046 * @author Kohsuke Kawaguchi
047 */
048@MetaInfServices
049@GroovyASTTransformation
050public class CaptureParameterNameTransformation implements ASTTransformation {
051    public void visit(ASTNode[] nodes, SourceUnit source) {
052        handleClasses(source.getAST().getClasses());
053    }
054
055    private void handleClasses(List<ClassNode> classNodes) {
056        for (ClassNode c : classNodes)
057            handleMethods(c.getMethods());
058    }
059
060    // set of annotation class names to capture
061    private static final Set<String> CONSTRUCTOR_ANN = Collections.singleton(DataBoundConstructor.class.getName());
062    private static final Set<String> INJECTED_PARAMETER_ANN = Collections.singleton(InjectedParameter.class.getName());
063
064    private void handleMethods(List<MethodNode> methods) {
065        for (MethodNode m : methods)
066            // copy the array as we'll modify them
067            if(hasAnnotation(m,CONSTRUCTOR_ANN) || hasInjectionAnnotation(m))
068                write(m);
069    }
070
071    private boolean hasInjectionAnnotation(MethodNode m) {
072        for (Parameter p : m.getParameters())
073            if(hasInjectedParameterAnnotation(p))
074                return true;
075        return false;
076    }
077
078
079    private boolean hasAnnotation(AnnotatedNode target, Set<String> annotationTypeNames) {
080        for (AnnotationNode a : target.getAnnotations())
081            if(annotationTypeNames.contains(a.getClassNode().getName()))
082                return true;
083        return false;
084    }
085
086    private boolean hasInjectedParameterAnnotation(Parameter p) {
087        for (AnnotationNode a : p.getAnnotations()) {
088            if (hasAnnotation(a.getClassNode(), INJECTED_PARAMETER_ANN))
089                return true;
090        }
091        return false;
092    }
093
094
095    /**
096     * Captures the parameter names as annotations on the class.
097     */
098    private void write(MethodNode c) {
099        ListExpression v = new ListExpression();
100        for( Parameter p : c.getParameters() )
101            v.addExpression(new ConstantExpression(p.getName()));
102
103        AnnotationNode a = new AnnotationNode(new ClassNode(CapturedParameterNames.class));
104        a.addMember("value",v);
105        c.addAnnotation(a);
106    }
107}
