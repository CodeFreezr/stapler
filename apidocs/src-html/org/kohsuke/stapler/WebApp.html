042/**
043 * Object scoped to the entire webapp. Mostly used for configuring behavior of Stapler.
044 *
045 * <p>
046 * In contrast, {@link Stapler} is a servlet, so there can be multiple instances per webapp.
047 *
048 * @author Kohsuke Kawaguchi
049 * @see WebApp#get(ServletContext)
050 * @see WebApp#getCurrent()  
051 * @see Stapler#getWebApp()
052 */
053public class WebApp {
054    /**
055     * Obtains the {@link WebApp} associated with the given {@link ServletContext}.
056     */
057    public static WebApp get(ServletContext context) {
058        Object o = context.getAttribute(WebApp.class.getName());
059        if(o==null) {
060            synchronized (WebApp.class) {
061                o = context.getAttribute(WebApp.class.getName());
062                if(o==null) {
063                    o = new WebApp(context);
064                    context.setAttribute(WebApp.class.getName(),o);
065                }
066            }
067        }
068        return (WebApp)o;
069    }
070
071    /**
072     * {@link ServletContext} for this webapp.
073     */
074    public final ServletContext context;
075
076    /**
077     * Duck-type wrappers for the given class.
078     */
079    public final Map<Class,Class[]> wrappers = new HashMap<Class,Class[]>();
080
081    /**
082     * MIME type -> encoding map that determines how static contents in the war file is served.
083     */
084    public final Map<String,String> defaultEncodingForStaticResources = new HashMap<String,String>();
085
086    /**
087     * Activated facets.
088     *
089     * TODO: is this really mutable?
090     */
091    public final List<Facet> facets = new Vector<Facet>();
092
093    /**
094     * Global {@link BindInterceptor}s.
095     *
096     * These are consulted after {@link StaplerRequest#getBindInterceptor()} is consulted.
097     * Global bind interceptors are useful to register webapp-wide conversion logic local to the application.
098     * @since 1.220
099     */
100    public final List<BindInterceptor> bindInterceptors = new CopyOnWriteArrayList<BindInterceptor>();
101
102    /**
103     * MIME type mapping from extensions (like "txt" or "jpg") to MIME types ("foo/bar").
104     *
105     * This overrides whatever mappings given in the servlet as far as stapler is concerned.
106     * This is case insensitive, and should be normalized to lower case.
107     */
108    public final Map<String,String> mimeTypes = new Hashtable<String,String>();
109
110    private volatile ClassLoader classLoader;
111
112    /**
113     * All {@link MetaClass}es.
114     *
115     * Note that this permanently holds a strong reference to its key, i.e. is a memory leak.
116     */
117    private final Map<Klass<?>,MetaClass> classMap = new HashMap<Klass<?>,MetaClass>();
118
119    /**
120     * Handles objects that are exported.
121     */
122    public final BoundObjectTable boundObjectTable = new BoundObjectTable();
123
124    private final CopyOnWriteArrayList<HttpResponseRenderer> responseRenderers = new CopyOnWriteArrayList<HttpResponseRenderer>();
125
126    private CrumbIssuer crumbIssuer = CrumbIssuer.DEFAULT;
127
128    /**
129     * Provides access to {@link Stapler} servlet instances. This is useful
130     * for sending a request over to stapler from a context outside Stapler,
131     * such as in {@link Filter}.
132     *
133     * Keyed by {@link ServletConfig#getServletName()}.
134     */
135    private final ConcurrentMap<String,Stapler> servlets = new ConcurrentHashMap<String,Stapler>();
136
137    public WebApp(ServletContext context) {
138        this.context = context;
139        // TODO: allow classloader to be given?
140        facets.addAll(Facet.discoverExtensions(Facet.class, Thread.currentThread().getContextClassLoader(), getClass().getClassLoader()));
141        responseRenderers.add(new HttpResponseRenderer.Default());
142    }
143
144    /**
145     * Returns the 'app' object, which is the user-specified object that
146     * sits at the root of the URL hierarchy and handles the request to '/'.
147     */
148    public Object getApp() {
149        return context.getAttribute("app");
150    }
151
152    public void setApp(Object app) {
153        context.setAttribute("app",app);
154    }
155
156    public CrumbIssuer getCrumbIssuer() {
157        return crumbIssuer;
158    }
159
160    public void setCrumbIssuer(CrumbIssuer crumbIssuer) {
161        this.crumbIssuer = crumbIssuer;
162    }
163
164    public CopyOnWriteArrayList<HttpResponseRenderer> getResponseRenderers() {
165        return responseRenderers;
166    }
167
168    public ClassLoader getClassLoader() {
169        ClassLoader cl = classLoader;
170        if(cl==null)
171            cl = Thread.currentThread().getContextClassLoader();
172        if(cl==null)
173            cl = Stapler.class.getClassLoader();
174        return cl;
175    }
176
177    /**
178     * If the facet of the given type exists, return it. Otherwise null.
179     */
180    public <T extends Facet> T getFacet(Class<T> type) {
181        for (Facet f : facets)
182            if(type==f.getClass())
183                return type.cast(f);
184        return null;
185    }
186
187    /**
188     * Sets the classloader used by {@link StaplerRequest#bindJSON(Class, JSONObject)} and its sibling methods.
189     */
190    public void setClassLoader(ClassLoader classLoader) {
191        this.classLoader = classLoader;
192    }
193
194    public MetaClass getMetaClass(Class c) {
195        return getMetaClass(Klass.java(c));
196    }
197    
198    public MetaClass getMetaClass(Klass<?> c) {
199        if(c==null)     return null;
200        synchronized(classMap) {
201            MetaClass mc = classMap.get(c);
202            if(mc==null) {
203                mc = new MetaClass(this,c);
204                classMap.put(c,mc);
205            }
206            return mc;
207        }
208    }
209
210    /**
211     * Obtains a {@link MetaClass} that represents the type of the given object.
212     *
213     * <p>
214     * This code consults all facets to handle scripting language objects correctly.
215     */
216    public MetaClass getMetaClass(Object o) {
217        return getMetaClass(getKlass(o));
218    }
219
220    public Klass<?> getKlass(Object o) {
221        for (Facet f : facets) {
222            Klass<?> k = f.getKlass(o);
223            if (k!=null)
224                return k;
225        }
226        return Klass.java(o.getClass());
227    }
228    
229    /**
230     * Convenience maintenance method to clear all the cached scripts for the given tearoff type.
231     *
232     * <p>
233     * This is useful when you want to have the scripts reloaded into the live system without
234     * the performance penalty of {@link MetaClass#NO_CACHE}.
235     *
236     * @see MetaClass#NO_CACHE
237     */
238    public void clearScripts(Class<? extends AbstractTearOff> clazz) {
239        synchronized (classMap) {
240            for (MetaClass v : classMap.values()) {
241                AbstractTearOff t = v.getTearOff(clazz);
242                if (t!=null)
243                    t.clearScripts();
244            }
245        }
246    }
247
248    void addStaplerServlet(String servletName, Stapler servlet) {
249        if (servletName==null)  servletName=""; // be defensive
250        servlets.put(servletName,servlet);
251    }
252
253    /**
254     * Gets a reference to some {@link Stapler} servlet in this webapp.
255     *
256     * <p>
257     * Most Stapler webapps will have one &lt;servlet> entry in web.xml
258     * and if that's the case, that'd be returned. In a fully general case,
259     * a webapp can have multiple servlets and more than one of them can be
260     * {@link Stapler}. This method returns one of those. Which one gets
261     * returned is unspecified.
262     *
263     * <p>
264     * This method is useful if you are in a {@link Filter} and using
265     * Stapler to handle the current request. For example,
266     *
267     * <pre>
268     * WebApp.get(servletContext).getSomeStapler().invoke(
269     *     request,response,
270     *     someJavaObject,
271     *     "/path/to/dispatch/request");
272     * </pre>
273     */
274    public Stapler getSomeStapler() {
275        return servlets.values().iterator().next();
276    }
277
278    /**
279     * Gets the current {@link WebApp} that the calling thread is associated with.
280     */
281    public static WebApp getCurrent() {
282        return Stapler.getCurrent().getWebApp();
283    }
