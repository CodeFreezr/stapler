001package org.kohsuke.stapler.compression;
002
003import javax.servlet.ServletOutputStream;
004import java.io.IOException;
005import java.io.OutputStream;
006
007/**
008 * {@link ServletOutputStream} that writes to the specified output stream.
009 *
010 * @author Kohsuke Kawaguchi
011 */
012public class FilterServletOutputStream extends ServletOutputStream {
013    private final OutputStream out;
014
015    public FilterServletOutputStream(OutputStream out) {
016        this.out = out;
017    }
018
019    @Override
020    public void write(int b) throws IOException {
021        out.write(b);
022    }
023
024    @Override
025    public void write(byte[] b) throws IOException {
026        out.write(b);
027    }
028
029    @Override
030    public void write(byte[] b, int off, int len) throws IOException {
031        out.write(b, off, len);
032    }
033
034    @Override
035    public void close() throws IOException {
036        out.close();
037    }
038
039    @Override
040    public void flush() throws IOException {
041        out.flush();
042    }
043}
