001package org.kohsuke.stapler.compression;
002
003import javax.servlet.ServletOutputStream;
004import javax.servlet.http.HttpServletResponse;
005import javax.servlet.http.HttpServletResponseWrapper;
006import java.io.IOException;
007import java.io.OutputStreamWriter;
008import java.io.PrintWriter;
009import com.jcraft.jzlib.GZIPOutputStream;
010
011/**
012 * {@link HttpServletResponse} that recognizes Content-Encoding: gzip in the response header
013 * and acts accoringly.
014 *
015 * @author Kohsuke Kawaguchi
016 */
017public class CompressionServletResponse extends HttpServletResponseWrapper {
018    /**
019     * If not null, we are compressing the stream.
020     */
021    private ServletOutputStream stream;
022    private PrintWriter writer;
023
024    public CompressionServletResponse(HttpServletResponse response) {
025        super(response);
026    }
027
028    @Override
029    public void setHeader(String name, String value) {
030        super.setHeader(name, value);
031        activateCompressionIfNecessary(name,value);
032    }
033
034    @Override
035    public void addHeader(String name, String value) {
036        super.addHeader(name, value);
037        activateCompressionIfNecessary(name, value);
038    }
039
040    private void activateCompressionIfNecessary(String name, String value) {
041        try {
042            if (name.equals("Content-Encoding") && value.equals("gzip")) {
043                if (stream==null)
044                    stream = new FilterServletOutputStream(new GZIPOutputStream(super.getOutputStream()));
045            }
046        } catch (IOException e) {
047            throw new RuntimeException(e);
048        }
049    }
050
051    @Override
052    public PrintWriter getWriter() throws IOException {
053        if (writer!=null)   return writer;
054        if (stream!=null) {
055            writer = new PrintWriter(new OutputStreamWriter(stream,getCharacterEncoding()));
056            return writer;
057        }
058        return super.getWriter();
059    }
060
061    @Override
062    public ServletOutputStream getOutputStream() throws IOException {
063        if (stream!=null)   return stream;
064        return super.getOutputStream();
065    }
066}
