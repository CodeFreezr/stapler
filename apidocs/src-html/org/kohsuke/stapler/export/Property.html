024package org.kohsuke.stapler.export;
025
026import org.kohsuke.stapler.export.TreePruner.ByDepth;
027
028import java.io.IOException;
029import java.lang.reflect.InvocationTargetException;
030import java.lang.reflect.Type;
031import java.lang.reflect.Array;
032import java.net.URL;
033import java.util.Arrays;
034import java.util.Calendar;
035import java.util.Collection;
036import java.util.Collections;
037import java.util.Date;
038import java.util.HashSet;
039import java.util.Map;
040import java.util.Set;
041
042/**
043 * Exposes one {@link Exported exposed property} of {@link ExportedBean} to
044 * {@link DataWriter}.
045 *  
046 * @author Kohsuke Kawaguchi
047 */
048public abstract class Property implements Comparable<Property> {
049    /**
050     * Name of the property.
051     */
052    public final String name;
053    final ModelBuilder owner;
054    /**
055     * Visibility depth level of this property.
056     *
057     * @see Exported#visibility()
058     */
059    public final int visibility;
060
061    /**
062     * Model to which this property belongs to.
063     * Never null.
064     */
065    public final Model parent;
066
067    /**
068     * @see Exported#inline()
069     */
070    public final boolean inline;
071
072    private String[] verboseMap;
073
074    Property(Model parent, String name, Exported exported) {
075        this.parent = parent;
076        this.owner = parent.parent;
077        this.name = exported.name().length()>1 ? exported.name() : name;
078        int v = exported.visibility();
079        if(v==0)
080            v = parent.defaultVisibility;
081        this.visibility = v;
082        this.inline = exported.inline();
083        String[] s = exported.verboseMap().split("/");
084        if (s.length<2)
085            this.verboseMap = null;
086        else
087            this.verboseMap = s;
088    }
089
090    public int compareTo(Property that) {
091        return this.name.compareTo(that.name);
092    }
093
094    public abstract Type getGenericType();
095    public abstract Class getType();
096
097    /**
098     * Gets the associated javadoc, if any, or null.
099     */
100    public abstract String getJavadoc();
101
102    /**
103     * Writes one property of the given object to {@link DataWriter}.
104     *
105     * @param pruner
106     *      Determines how to prune the object graph tree.
107     */
108    public void writeTo(Object object, TreePruner pruner, DataWriter writer) throws IOException {
109        TreePruner child = pruner.accept(object, this);
110        if (child==null)        return;
111
112        try {
113            writer.name(name);
114            writeValue(getValue(object),child,writer);
115        } catch (IllegalAccessException e) {
116            IOException x = new IOException("Failed to write " + name);
117            x.initCause(e);
118            throw x;
119        } catch (InvocationTargetException e) {
120            IOException x = new IOException("Failed to write " + name);
121            x.initCause(e);
122            throw x;
123        }
124    }
125
126    /**
127     * @deprecated as of 1.139
128     */
129    public void writeTo(Object object, int depth, DataWriter writer) throws IOException {
130        writeTo(object,new ByDepth(depth),writer);
131    }
132
133    /**
134     * Writes one value of the property to {@link DataWriter}.
135     */
136    private void writeValue(Object value, TreePruner pruner, DataWriter writer) throws IOException {
137        writeValue(value,pruner,writer,false);
138    }
139
140    /**
141     * Writes one value of the property to {@link DataWriter}.
142     */
143    private void writeValue(Object value, TreePruner pruner, DataWriter writer, boolean skipIfFail) throws IOException {
144        if(value==null) {
145            writer.valueNull();
146            return;
147        }
148
149        if(value instanceof CustomExportedBean) {
150            writeValue(((CustomExportedBean)value).toExportedObject(),pruner,writer);
151            return;
152        }
153
154        Class c = value.getClass();
155
156        if(STRING_TYPES.contains(c)) {
157            writer.value(value.toString());
158            return;
159        }
160        if(PRIMITIVE_TYPES.contains(c)) {
161            writer.valuePrimitive(value);
162            return;
163        }
164        if(c.getComponentType()!=null) { // array
165            writer.startArray();
166            if (value instanceof Object[]) {
167                // typical case
168                for (Object item : (Object[]) value)
169                    writeValue(item,pruner,writer,true);
170            } else {
171                // more generic case
172                int len = Array.getLength(value);
173                for (int i=0; i<len; i++)
174                    writeValue(Array.get(value,i),pruner,writer,true);
175            }
176            writer.endArray();
177            return;
178        }
179        if(value instanceof Collection) {
180            writer.startArray();
181            for (Object item : (Collection) value)
182                writeValue(item,pruner,writer,true);
183            writer.endArray();
184            return;
185        }
186        if(value instanceof Map) {
187            if (verboseMap!=null) {// verbose form
188                writer.startArray();
189                for (Map.Entry e : ((Map<?,?>) value).entrySet()) {
190                    writer.startObject();
191                    writer.name(verboseMap[0]);
192                    writeValue(e.getKey(),pruner,writer);
193                    writer.name(verboseMap[1]);
194                    writeValue(e.getValue(),pruner,writer);
195                    writer.endObject();
196                }
197                writer.endArray();
198            } else {// compact form
199                writer.startObject();
200                for (Map.Entry e : ((Map<?,?>) value).entrySet()) {
201                    writer.name(e.getKey().toString());
202                    writeValue(e.getValue(),pruner,writer);
203                }
204                writer.endObject();
205            }
206            return;
207        }
208        if(value instanceof Date) {
209            writer.valuePrimitive(((Date) value).getTime());
210            return;
211        }
212        if(value instanceof Calendar) {
213            writer.valuePrimitive(((Calendar) value).getTimeInMillis());
214            return;
215        }
216        if(value instanceof Enum) {
217            writer.value(value.toString());
218            return;
219        }
220
221        // otherwise handle it as a bean
222        writer.startObject();
223        Model model = null;
224        try {
225            model = owner.get(c);
226        } catch (NotExportableException e) {
227            if(!skipIfFail)
228                throw e;
229            // otherwise ignore this error by writing empty object
230        }
231        if(model!=null)
232            model.writeNestedObjectTo(value, pruner, writer, Collections.<String>emptySet());
233        writer.endObject();
234    }
235
236    /**
237     * Gets the value of this property from the bean.
238     */
239    protected abstract Object getValue(Object bean) throws IllegalAccessException, InvocationTargetException;
240
241    /*package*/ static final Set<Class> STRING_TYPES = new HashSet<Class>(Arrays.asList(
242        String.class,
243        URL.class
244    ));
245
246    /*package*/ static final Set<Class> PRIMITIVE_TYPES = new HashSet<Class>(Arrays.asList(
247        Integer.class,
248        Long.class,
249        Boolean.class,
250        Short.class,
251        Character.class,
252        Float.class,
253        Double.class
254    ));
