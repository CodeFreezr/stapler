001package org.kohsuke.stapler.export;
002
003import org.jvnet.tiger_types.Types;
004
005import javax.annotation.Nullable;
006import java.lang.reflect.Type;
007
008/**
009 * Setting that controls how the '_class' attribute will be produced in the output.
010 *
011 * <p>
012 * Three basic constants are defined:
013 *
014 * <dl>
015 *     <dt>{@link #NONE}
016 *     <dd>No type information will show up in the output whatsoever
017 *
018 *     <dt>{@link #ALWAYS}
019 *     <dd>Every object gets the type information explicitly written out
020 *
021 *     <dt>{@link #IF_NEEDED}
022 *     <dd>Type information will be produced only when it is necessary,
023 *         for example when the declared type and the actual type differ.
024 * </dl>
025 *
026 * <p>
027 * In addition, {@link #simple()} can be used to produce just the simple name of the class,
028 * whereas by default {@linkplain Class#getName() full class name} will be printed.
029 *
030 * @author Kohsuke Kawaguchi
031 * @see ExportConfig#withClassAttribute(ClassAttributeBehaviour)
032 */
033public abstract class ClassAttributeBehaviour {
034    private final String name;
035
036    // no subtyping outside this package
037    private ClassAttributeBehaviour(String name) {
038        this.name = name;
039    }
040
041    @Override
042    public String toString() {
043        return super.toString()+'['+name+']';
044    }
045
046    /**
047     * Determines if this type attribute should be written.
048     *
049     * @return
050     *      non-null if the type should be written, null if it shouldn't.
051     * @see DataWriter#type(Type, Class)
052     */
053    abstract @Nullable Class map(@Nullable Type expected, @Nullable Class actual);
054
055    String print(Type expected, Class actual) {
056        return print(map(expected,actual));
057    }
058
059    protected String print(Class t) {
060        return t==null ? null : t.getName();
061    }
062
063    public static final ClassAttributeBehaviour NONE = new ClassAttributeBehaviour("NONE") {
064        @Override
065        Class map(Type expected, Class actual) {
066            return null;
067        }
068    };
069
070    public static final ClassAttributeBehaviour ALWAYS = new ClassAttributeBehaviour("ALWAYS") {
071        @Override
072        Class map(Type expected, Class actual) {
073            return actual;
074        }
075    };
076
077    public static final ClassAttributeBehaviour IF_NEEDED = new ClassAttributeBehaviour("IF_NEEDED") {
078        @Override
079        Class map(Type expected, Class actual) {
080            if (actual==null)
081                return null;    // nothing to write
082            if (expected==actual)
083                return null;    // fast pass when we don't need it
084            if (expected==null)
085                return actual;  // we need to print it
086            if (Types.erasure(expected)==actual)
087                return null;    // slow pass
088            return actual;
089        }
090    };
091
092    public ClassAttributeBehaviour simple() {
093        final ClassAttributeBehaviour outer = this;
094        return new ClassAttributeBehaviour(this.name+"+simple") {
095            @Override
096            Class map(Type expected, Class actual) {
097                return outer.map(expected,actual);
098            }
099
100            @Override
101            protected String print(Class t) {
102                return t==null ? null : t.getSimpleName();
103            }
104        };
105    }
106}
