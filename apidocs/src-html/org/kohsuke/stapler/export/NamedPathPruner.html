024package org.kohsuke.stapler.export;
025
026import java.util.Map;
027import java.util.TreeMap;
028
029/**
030 * Tree pruner which operates according to a textual description of what tree leaves should be included.
031 */
032public final class NamedPathPruner extends TreePruner {
033    
034    static class Tree {
035        final Map<String,Tree> children = new TreeMap<String,Tree>();
036        public @Override String toString() {return children.toString();}
037    }
038
039    // Simple recursive descent parser:
040    static Tree parse(String spec) throws IllegalArgumentException {
041        Reader r = new Reader(spec);
042        Tree t = new Tree();
043        list(r, t);
044        r.expect(Token.EOF);
045        return t;
046    }
047    private static void list(Reader r, Tree t) throws IllegalArgumentException {
048        node(r, t);
049        if (r.accept(Token.COMMA)) {
050            list(r, t);
051        }
052    }
053    private static void node(Reader r, Tree t) throws IllegalArgumentException {
054        Object actual = r.peek();
055        if (actual instanceof Token) {
056            throw new IllegalArgumentException("expected name at " + r.pos);
057        }
058        r.advance();
059        Tree subtree = new Tree();
060        t.children.put((String) actual, subtree);
061        if (r.accept(Token.LBRACE)) {
062            list(r, subtree);
063            r.expect(Token.RBRACE);
064        }
065    }
066    private enum Token {COMMA, LBRACE, RBRACE, EOF}
067    private static class Reader {
068        private final String text;
069        int pos, next;
070        Reader(String text) {
071            this.text = text;
072            pos = 0;
073        }
074        Object peek() {
075            if (pos == text.length()) {
076                return Token.EOF;
077            }
078            switch (text.charAt(pos)) {
079            case ',':
080                next = pos + 1;
081                return Token.COMMA;
082            case '[':
083                next = pos + 1;
084                return Token.LBRACE;
085            case ']':
086                next = pos + 1;
087                return Token.RBRACE;
088            default:
089                next = text.length();
090                for (char c : new char[] {',', '[', ']'}) {
091                    int x = text.indexOf(c, pos);
092                    if (x != -1 && x < next) {
093                        next = x;
094                    }
095                }
096                return text.substring(pos, next);
097            }
098        }
099        void advance() {
100            pos = next;
101        }
102        void expect(Token tok) throws IllegalArgumentException {
103            Object actual = peek();
104            if (actual != tok) {
105                throw new IllegalArgumentException("expected " + tok + " at " + pos);
106            }
107            advance();
108        }
109        boolean accept(Token tok) {
110            if (peek() == tok) {
111                advance();
112                return true;
113            } else {
114                return false;
115            }
116        }
117    }
118    
119    private final Tree tree;
120
121    /**
122     * Constructs a pruner by parsing a textual specification.
123     * This lists the properties which should be included at each level of the hierarchy.
124     * Properties are separated by commas and nested objects are inside square braces.
125     * For example, {@code a,b[c,d]} will emit the top-level property {@code a} but
126     * none of its children, and the top-level property {@code b} and only those
127     * of its children named {@code c} and {@code d}.
128     * @param spec textual specification of tree
129     * @throws IllegalArgumentException if the syntax is incorrect
130     */
131    public NamedPathPruner(String spec) throws IllegalArgumentException {
132        this(parse(spec));
133    }
134    
135    private NamedPathPruner(Tree tree) {
136        this.tree = tree;
137    }
138
139    public @Override TreePruner accept(Object node, Property prop) {
140        Tree subtree = tree.children.get(prop.name);
141        if (subtree==null)  subtree=tree.children.get("*");
142        return subtree != null ? new NamedPathPruner(subtree) : null;
143    }
144
145}
