024package org.kohsuke.stapler.export;
025
026import java.util.Map;
027import java.util.TreeMap;
028import java.util.regex.Pattern;
029
030/**
031 * Tree pruner which operates according to a textual description of what tree leaves should be included.
032 */
033public final class NamedPathPruner extends TreePruner {
034    
035    static class Tree {
036        final Map<String,Tree> children = new TreeMap<String,Tree>();
037        Range range = Range.ALL;
038        public @Override String toString() {return children.toString();}
039    }
040
041    // Simple recursive descent parser:
042    static Tree parse(String spec) throws IllegalArgumentException {
043        Reader r = new Reader(spec);
044        Tree t = new Tree();
045        list(r, t);
046        r.expect(Token.EOF);
047        return t;
048    }
049    private static void list(Reader r, Tree t) throws IllegalArgumentException {
050        node(r, t);
051        if (r.accept(Token.COMMA)) {
052            list(r, t);
053        }
054    }
055    private static void node(Reader r, Tree t) throws IllegalArgumentException {
056        Object actual = r.peek();
057        if (actual instanceof Token) {
058            throw new IllegalArgumentException("expected name at " + r.pos);
059        }
060        r.advance();
061        Tree subtree = new Tree();
062        t.children.put((String) actual, subtree);
063        if (r.accept(Token.LBRACE)) {
064            list(r, subtree);
065            r.expect(Token.RBRACE);
066        }
067        if (r.accept(Token.QLBRACE)) {
068            subtree.range = parseRange(r);
069            r.expect(Token.QRBRACE);
070        }
071    }
072    static Range parseRange(Reader r) {
073        int min;
074
075        if (r.accept(Token.COMMA)) {
076            return new Range(0, r.expectNumber());  // {,M}
077        } else {
078            min = r.expectNumber();
079            if (r.peek()==Token.QRBRACE)
080                return new Range(min,min+1);        // {N}
081
082            r.expect(Token.COMMA);
083            if (r.peek()==Token.QRBRACE)
084                return new Range(min,Integer.MAX_VALUE);    // {N,}
085            else
086                return new Range(min,r.expectNumber());     // {N,M}
087        }
088    }
089
090    private enum Token {COMMA /* , */, LBRACE /* [ */, RBRACE /* ] */, QLBRACE /* { */, QRBRACE /* } */, EOF}
091    private static class Reader {
092        private final String text;
093        int pos, next;
094        Reader(String text) {
095            this.text = text;
096            pos = 0;
097        }
098        Object peek() {
099            if (pos == text.length()) {
100                return Token.EOF;
101            }
102            switch (text.charAt(pos)) {
103            case ',':
104                next = pos + 1;
105                return Token.COMMA;
106            case '[':
107                next = pos + 1;
108                return Token.LBRACE;
109            case ']':
110                next = pos + 1;
111                return Token.RBRACE;
112            case '{':
113                next = pos + 1;
114                return Token.QLBRACE;
115            case '}':
116                next = pos + 1;
117                return Token.QRBRACE;
118            default:
119                next = text.length();
120                for (char c : new char[] {',', '[', ']', '{', '}'}) {
121                    int x = text.indexOf(c, pos);
122                    if (x != -1 && x < next) {
123                        next = x;
124                    }
125                }
126                return text.substring(pos, next);
127            }
128        }
129        void advance() {
130            pos = next;
131        }
132        void expect(Token tok) throws IllegalArgumentException {
133            Object actual = peek();
134            if (actual != tok) {
135                throw new IllegalArgumentException("expected " + tok + " at " + pos);
136            }
137            advance();
138        }
139        int expectNumber() {
140            Object t = peek();
141            if (!(t instanceof String) || !Pattern.matches("[0-9]+$", (String) t)) {
142                throw new IllegalArgumentException("expected number at " + pos);
143            }
144            advance();
145            return Integer.parseInt((String) t);
146        }
147        boolean accept(Token tok) {
148            if (peek() == tok) {
149                advance();
150                return true;
151            } else {
152                return false;
153            }
154        }
155    }
156    
157    private final Tree tree;
158
159    /**
160     * Constructs a pruner by parsing a textual specification.
161     * This lists the properties which should be included at each level of the hierarchy.
162     * Properties are separated by commas and nested objects are inside square braces.
163     * For example, {@code a,b[c,d]} will emit the top-level property {@code a} but
164     * none of its children, and the top-level property {@code b} and only those
165     * of its children named {@code c} and {@code d}.
166     * @param spec textual specification of tree
167     * @throws IllegalArgumentException if the syntax is incorrect
168     */
169    public NamedPathPruner(String spec) throws IllegalArgumentException {
170        this(parse(spec));
171    }
172    
173    private NamedPathPruner(Tree tree) {
174        this.tree = tree;
175    }
176
177    public @Override TreePruner accept(Object node, Property prop) {
178        Tree subtree = tree.children.get(prop.name);
179        if (subtree==null)  subtree=tree.children.get("*");
180        return subtree != null ? new NamedPathPruner(subtree) : null;
181    }
182
183    public @Override Range getRange() {
184        return tree.range;
185    }
186
187}
