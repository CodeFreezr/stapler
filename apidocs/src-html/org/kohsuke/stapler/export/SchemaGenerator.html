024package org.kohsuke.stapler.export;
025
026import com.sun.xml.txw2.TXW;
027import com.sun.xml.txw2.output.ResultFactory;
028import java.beans.Introspector;
029import org.kohsuke.stapler.export.XSD.ComplexType;
030import org.kohsuke.stapler.export.XSD.ContentModel;
031import org.kohsuke.stapler.export.XSD.Element;
032import org.kohsuke.stapler.export.XSD.Schema;
033import org.kohsuke.stapler.export.XSD.Annotated;
034
035import javax.xml.namespace.QName;
036import javax.xml.transform.Result;
037import java.util.Calendar;
038import java.util.Collection;
039import java.util.HashSet;
040import java.util.Map;
041import java.util.Set;
042import java.util.Stack;
043
044/**
045 * Generates XML Schema that describes the XML representation of exported beans.
046 *
047 * @author Kohsuke Kawaguchi
048 */
049public class SchemaGenerator {
050    private final Stack<Model> queue = new Stack<Model>();
051    private final Set<Model> written = new HashSet<Model>();
052    /**
053     * Enumerations to be generated.
054     */
055    private final Set<Class> enums = new HashSet<Class>();
056
057    private final ModelBuilder builder;
058    private final Model<?> top;
059
060    public SchemaGenerator(Model<?> m) {
061        this.builder = m.parent;
062        this.top = m;
063    }
064
065    /**
066     * Generates the complete schema to the specified result.
067     */
068    public void generateSchema(Result r) {
069        Schema s = TXW.create(Schema.class, ResultFactory.createSerializer(r));
070        s._namespace(XSD.URI,"xsd");
071
072        queue.clear();
073        written.clear();
074
075        // element decl for the root element
076        s.element().name(Introspector.decapitalize(top.type.getSimpleName())).type(getXmlTypeName(top.type));
077
078        // write all beans
079        while(!queue.isEmpty())
080            writeBean(s, (Model<?>) queue.pop());
081
082        // then enums
083        for (Class e : enums)
084            writeEnum(s,e);
085
086        s.commit();
087    }
088
089    private void writeEnum(Schema s, Class e) {
090        XSD.Restriction facets = s.simpleType().name(getXmlToken(e)).restriction().base(XSD.Types.STRING);
091        for (Object constant : e.getEnumConstants()) {
092            facets.enumeration().value(constant.toString());
093        }
094    }
095
096    private void writeBean(Schema s, Model<?> m) {
097        ComplexType ct = s.complexType().name(getXmlToken(m.type));
098        final ContentModel cm;
099
100        if(m.superModel==null)
101            cm = ct.sequence();
102        else {
103            addToQueue(m.superModel);
104            cm = ct.complexContent().extension().base(getXmlTypeName(m.superModel.type)).sequence();
105        }
106
107        for (Property p : m.getProperties()) {
108            Class t = p.getType();
109            final boolean isCollection;
110            final Class itemType;
111            if(t.isArray()) {
112                isCollection = true;
113                itemType = t.getComponentType();
114            } else
115            if(Collection.class.isAssignableFrom(t)) {
116                isCollection = true;
117                itemType = TypeUtil.erasure(
118                    TypeUtil.getTypeArgument(TypeUtil.getBaseClass(p.getGenericType(),Collection.class),0));
119            } else {
120                isCollection = false;
121                itemType = t;
122            }
123
124            Element e = cm.element()
125                .name(isCollection?XMLDataWriter.toSingular(p.name):p.name)
126                .type(getXmlTypeName(itemType));
127
128            if(!t.isPrimitive())
129                e.minOccurs(0);
130            if(isCollection)
131                e.maxOccurs("unbounded");
132
133            annotate(e,p.getJavadoc());
134        }
135
136        if (m.superModel==null)
137            ct.attribute().name(DataWriter.CLASS_PROPERTY_NAME).type(XSD.Types.STRING).use("optional");
138    }
139
140    /**
141     * Annotates the schema element by javadoc, if that exists.
142     */
143    private void annotate(Annotated e, String javadoc) {
144        if(javadoc==null)   return;
145        e.annotation().documentation(javadoc);
146    }
147
148    /**
149     * Adds the schema for the XML representation of the given class.
150     */
151    public void add(Class<?> c) {
152        addToQueue(builder.get(c));
153    }
154
155    private void addToQueue(Model m) {
156        if (written.add(m))
157            queue.push(m);
158    }
159
160    public QName getXmlTypeName(Class<?> t) {
161        if(Property.STRING_TYPES.contains(t))       return XSD.Types.STRING;
162        if(t==Boolean.class || t==boolean.class)    return XSD.Types.BOOLEAN;
163        if(t==Integer.class || t==int.class)        return XSD.Types.INT;
164        if(t==Long.class || t==long.class)          return XSD.Types.LONG;
165        if(Map.class.isAssignableFrom(t))           return XSD.Types.ANYTYPE;
166        if(Calendar.class.isAssignableFrom(t))      return XSD.Types.LONG;
167        if(Enum.class.isAssignableFrom(t)) {
168            enums.add(t);
169            return new QName(getXmlToken(t));
170        }
171
172        // otherwise bean
173        try {
174            addToQueue(builder.get(t));
175        } catch (IllegalArgumentException e) {
176            // not an exposed bean by itself
177            return XSD.Types.ANYTYPE;
178        }
179        
180        return new QName(getXmlToken(t));
181    }
182
183    private String getXmlToken(Class<?> t) {
184        return t.getName().replace('$','-');
185    }
186
187    private String getXmlElementName(Class<?> type) {
188        return getXmlToken(type);
189    }
190}
