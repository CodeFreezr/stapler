001package org.kohsuke.stapler.export;
002
003import java.io.IOException;
004import java.lang.reflect.InvocationTargetException;
005
006/**
007 * Allows caller to intercept exporting of properties.
008 *
009 * Implementation can choose to ignore properties in case of failure during serialization.
010 *
011 * @author Vivek Pandey
012 * @author James Dumay
013 */
014public abstract class ExportInterceptor {
015    /**
016     * Constant to tell if return of {@link ExportInterceptor#getValue(Property, Object, ExportConfig)} should be skipped.
017     *
018     * Constant to skip serializaing a property in case of error
019     */
020    public static final Object SKIP = new Object();
021
022    /**
023     * Subclasses must call {@link Property#getValue(Object)}  to retrieve the property.
024     *
025     * If the subclass decides the value can be included in the request return the value
026     * otherwise, return {@link #SKIP}  to skip the property.
027     *
028     * @param property to get the value from model object
029     * @param model object with this property
030     * @return the value of the property, if {@link #SKIP} is returned, this property will be skipped
031     * @throws IOException if there was a problem with serialization that should prevent
032     *         the serialization from proceeding
033     * @see Exported#skipNull()
034     */
035    public abstract Object getValue(Property property, Object model, ExportConfig config) throws IOException;
036
037    public static final ExportInterceptor DEFAULT = new ExportInterceptor() {
038        @Override
039        public Object getValue(Property property, Object model, ExportConfig config) throws IOException {
040            try {
041                return property.getValue(model);
042            } catch (IllegalAccessException | InvocationTargetException e) {
043                if(config.isSkipIfFail()) {
044                    return SKIP;
045                }
046                throw new IOException("Failed to write " + property.name + ":" + e.getMessage(), e);
047            }
048        }
049    };
050}
