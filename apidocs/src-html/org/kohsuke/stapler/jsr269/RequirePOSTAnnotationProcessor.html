027package org.kohsuke.stapler.jsr269;
028
029import java.util.Set;
030import javax.annotation.processing.Processor;
031import javax.annotation.processing.RoundEnvironment;
032import javax.annotation.processing.SupportedAnnotationTypes;
033import javax.lang.model.SourceVersion;
034import javax.lang.model.element.Element;
035import javax.lang.model.element.ElementKind;
036import javax.lang.model.element.ExecutableElement;
037import javax.lang.model.element.Modifier;
038import javax.lang.model.element.TypeElement;
039import javax.lang.model.type.TypeMirror;
040import javax.tools.Diagnostic;
041import org.kohsuke.MetaInfServices;
042import org.kohsuke.stapler.interceptor.RequirePOST;
043
044@SupportedAnnotationTypes("org.kohsuke.stapler.interceptor.RequirePOST")
045@MetaInfServices(Processor.class)
046public class RequirePOSTAnnotationProcessor extends AbstractProcessorImpl {
047
048    @Override public boolean process(Set<? extends TypeElement> annotations, RoundEnvironment roundEnv) {
049        for (Element e : roundEnv.getElementsAnnotatedWith(RequirePOST.class)) {
050            if (e.getKind() != ElementKind.METHOD) {
051                continue; // defensive against broken code
052            }
053            ExecutableElement method = (ExecutableElement) e;
054            if (method.getModifiers().contains(Modifier.ABSTRACT)) {
055                processingEnv.getMessager().printMessage(Diagnostic.Kind.ERROR, "@RequirePOST is meaningless on an abstract method", e);
056            }
057        }
058        for (Element e : roundEnv.getRootElements()) {
059            if (e.getKind() != ElementKind.CLASS) {
060                continue;
061            }
062            for (Element e2 : ((TypeElement) e).getEnclosedElements()) {
063                if (e2.getKind() != ElementKind.METHOD) {
064                    continue;
065                }
066                ExecutableElement method = (ExecutableElement) e2;
067                if (method.getAnnotation(RequirePOST.class) != null) {
068                    continue;
069                }
070                if (method.getModifiers().contains(Modifier.ABSTRACT)) {
071                    continue;
072                }
073                checkForOverrides((ExecutableElement) e2);
074            }
075        }
076        return true;
077    }
078
079    @Override public SourceVersion getSupportedSourceVersion() {
080        return SourceVersion.latest();
081    }
082
083    private void checkForOverrides(ExecutableElement concrete) {
084        checkForOverrides(concrete, ((TypeElement) concrete.getEnclosingElement()).getSuperclass());
085    }
086
087    private void checkForOverrides(ExecutableElement concrete, TypeMirror superclass) {
088        TypeElement superclassE = (TypeElement) processingEnv.getTypeUtils().asElement(superclass);
089        if (superclassE == null) {
090            return;
091        }
092        for (Element e : superclassE.getEnclosedElements()) {
093            if (e.getKind() != ElementKind.METHOD) {
094                continue;
095            }
096            ExecutableElement method = (ExecutableElement) e;
097            if (processingEnv.getElementUtils().overrides(concrete, method, superclassE)) {
098                if (method.getAnnotation(RequirePOST.class) != null) {
099                    processingEnv.getMessager().printMessage(Diagnostic.Kind.WARNING, "@RequirePOST ignored unless also specified on an overriding method", concrete);
100                }
101                return;
102            }
103        }
104        checkForOverrides(concrete, superclassE.getSuperclass());
105    }
106
107}
