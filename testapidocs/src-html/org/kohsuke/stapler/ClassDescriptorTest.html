031    @Test public void loadParametersFromAsm() throws Exception {
032        // get private method that is being tested
033        Method lpfa = ClassDescriptor.ASM.class.getDeclaredMethod(
034                "loadParametersFromAsm", Method.class);
035        lpfa.setAccessible(true);
036        // collect test cases
037        Map<String,Method> testCases = new HashMap<String,Method>();
038        for (Method m : ClassDescriptorTest.class.getDeclaredMethods())
039            if (m.getName().startsWith("methodWith"))
040                testCases.put(m.getName().substring(10), m);
041        // expected results
042        Map<String,String[]> expected = new HashMap<String,String[]>();
043        expected.put("NoParams", new String[0]);
044        expected.put("NoParams_static", new String[0]);
045        expected.put("ManyParams", new String[] { "a", "b", "c", "d", "e", "f", "g", "h", "i" });
046        expected.put("Params_static", new String[] { "abc", "def", "ghi" });
047        // run tests
048        for (Map.Entry<String,String[]> entry : expected.entrySet()) {
049            Method testMethod = testCases.get(entry.getKey());
050            assertNotNull("Method missing for " + entry.getKey(), testMethod);
051            String[] result = (String[])lpfa.invoke(null, testMethod);
052            assertNotNull("Null result for " + entry.getKey());
053            if (!Arrays.equals(entry.getValue(), result)) {
054                StringBuilder buf = new StringBuilder('|');
055                for (String s : result) buf.append(s).append('|');
056                fail("Unexpected result for " + entry.getKey() + ": " + buf);
057            }
058        }
059    }

083    /**
084     * D.x() overries B.x()
085     */
086    @Test
087    public void overridingMethod() throws Exception {
088        FunctionList methods = new ClassDescriptor(D.class).methods.name("x");
089        assertEquals(1, methods.size());
090
091        // we should be able to see both annotations
092        Function f = methods.get(0);
093        assertEquals(3, f.getAnnotation(AnnA.class).value());
094        assertNotNull(f.getAnnotation(AnnB.class));
095        // similarly parameter annotations should be fused together
096        assertSame(Nullable.class, f.getParameterAnnotations()[0][0].annotationType());
097
098        // method should be dispatched to D.x() which overrides B.x()
099        assertEquals(2, f.bindAndInvoke(new D(), null, null, "Hello"));
100    }
101
102    public static class B<T> {
103        @AnnA @AnnB
104        public int x(@Nullable T t) { return 1; }
105    }
106
107    public static class D extends B<String> {
108        @AnnA(3)
109        public int x(String t) { return 2; }
110    }
111
112    @Retention(RetentionPolicy.RUNTIME)
113    @interface AnnA {
114        int value() default 0;
115    }
116
117    @Retention(RetentionPolicy.RUNTIME)
118    @interface AnnB {
119        int value() default 0;
120    }
