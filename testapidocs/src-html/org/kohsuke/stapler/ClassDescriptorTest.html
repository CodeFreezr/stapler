001package org.kohsuke.stapler;

003import java.lang.reflect.Method;
004import java.util.Arrays;
005import java.util.HashMap;
006import java.util.Map;
007import junit.framework.TestCase;
008
009/**
010 * @author Alan Harder
011 */
012public class ClassDescriptorTest extends TestCase {
013    public ClassDescriptorTest() {
014    }
015
016    public ClassDescriptorTest(int a, int b, String x) {
017    }
018
019    public void testLoadConstructorParam() throws Exception {
020        assertEquals(0,ClassDescriptor.loadParameterNames(getClass().getConstructor()).length);
021        String[] names = ClassDescriptor.loadParameterNames(getClass().getConstructor(int.class, int.class, String.class));
022        assertEquals("[a, b, x]",Arrays.asList(names).toString());
023    }
024
025    public void testLoadParametersFromAsm() throws Exception {
026        // get private method that is being tested
027        Method lpfa = ClassDescriptor.class.getDeclaredClasses()[0].getDeclaredMethod(
028                "loadParametersFromAsm", Method.class);
029        lpfa.setAccessible(true);
030        // collect test cases
031        Map<String,Method> testCases = new HashMap<String,Method>();
032        for (Method m : ClassDescriptorTest.class.getDeclaredMethods())
033            if (m.getName().startsWith("methodWith"))
034                testCases.put(m.getName().substring(10), m);
035        // expected results
036        Map<String,String[]> expected = new HashMap<String,String[]>();
037        expected.put("NoParams", new String[0]);
038        expected.put("NoParams_static", new String[0]);
039        expected.put("ManyParams", new String[] { "a", "b", "c", "d", "e", "f", "g", "h", "i" });
040        expected.put("Params_static", new String[] { "abc", "def", "ghi" });
041        // run tests
042        for (Map.Entry<String,String[]> entry : expected.entrySet()) {
043            Method testMethod = testCases.get(entry.getKey());
044            assertNotNull("Method missing for " + entry.getKey(), testMethod);
045            String[] result = (String[])lpfa.invoke(null, testMethod);
046            assertNotNull("Null result for " + entry.getKey());
047            if (!Arrays.equals(entry.getValue(), result)) {
048                StringBuilder buf = new StringBuilder('|');
049                for (String s : result) buf.append(s).append('|');
050                fail("Unexpected result for " + entry.getKey() + ": " + buf);
051            }
052        }
053    }
054
055    private void methodWithNoParams() { }
056    private static void methodWithNoParams_static() { }
057    private void methodWithManyParams(String a, boolean b, int c, long d,
058            Boolean e, Integer f, Long g, Object h, ClassDescriptorTest i) { }
059    private static void methodWithParams_static(String abc, long def, Object ghi) { }
060}
