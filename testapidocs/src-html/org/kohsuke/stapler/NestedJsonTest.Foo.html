001package org.kohsuke.stapler;
002
003import junit.framework.TestCase;
004import net.sf.json.JSONArray;
005import net.sf.json.JSONObject;
006
007import javax.servlet.ServletException;
008import java.util.Collections;
009import java.util.List;
010
011/**
012 * Tests the instantiation of nested objects.
013 *
014 * @author Kohsuke Kawaguchi
015 */
016public class NestedJsonTest extends TestCase {
017    public static final class Foo {
018        public Bar bar;
019
020        @DataBoundConstructor
021        public Foo(Bar bar) {
022            this.bar = bar;
023        }
024    }
025
026    public static interface Bar {}
027
028    public static final class BarImpl implements Bar {
029        public final int i;
030
031        @DataBoundConstructor
032        public BarImpl(int i) {
033            this.i = i;
034        }
035    }
036
037    public void testCreateObject() throws Exception {
038        Foo o = createRequest().bindJSON(Foo.class, createDataSet());
039
040        assertNotNull(o);
041        assertTrue(o.bar instanceof BarImpl);
042        assertEquals(123, ((BarImpl)o.bar).i);
043    }
044
045    public void testInstanceFill() throws Exception {
046        Foo o = new Foo(null);
047        createRequest().bindJSON(o, createDataSet());
048        
049        assertTrue(o.bar instanceof BarImpl);
050        assertEquals(123, ((BarImpl)o.bar).i);
051    }
052
053    public void testCreateList() throws Exception {
054        // Just one
055        List<Foo> list = createRequest().bindJSONToList(Foo.class, createDataSet());
056        assertNotNull(list);
057        assertEquals(1, list.size());
058        assertTrue(list.get(0).bar instanceof BarImpl);
059        assertEquals(123, ((BarImpl)list.get(0).bar).i);
060
061        // Longer list
062        JSONArray data = new JSONArray();
063        data.add(createDataSet());
064        data.add(createDataSet());
065        data.add(createDataSet());
066
067        list = createRequest().bindJSONToList(Foo.class, data);
068        assertNotNull(list);
069        assertEquals(3, list.size());
070        assertEquals(123, ((BarImpl)list.get(2).bar).i);
071    }
072
073    private RequestImpl createRequest() throws Exception {
074        return new RequestImpl(createStapler(), new MockRequest(), Collections.EMPTY_LIST,null);
075    }
076
077    private JSONObject createDataSet() {
078        JSONObject bar = new JSONObject();
079        bar.put("i",123);
080        JSONObject foo = new JSONObject();
081        foo.put("bar",bar);
082        foo.getJSONObject("bar").put("$class", BarImpl.class.getName());
083
084        return foo;
085    }
086
087    private Stapler createStapler() throws ServletException {
088        Stapler stapler = new Stapler();
089        stapler.init(new ServletConfigImpl());
090        return stapler;
091    }
092}
