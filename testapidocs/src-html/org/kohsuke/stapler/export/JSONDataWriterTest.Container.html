001package org.kohsuke.stapler.export;
002
003import org.junit.Test;
004
005import java.io.IOException;
006import java.io.StringWriter;
007
008import static org.junit.Assert.assertEquals;
009
010public class JSONDataWriterTest {
011    private ExportConfig config = new ExportConfig()
012            .withClassAttribute(ClassAttributeBehaviour.IF_NEEDED.simple());
013
014    private <T> String serialize(T bean, Class<T> clazz) throws IOException {
015        StringWriter w = new StringWriter();
016        Model<T> model = new ModelBuilder().get(clazz);
017        model.writeTo(bean, Flavor.JSON.createDataWriter(bean, w, config));
018        return w.toString();
019    }
020
021    @ExportedBean public static class X {
022        @Exported public String a = "aval";
023        public String b = "bval";
024        @Exported public String getC() {return "cval";}
025        public String getD() {return "dval";}
026    }
027
028    @Test
029    public void testSimpleUsage() throws Exception {
030        assertEquals("{\"_class\":\"X\",\"a\":\"aval\",\"c\":\"cval\"}",
031                serialize(new X(), X.class));
032    }
033
034    @ExportedBean(defaultVisibility=2) public static abstract class Super {
035        @Exported public String basic = "super";
036        @Exported public abstract String generic();
037    }
038    public static class Sub extends Super {
039        public String generic() {return "sub";}
040        @Exported public String specific() {return "sub";}
041    }
042    @ExportedBean public static class Container {
043        @Exported public Super polymorph = new Sub();
044    }
045
046    @Test
047    public void testInheritance() throws Exception {
048        assertEquals("{\"_class\":\"Container\",\"polymorph\":{\"_class\":\"Sub\",\"basic\":\"super\",\"generic\":\"sub\",\"specific\":\"sub\"}}",
049                serialize(new Container(), Container.class));
050    }
051
052    public static class Sub2 extends Super {
053        @Exported @Override public String generic() {return "sub2";}
054    }
055
056    @Test
057    public void testInheritance2() throws Exception { // JENKINS-13336
058        assertEquals("{\"_class\":\"Sub2\",\"basic\":\"super\",\"generic\":\"sub2\"}",
059                serialize(new Sub2(), Sub2.class));
060    }
061}
