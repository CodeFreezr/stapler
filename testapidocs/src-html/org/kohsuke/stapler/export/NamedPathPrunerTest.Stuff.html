001package org.kohsuke.stapler.export;
002
003import java.io.StringWriter;
004import java.util.Arrays;
005import java.util.List;
006import junit.framework.TestCase;
007
008public class NamedPathPrunerTest extends TestCase {
009
010    public NamedPathPrunerTest(String name) {
011        super(name);
012    }
013    
014    public void testParse() throws Exception {
015        assertEquals("{a={}, b={c={}}}", NamedPathPruner.parse("a,b[c]").toString());
016        assertEquals("{a={}, b={c={}, d={}}}", NamedPathPruner.parse("a,b[c,d]").toString());
017        assertEquals("{a={}, b={c={}, d={}}, e={}}", NamedPathPruner.parse("a,b[c,d],e").toString());
018        assertEquals("{a={}, b={c={}, d={}}, e={}}", NamedPathPruner.parse("a,b[c,d]{,10},e").toString());
019        assertParseError("");
020        assertParseError("a,");
021        assertParseError(",b");
022        assertParseError("a[");
023        assertParseError("a[b,c");
024        assertParseError("a[]");
025        assertParseError("a[b,,]");
026        assertParseError("a]");
027        assertParseError("a{}");
028        assertParseError("a{b}");
029    }
030    private static void assertParseError(String spec) {
031        try {
032            NamedPathPruner.parse(spec);
033            fail();
034        } catch (IllegalArgumentException x) {
035            // pass
036        }
037    }
038    
039    public void testPruning() throws Exception {
040        Jhob job1 = new Jhob("job1", "Job #1", "whatever");
041        Jhob job2 = new Jhob("job2", "Job #2", "junk");
042        Vhew view1 = new Vhew("All", "crap", new Jhob[] {job1, job2});
043        Vhew view2 = new Vhew("Some", "less", new Jhob[] {job1});
044        Stuff bean = new Stuff(new Jhob[] {job1, job2}, Arrays.asList(view1, view2));
045        assertResult("{jobs:[{displayName:Job #1,name:job1},{displayName:Job #2,name:job2}],"
046                + "views:[{jobs:[{name:job1},{name:job2}],name:All},{jobs:[{name:job1}],name:Some}]}",
047                bean, "jobs[name,displayName],views[name,jobs[name]]");
048        assertResult("{jobs:[{displayName:Job #1,name:job1}],views:[{jobs:[],name:All},"
049                + "{jobs:[],name:Some}]}",
050                bean, "jobs[name,displayName]{,1},views[name,jobs[name]{,0}]");
051    }
052
053    public void testRange() throws Exception {
054        Jhob[] jobs = new Jhob[100];
055        for (int i=0; i<jobs.length; i++)
056            jobs[i] = new Jhob("job"+i,"aaa","bbb");
057        Vhew v = new Vhew("view","aaa",jobs);
058
059        assertResult("{jobs:[{name:job0},{name:job1},{name:job2}]}",    v, "jobs[name]{,3}");
060        assertResult("{jobs:[{name:job3},{name:job4},{name:job5}]}",    v, "jobs[name]{3,6}");
061        assertResult("{jobs:[{name:job38}]}",                           v, "jobs[name]{38}");
062        assertResult("{jobs:[{name:job97},{name:job98},{name:job99}]}", v, "jobs[name]{97,}");
063    }
064    
065    @ExportedBean public static class Stuff {
066        @Exported public Jhob[] jobs;
067        @Exported public List<Vhew> views;
068        public Stuff(Jhob[] jobs, List<Vhew> views) {
069            this.jobs = jobs.clone();
070            this.views = views;
071        }
072    }
073    @ExportedBean public static class Jhob {
074        @Exported public String name;
075        @Exported public String displayName;
076        @Exported public String trash;
077        public Jhob(String name, String displayName, String trash) {
078            this.name = name;
079            this.displayName = displayName;
080            this.trash = trash;
081        }
082    }
083    @ExportedBean public static class Vhew {
084        @Exported public String name;
085        @Exported public String trash;
086        @Exported public Jhob[] jobs;
087        public Vhew(String name, String trash, Jhob[] jobs) {
088            this.name = name;
089            this.trash = trash;
090            this.jobs = jobs.clone();
091        }
092    }
093    
094    @SuppressWarnings({"unchecked", "rawtypes"}) // API design flaw prevents this from type-checking
095    private static void assertResult(String expected, Object bean, String spec) throws Exception {
096        Model model = new ModelBuilder().get(bean.getClass());
097        StringWriter w = new StringWriter();
098        model.writeTo(bean, new NamedPathPruner(spec), Flavor.JSON.createDataWriter(bean, w));
099        assertEquals(expected, w.toString().replace("\\\"", "").replace("\"", ""));
100    }
