029import java.io.ByteArrayInputStream;
030import java.io.ByteArrayOutputStream;
031import java.io.StringReader;
032import java.io.StringWriter;
033import javax.xml.XMLConstants;
034import javax.xml.transform.stream.StreamResult;
035import javax.xml.transform.stream.StreamSource;
036import javax.xml.validation.SchemaFactory;
037import static org.junit.Assert.*;
038import org.junit.Test;
039import org.xml.sax.SAXParseException;
040
041public class SchemaGeneratorTest {
042
043    @Test public void basics() throws Exception {
044        validate(new XMLDataWriterTest.X(), XMLDataWriterTest.X.class);
045    }
046
047    /* TODO currently fails
048    @Test public void inheritance() throws Exception {
049        validate(new XMLDataWriterTest.Container(), XMLDataWriterTest.Container.class);
050    }
051    */
052    
053    private static <T> void validate(T bean, Class<T> clazz) throws Exception {
054        Model<T> model = new ModelBuilder().get(clazz);
055        ByteArrayOutputStream schema = new ByteArrayOutputStream();
056        new SchemaGenerator(model).generateSchema(new StreamResult(schema));
057        StringWriter xml = new StringWriter();
058        model.writeTo(bean, Flavor.XML.createDataWriter(bean, xml));
059        try {
060            SchemaFactory.newInstance(XMLConstants.W3C_XML_SCHEMA_NS_URI).newSchema(new StreamSource(new ByteArrayInputStream(schema.toByteArray()))).newValidator().validate(new StreamSource(new StringReader(xml.toString())));
061        } catch (SAXParseException x) {
062            fail(x + "\n" + xml + "\n" + schema);
063        }
064    }
065
066}
