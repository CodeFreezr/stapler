024package org.kohsuke.stapler.export;
025
026import static org.junit.Assert.assertEquals;
027
028import java.io.StringWriter;
029import java.util.Arrays;
030import java.util.Collection;
031import java.util.List;
032
033import org.junit.Test;
034
035public class ModelTest {
036    ModelBuilder builder = new ModelBuilder();
037
038    @Test // JENKINS-26775
039    public void sytheticMethodShouldNotBeExported() {
040        Model<Impl> model = builder.get(Impl.class);
041        assertEquals("Redundant properties discovered: " + model.getProperties(), 1, model.getProperties().size());
042    }
043
044    public static interface GenericInterface<T extends Number> {
045        Collection<T> get();
046    }
047
048    @ExportedBean
049    public static class Impl implements GenericInterface<Integer> {
050        @Exported
051        public List<Integer> get() {
052            return Arrays.asList(42);
053        }
054    }
055
056    //===========================================
057
058    @Test
059    public void merge() throws Exception {
060        StringWriter sw = new StringWriter();
061        builder.get(B.class).writeTo(b, Flavor.JSON.createDataWriter(b, sw));
062        // B.x should maskc C.x, so x should be 40
063        // but C.y should be printed as merged
064        assertEquals("{'y':20,'z':30,'x':40}", sw.toString().replace('"','\''));
065    }
066
067    /**
068     * y is a property from a merged object but that shouldn't be visible to {@link NamedPathPruner}.
069     */
070    @Test
071    public void merge_pathPrune() throws Exception {
072        StringWriter sw = new StringWriter();
073        builder.get(B.class).writeTo(b, new NamedPathPruner("z,y"), Flavor.JSON.createDataWriter(b, sw));
074        assertEquals("{'y':20,'z':30}", sw.toString().replace('"','\''));
075    }
076
077    B b = new B();
078    public static class B extends A {
079        @Exported
080        public int x = 40;
081    }
082
083    @ExportedBean
084    public static class A {
085        @Exported(merge=true)
086        public C c = new C();
087
088        @Exported
089        public int z = 30;
090    }
091
092    @ExportedBean
093    public static class C {
094        @Exported public int x = 10;
095        @Exported public int y = 20;
096    }
097}
