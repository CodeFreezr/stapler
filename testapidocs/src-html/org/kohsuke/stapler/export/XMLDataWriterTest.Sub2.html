001package org.kohsuke.stapler.export;
002
003import junit.framework.TestCase;
004import org.xml.sax.InputSource;
005import org.xml.sax.helpers.DefaultHandler;
006
007import javax.xml.parsers.SAXParser;
008import javax.xml.parsers.SAXParserFactory;
009import java.io.IOException;
010import java.io.StringReader;
011import java.io.StringWriter;
012
013public class XMLDataWriterTest extends TestCase {
014
015    public XMLDataWriterTest(String n) {
016        super(n);
017    }
018
019    private static <T> String serialize(T bean, Class<T> clazz) throws IOException {
020        StringWriter w = new StringWriter();
021        Model<T> model = new ModelBuilder().get(clazz);
022        model.writeTo(bean, Flavor.XML.createDataWriter(bean, w));
023        return w.toString();
024    }
025
026    @ExportedBean public static class X {
027        @Exported public String a = "aval";
028        public String b = "bval";
029        @Exported public String getC() {return "cval";}
030        public String getD() {return "dval";}
031    }
032    public void testSimpleUsage() throws Exception {
033        assertEquals("<x><a>aval</a><c>cval</c></x>",
034                serialize(new X(), X.class));
035    }
036
037    @ExportedBean(defaultVisibility=2) public static abstract class Super {
038        @Exported public String basic = "super";
039        @Exported public abstract String generic();
040    }
041    public static class Sub extends Super {
042        public String generic() {return "sub";}
043        @Exported public String specific() {return "sub";}
044    }
045    @ExportedBean public static class Container {
046        @Exported public Super polymorph = new Sub();
047    }
048    public void testInheritance() throws Exception {
049        assertEquals("<container><polymorph><basic>super</basic><generic>sub</generic>" +
050                "<specific>sub</specific></polymorph></container>",
051                serialize(new Container(), Container.class));
052    }
053
054    public static class Sub2 extends Super {
055        @Exported @Override public String generic() {return "sub2";}
056    }
057    public void testInheritance2() throws Exception { // JENKINS-13336
058        assertEquals("<sub2><basic>super</basic><generic>sub2</generic></sub2>",
059                serialize(new Sub2(), Sub2.class));
060    }
061
062    private void assertValidXML(String s) throws Exception {
063        SAXParser p = SAXParserFactory.newInstance().newSAXParser();
064        p.parse(new InputSource(new StringReader(s)),new DefaultHandler());
065    }
066
067    /**
068     * Can we write out anonymous classes as the root object?
069     */
070    public void testAnonymousClass() throws Exception {
071        assertValidXML(serialize(new X() {},X.class));
072    }
073
074    @ExportedBean
075    public static class PA {
076        @Exported public int[] v = new int[]{1,2,3};
077    }
078
079    public void testPrimitiveArrays() throws Exception {
080        assertEquals("<PA><v>1</v><v>2</v><v>3</v></PA>",serialize(new PA(),PA.class));
081    }
082
083    public void testMakeXmlName() {
084        assertEquals("_",   XMLDataWriter.makeXmlName(""));
085        assertEquals("abc", XMLDataWriter.makeXmlName("abc"));
086        assertEquals("abc", XMLDataWriter.makeXmlName("/abc"));
087        assertEquals("abc", XMLDataWriter.makeXmlName("/a/b/c/"));
088    }
089
090    @ExportedBean public static class Arrays {
091        @Exported public String[] categories = {"general", "specific"};
092        @Exported public String[] styles = {"ornate", "plain"};
093    }
094
095    public void testToSingular() throws Exception {
096        assertEquals("<arrays><category>general</category><category>specific</category><style>ornate</style><style>plain</style></arrays>",
097                serialize(new Arrays(), Arrays.class));
098    }
099
100}
