003import junit.framework.TestCase;
004import org.kohsuke.stapler.test.AbstractStaplerTest;
005import org.mockito.Mockito;

007import javax.servlet.ServletConfig;
008import javax.servlet.ServletContext;
009import javax.servlet.http.HttpServletRequest;
010import javax.servlet.http.HttpServletResponse;
011import java.util.ArrayList;
012
013/**
014 * This class needs to be in this package to access package-protected stuff.
015 * You should extend from {@link AbstractStaplerTest}.
016 *
017 * @author Kohsuke Kawaguchi
018 */
019public abstract class AbstractStaplerTestBase extends TestCase {
020
021    protected WebApp webApp;
022    protected RequestImpl request;
023    protected ResponseImpl response;
024    protected Stapler stapler = new Stapler();
025    protected HttpServletRequest rawRequest;
026    protected HttpServletResponse rawResponse;
027
028    @Override
029    protected void setUp() throws Exception {
030        super.setUp();
031        ServletContext servletContext = Mockito.mock(ServletContext.class);
032
033        webApp = new WebApp(servletContext);
034
035        ServletConfig servletConfig = Mockito.mock(ServletConfig.class);
036        Mockito.when(servletConfig.getServletContext()).thenReturn(servletContext);
037        stapler.init(servletConfig);
038
039        rawRequest = Mockito.mock(HttpServletRequest.class);
040        rawResponse = Mockito.mock(HttpServletResponse.class);
041
042        this.request = new RequestImpl(stapler, rawRequest,new ArrayList<AncestorImpl>(),new TokenList(""));
043        Stapler.CURRENT_REQUEST.set(this.request);
044
045        this.response = new ResponseImpl(stapler, rawResponse);
046        Stapler.CURRENT_RESPONSE.set(this.response);
047    }
048}
