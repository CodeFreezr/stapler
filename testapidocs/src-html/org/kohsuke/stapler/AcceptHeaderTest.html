001package org.kohsuke.stapler;
002
003import org.junit.Assert;
004import org.junit.Test;
005
006/**
007 * @author Vivek Pandey
008 */
009public class AcceptHeaderTest {
010
011    @Test
012    public void simpleHeader() {
013        AcceptHeader.Atom atom = new AcceptHeader("text/html").match("text/html");
014        Assert.assertNotNull(atom);
015        Assert.assertEquals("text/html", atom.toString());
016    }
017
018    @Test
019    public void headerSelect1() {
020        String type = new AcceptHeader("text/html").select("text/html", "text/plain");
021        Assert.assertNotNull(type);
022        Assert.assertEquals("text/html", type);
023    }
024
025    @Test
026    public void headerSelect2() {
027        String type = new AcceptHeader("text/*").select("text/html");
028        Assert.assertNotNull(type);
029        Assert.assertEquals("text/html", type);
030    }
031
032    @Test(expected = HttpResponses.HttpResponseException.class)
033    public void headerSelectUnsupportedMediaType() {
034        new AcceptHeader("text/html").select("application/json");
035    }
036    @Test
037    public void simpleHeaderWithParam() {
038        AcceptHeader acceptHeader = new AcceptHeader("application/json;charset=utf-8");
039
040        AcceptHeader.Atom atom = acceptHeader.match("application/json");
041        Assert.assertNotNull(atom);
042        Assert.assertEquals("application/json;charset=utf-8", atom.toString());
043
044        String type = acceptHeader.select("application/json");
045        Assert.assertNotNull(type);
046        Assert.assertEquals("application/json;charset=utf-8", atom.toString());
047    }
048
049
050    @Test
051    public void wildCard() {
052        String type = new AcceptHeader("text/*").select("text/html");
053        Assert.assertNotNull(type);
054        Assert.assertEquals("text/html", type);
055    }
056
057    @Test
058    public void qualityFactor1() {
059        String type = new AcceptHeader("image/*;q=0.5, image/png;q=1").select("image/jpeg", "image/png");
060        Assert.assertNotNull(type);
061        Assert.assertEquals(type, "image/png");
062    }
063
064    @Test
065    public void qualityFactor2() {
066        String type = new AcceptHeader("text/*;q=0.5, *;q=0.1").select("application/xbel+xml", "text/xml");
067        Assert.assertNotNull(type);
068        Assert.assertEquals(type, "text/xml");
069    }
070
071    @Test
072    public void qualityFactor3() {
073        String type = new AcceptHeader("text/xml;q=0.5, application/json;q=0.1, text/html;q=0.1").select("application/json", "text/xml");
074        Assert.assertNotNull(type);
075        Assert.assertEquals(type, "text/xml");
076    }
077
078    @Test
079    public void qualityFactor4(){
080        String type = new AcceptHeader("text/html,application/xhtml+xml,application/xml;q=0.9,*/*;q=0.8").select("application/xml");
081        Assert.assertNotNull(type);
082        Assert.assertEquals(type, "application/xml");
083
084        type = new AcceptHeader("text/html,application/xhtml+xml,application/xml;q=0.9,*/*;q=0.8").select("foo/bar");
085        Assert.assertNotNull(type);
086        Assert.assertEquals(type, "foo/bar");
087    }
088
089    @Test
090    public void qualityFactor5(){
091         AcceptHeader acceptHeader = new AcceptHeader("application/xml,application/xhtml+xml,text/html;q=0.9,text/plain;q=0.8,image/png,*/*;q=0.5");
092
093        String type = acceptHeader.select("text/html", "image/png");
094        Assert.assertNotNull(type);
095        Assert.assertEquals(type, "image/png");
096
097        type = acceptHeader.select("text/plain", "image/png");
098        Assert.assertNotNull(type);
099        Assert.assertEquals(type, "image/png");
100
101        type = acceptHeader.select("text/html", "image/png");
102        Assert.assertNotNull(type);
103        Assert.assertEquals(type, "image/png");
104
105        type = acceptHeader.select("text/plain", "image/png","application/xml");
106        Assert.assertNotNull(type);
107        Assert.assertEquals(type, "image/png");
108
109    }
110
111    @Test
112    public void qualityFactor6(){
113        AcceptHeader acceptHeader = new AcceptHeader("text/html,application/xhtml+xml,application/xml;q=0.9,image/webp,*/*;q=0.8");
114
115        String type = acceptHeader.select("text/plain", "text/html");
116        Assert.assertNotNull(type);
117        Assert.assertEquals(type, "text/html"); // 1 > 0.8
118
119        type = acceptHeader.select("text/plain", "application/xml");
120        Assert.assertNotNull(type);
121        Assert.assertEquals(type, "application/xml"); //0.9 > 0.8
122    }
123
124}
