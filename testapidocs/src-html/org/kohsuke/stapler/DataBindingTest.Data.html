226    public static class Point {
227        int x,y;
228
229        @DataBoundConstructor
230        public Point(int x, int y) {
231            this.x = x;
232            this.y = y;
233        }
234
235        @Override
236        public boolean equals(Object o) {
237            if (this == o) return true;
238            if (o == null || getClass() != o.getClass()) return false;
239
240            Point rhs = (Point) o;
241            return x == rhs.x && y == rhs.y;
242        }
243
244        @Override
245        public int hashCode() {
246            int result = x;
247            result = 31 * result + y;
248            return result;
249        }
250    }

252    public void testSetterInvocation() {
253        SetterBinding r = bind("{x:1,y:2,z:3,w:1, children:[{x:5,y:5,z:5},{x:6,y:6,z:6}], anotherObject:{$class:'org.kohsuke.stapler.DataBindingTest$Point', x:1,y:1} }",SetterBinding.class);
254        assertEquals(1,r.x);
255        assertEquals(2,r.y);
256        assertEquals(3,r.z);
257        assertEquals(0,r.w);
258
259        assertEquals(2,r.children.size());
260        SetterBinding c1 = r.children.get(0);
261        assertEquals(5, c1.x);
262        assertEquals(5, c1.y);
263        assertEquals(5, c1.z);
264
265        SetterBinding c2 = r.children.get(1);
266        assertEquals(6, c2.x);
267        assertEquals(6, c2.y);
268        assertEquals(6, c2.z);
269
270        Point o = (Point)r.o;
271        assertEquals(1,o.x);
272        assertEquals(1,o.y);
273    }

322    public void testInterceptor2() {
323        RequestImpl req = createFakeRequest();
324        req.setBindInterceptor(new BindInterceptor() {
325            @Override
326            public Object onConvert(Type targetType, Class targetTypeErasure, Object jsonSource) {
327                if (targetType==String.class)
328                    return String.valueOf(((JSONObject) jsonSource).getInt("x"));
329                return DEFAULT;
330            }
331        });
332
333        String[] r = (String[]) req.bindJSON(String[].class, String[].class, JSONArray.fromObject("[{x:1},{x:2}]"));
334        assertTrue(Arrays.equals(r,new String[]{"1","2"}));
