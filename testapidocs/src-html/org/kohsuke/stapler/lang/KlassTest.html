001package org.kohsuke.stapler.lang;
002
003import org.junit.Assert;
004import org.junit.Test;
005
006import java.util.List;
007
008/**
009 * Contains tests for {@link Klass}.
010 * @author Oleg Nenashev.
011 */
012public class KlassTest {
013
014    @Test
015    public void shouldProperlyAccessJavaDeclaredFields() throws Exception {
016        final Klass<Class> classInstance = Klass.java(FooClass.class);
017        final List<FieldRef> declaredFields = classInstance.getDeclaredFields();
018        for (FieldRef ref : declaredFields) {
019            if ("fooField".equals(ref.getName())) {
020                //TODO: check field parameters once Stapler provides such info
021                return;
022            }
023        }
024        Assert.fail("Have not found 'fooField' in the returned field list");
025    }
026
027    private static final class FooClass {
028        private int fooField;
029    }
030}
