046    public final VerbMatch verbMatch = new VerbMatch();
047
048    public class VerbMatch {
049        @WebMethod(name="") @GET
050        public HttpResponse doGet() {
051            return HttpResponses.plainText("Got GET");
052        }
053
054        @WebMethod(name="") @POST
055        public HttpResponse doPost() {
056            return HttpResponses.plainText("Got POST");
057        }
058    }
059
060    /**
061     * Tests the dispatching of WebMethod based on verb
062     */
063    public void testVerbMatch() throws Exception {
064        WebClient wc = new WebClient();
065
066        check(wc, HttpMethod.GET);
067        check(wc, HttpMethod.POST);
068        try {
069            check(wc, HttpMethod.DELETE);
070            fail("There's no route for DELETE");
071        } catch (FailingHttpStatusCodeException e) {
072            assertEquals(404, e.getStatusCode());
073        }
074    }
075
076    private void check(WebClient wc, HttpMethod m) throws java.io.IOException {
077        TextPage p = wc.getPage(new WebRequestSettings(new URL(url, "verbMatch/"), m));
078        assertEquals("Got "+m.name()+"\n", p.getContent());
079    }

114    public final InterceptorStage interceptorStage = new InterceptorStage();
115
116    public class InterceptorStage {
117        @POST
118        public HttpResponse doFoo(@JsonBody Point body) {
119            return HttpResponses.plainText(body.x+","+body.y);
120        }
121    }
122
123    public static class Point {
124        public int x,y;
125    }
126
127    /**
128     * POST annotation selection needs to happen before databinding of the parameter happens.
129     */
130    public void testInterceptorStage() throws Exception {
131        WebClient wc = new WebClient();
132        try {
133            wc.getPage(new URL(url, "interceptorStage/foo"));
134            fail("Expected 404");
135        } catch (FailingHttpStatusCodeException e) {
136            assertEquals(404, e.getStatusCode());
137        }
138
139        WebRequestSettings req = new WebRequestSettings(new URL(url, "interceptorStage/foo"), HttpMethod.POST);
140        req.setAdditionalHeader("Content-Type","application/json");
141        req.setRequestBody("{x:3,y:5}");
142        TextPage p = wc.getPage(req);
143        assertEquals("3,5\n",p.getContent());
144
145    }

151    public final Inheritance inheritance = new Inheritance2();
152    public class Inheritance {
153        @WebMethod(name="foo")
154        public HttpResponse doBar(@QueryParameter String q) {
155            return HttpResponses.plainText("base");
156        }
157    }
158
159    public class Inheritance2 extends Inheritance {
160        @Override
161        public HttpResponse doBar(String q) {
162            return HttpResponses.plainText(q);
163        }
164    }
165
166    public void testInheritance() throws Exception {
167        WebClient wc = new WebClient();
168
169        // the request should get to the overriding method and it should still see all the annotations in the base type
170        TextPage p = wc.getPage(new URL(url, "inheritance/foo?q=abc"));
171        assertEquals("abc\n", p.getContent());
172
173        // doBar is a web method for 'foo', so bar endpoint shouldn't respond
174        try {
175            wc.getPage(new URL(url, "inheritance/bar"));
176            fail();
177        } catch (FailingHttpStatusCodeException e) {
178            assertEquals(404,e.getStatusCode());
179        }
180    }

186    public final RequirePostOnBase requirePostOnBase = new RequirePostOnBase2();
187    public abstract class RequirePostOnBase {
188        int hit;
189        @RequirePOST
190        public abstract void doSomething();
191    }
192
193    public class RequirePostOnBase2 extends RequirePostOnBase {
194        @Override
195        public void doSomething() {
196            hit++;
197        }
198    }
199
200    public void testRequirePostOnBase() throws Exception {
201        WebClient wc = new WebClient();
202        URL url = new URL(this.url, "requirePostOnBase/something");
203
204        try {
205            wc.getPage(url);
206            fail();
207        } catch (FailingHttpStatusCodeException e) {
208            assertEquals(HttpServletResponse.SC_METHOD_NOT_ALLOWED, e.getStatusCode());
209        }
210
211        // POST should succeed
212        wc.getPage(new WebRequestSettings(url, HttpMethod.POST));
213        assertEquals(1, requirePostOnBase.hit);
214    }
