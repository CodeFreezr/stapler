010public class JRubyClassInfoTest extends TestCase {
011    public void testCamelize() throws Exception {
012        assertEquals("foo_bar_zot", decamelize("FooBarZot"));
013        assertEquals("http_connection",decamelize("HTTPConnection"));
014        assertEquals("managed_jdk_listener",decamelize("ManagedJDKListener"));
015        assertEquals("managed_jdk15_listener",decamelize("ManagedJDK15Listener"));
016        assertEquals("abc000thunder",decamelize("Abc000thunder"));
017
018        assertEquals("nested_nested/http_connection",decamelize("NestedNested/HTTPConnection"));
019        assertEquals("a/http_connection",decamelize("A/HTTPConnection"));
020        assertEquals("a/http_connection",decamelize("a/HTTPConnection"));
021        assertEquals("aa/http_connection",decamelize("AA/HTTPConnection"));
022        assertEquals("aa/http_connection",decamelize("Aa/HTTPConnection"));
023        assertEquals("aa_a/http_connection",decamelize("AaA/HTTPConnection"));
024        assertEquals("/http_connection",decamelize("/HTTPConnection"));
025        assertEquals("nested/a",decamelize("Nested/A"));
026        assertEquals("nested/a",decamelize("Nested/a"));
027        assertEquals("nested/",decamelize("Nested/"));
028        assertEquals("/",decamelize("/"));
029        assertEquals("",decamelize(""));
030        try {
031            decamelize(null);
032            fail();
033        } catch (NullPointerException npe) {
034        }
035    }
