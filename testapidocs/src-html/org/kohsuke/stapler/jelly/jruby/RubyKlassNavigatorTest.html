001package org.kohsuke.stapler.jelly.jruby;
002
003import org.jruby.CompatVersion;
004import org.jruby.Ruby;
005import org.jruby.RubyClass;
006import org.jruby.RubyModule;
007import org.jruby.embed.LocalContextScope;
008import org.jruby.embed.ScriptingContainer;
009import org.junit.Assert;
010import org.junit.Test;
011import org.kohsuke.stapler.lang.FieldRef;
012import org.kohsuke.stapler.lang.Klass;
013import org.jvnet.hudson.test.Issue;
014
015import java.util.List;
016
017import static org.hamcrest.CoreMatchers.*;
018import org.hamcrest.collection.IsEmptyCollection;
019import static org.junit.Assume.assumeThat;
020
021/**
022 * Tests handling of Ruby classes and modules with {@link RubyKlassNavigator}.
023 * @author Oleg Nenashev.
024 */
025public class RubyKlassNavigatorTest {
026
027    /**
028     * Verifies that field retrieval do not fail horribly for {@link RubyModule}.
029     * Effective use-case - Ruby Runtime Plugin for Jenkins.
030     * @throws Exception Test failure
031     */
032    @Test
033    @Issue("JENKINS-39414")
034    public void shouldProperlyHandleRubyModules() throws Exception {
035        final ScriptingContainer ruby = createRubyInstance();
036        final RubyKlassNavigator navigator = new RubyKlassNavigator(ruby.getProvider().getRuntime(), ClassLoader.getSystemClassLoader());
037        final MyRubyModule myModule = new MyRubyModule(ruby.getRuntime(), new MyRubyClass(ruby.getRuntime()), true);
038
039
040        final Klass<RubyModule> classInstance = new Klass<RubyModule>(myModule, navigator);
041        final List<FieldRef> declaredFields = classInstance.getDeclaredFields();
042
043        assumeThat("Access to fields in Ruby Modules has not been implemented yet",
044                declaredFields, not(IsEmptyCollection.<FieldRef>empty()));
045        for (FieldRef ref : declaredFields) {
046            if ("fooField".equals(ref.getName())) {
047                //TODO: check fields once implemented in Stapler
048                return;
049            }
050        }
051        Assert.fail("Have not found 'fooField' in the returned field list");
052    }
053
054    private static final class MyRubyModule extends RubyModule {
055
056        private int fooField = 123;
057
058        MyRubyModule(Ruby runtime, RubyClass metaClass, boolean objectSpace) {
059            super(runtime, metaClass, objectSpace);
060        }
061    }
062
063    private static final class MyRubyClass extends RubyClass {
064        MyRubyClass(Ruby runtime) {
065            super(runtime);
066        }
067    }
068
069    private ScriptingContainer createRubyInstance() {
070        // Similar to Ruby Runtime Plugin, but with non-Jenkins classloader
071        final ScriptingContainer ruby = new ScriptingContainer(LocalContextScope.SINGLETHREAD);
072        ruby.setCompatVersion(CompatVersion.RUBY1_9);
073        ruby.setClassLoader(ClassLoader.getSystemClassLoader());
074        return ruby;
075    }
076}
