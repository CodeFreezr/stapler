037    /**
038     * Verifies that declared method retrieval works well.
039     * Effective use-case - Ruby Runtime Plugin for Jenkins.
040     * @throws Exception Test failure
041     */
042    @Test
043    public void shouldProperlyHandleDeclaredMethodsInRubyModules() throws Exception {
044        final ScriptingContainer ruby = createRubyInstance();
045        final RubyKlassNavigator navigator = new RubyKlassNavigator(ruby.getProvider().getRuntime(), ClassLoader.getSystemClassLoader());
046        final MyRubyModule myModule = new MyRubyModule(ruby.getRuntime(), new MyRubyClass(ruby.getRuntime()), true);
047
048
049        final Klass<RubyModule> classInstance = new Klass<RubyModule>(myModule, navigator);
050        
051        final List<MethodRef> declaredMethods = classInstance.getDeclaredMethods();
052        for (MethodRef ref : declaredMethods) {
053            if (ref instanceof RubyMethodRef) {
054                // Ruby engine API allows creating methods with null names, not our bug BTW...
055                if ("doDynamic".equals(ref.getName())) {
056                    //TODO: More consistency checks
057                    return;
058                }
059            }
060        }
061        Assert.fail("Have not found 'doDynamic' in the returned function list");
062    }

065     * Verifies that field retrieval do not fail horribly for {@link RubyModule}.
066     * Effective use-case - Ruby Runtime Plugin for Jenkins.
067     * @throws Exception Test failure
068     */
069    @Test
070    @Issue("JENKINS-39414")
071    public void shouldProperlyHandleFieldsInRubyModules() throws Exception {
072        final ScriptingContainer ruby = createRubyInstance();
073        final RubyKlassNavigator navigator = new RubyKlassNavigator(ruby.getProvider().getRuntime(), ClassLoader.getSystemClassLoader());
074        final MyRubyModule myModule = new MyRubyModule(ruby.getRuntime(), new MyRubyClass(ruby.getRuntime()), true);
075
076
077        final Klass<RubyModule> classInstance = new Klass<RubyModule>(myModule, navigator);
078        final List<FieldRef> declaredFields = classInstance.getDeclaredFields();
079
080        assumeThat("Access to fields in Ruby Modules has not been implemented yet",
081                declaredFields, not(IsEmptyCollection.<FieldRef>empty()));
082        for (FieldRef ref : declaredFields) {
083            if ("fooField".equals(ref.getName())) {
084                //TODO: check fields once implemented in Stapler

086            }
087        }
088        Assert.fail("Have not found 'fooField' in the returned field list");
089    }

107        final List<Function> declaredFunctions = classInstance.getFunctions();

118    }

120    private static final class MyRubyModule extends RubyModule {
121                
122        private int fooField = 123;
123        
124        MyRubyModule(Ruby runtime, RubyClass metaClass, boolean objectSpace) {
125            super(runtime, metaClass, objectSpace);
126            
127            // Register a Mock method for the class
128            getMethodsForWrite().put("doDynamic", new DynamicMethod(this, Visibility.PUBLIC, CallConfiguration.FrameFullScopeFull, "doDynamic") {
129                @Override
130                public IRubyObject call(ThreadContext context, IRubyObject self, RubyModule clazz, String name, IRubyObject[] args, Block block) {
131                    return doDynamic("foo");
132                }
133                
134                @Override
135                public DynamicMethod dup() {
136                    throw new UnsupportedOperationException("Not supported yet.");
137                }
138            });
139        }
140        
141        public final IRubyObject doDynamic(String token) {
142            // Just return a routable object
143            return new RubyFixnum(getRuntime(), 0);
144        }
145    }
146
147    private static final class MyRubyClass extends RubyClass {     
148        MyRubyClass(Ruby runtime) {
149            super(runtime);
150        }   
151    }
152
153    private ScriptingContainer createRubyInstance() {
154        // Similar to Ruby Runtime Plugin, but with non-Jenkins classloader
155        final ScriptingContainer ruby = new ScriptingContainer(LocalContextScope.SINGLETHREAD);
156        ruby.setCompatVersion(CompatVersion.RUBY1_9);
157        ruby.setClassLoader(ClassLoader.getSystemClassLoader());
158        return ruby;
159    }
