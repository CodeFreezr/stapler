001package org.kohsuke.stapler.jelly.jruby;
002
003import org.apache.commons.jelly.JellyContext;
004import org.apache.commons.jelly.Script;
005import org.apache.commons.jelly.XMLOutput;
006import org.kohsuke.stapler.MetaClass;
007import org.kohsuke.stapler.jelly.JellyClassLoaderTearOff;
008import org.kohsuke.stapler.jelly.jruby.erb.ERbClassTearOff;
009import org.kohsuke.stapler.test.AbstractStaplerTest;
010
011import java.io.StringWriter;
012
013/**
014 * @author Kohsuke Kawaguchi
015 */
016public class SanityTest extends AbstractStaplerTest {
017    public void testSanityERb() throws Exception {
018        MetaClass mc = webApp.getMetaClass(SanityTest.class);
019        ERbClassTearOff jr = mc.getTearOff(ERbClassTearOff.class);
020        Script s = jr.parseScript(getClass().getResource("test_sanity.erb"));
021
022        JellyContext context = mc.classLoader.getTearOff(JellyClassLoaderTearOff.class).createContext();
023        context.setVariable("name","ERB");
024
025        StringWriter out = new StringWriter();
026        s.run(context, XMLOutput.createXMLOutput(out));
027
028        assertTrue(out.toString().contains("Hello ERB!"));
029        assertTrue(out.toString().contains("Hello from Jelly to ERB"));
030        assertTrue(out.toString().contains("<i>Nested in ERB (47)</i>"));
031    }
032
033    public void testSanityHaml() throws Exception {
034        MetaClass mc = webApp.getMetaClass(SanityTest.class);
035        ERbClassTearOff jr = mc.getTearOff(ERbClassTearOff.class);
036        Script s = jr.parseScript(getClass().getResource("test_sanity.haml"));
037
038        JellyContext context = mc.classLoader.getTearOff(JellyClassLoaderTearOff.class).createContext();
039        context.setVariable("name","Haml");
040
041        StringWriter out = new StringWriter();
042        s.run(context, XMLOutput.createXMLOutput(out));
043
044        assertTrue(out.toString().contains("Hello Haml!"));
045        assertTrue(out.toString().contains("Hello from Jelly to Haml"));
046        assertTrue(out.toString().contains("Nested in Haml (47)"));
047    }
048}
