001package org.kohsuke.stapler.jelly.issue76;
002
003import com.gargoylesoftware.htmlunit.TextPage;
004import com.gargoylesoftware.htmlunit.WebClient;
005import com.gargoylesoftware.htmlunit.html.HtmlPage;
006import org.kohsuke.stapler.test.JettyTestCase;
007
008import java.net.URL;
009
010/**
011 * @author Kohsuke Kawaguchi
012 */
013public class Issue76Test extends JettyTestCase {
014    // bound to /robot
015    public final Robot robot = new Robot();
016
017    // protected version of /robot which mimics the URL structure but tweaks its routes
018    public final Protection protectedRobot = new Protection(robot);
019
020    public void testRouting() throws Exception {
021        WebClient wc = new WebClient();
022        HtmlPage p = wc.getPage(new URL(url, "robot/head/eye/3/"));
023        assertTrue(p.getWebResponse().getContentAsString().contains("This is eye 3"));
024
025        // protected parts have different index view
026        TextPage tp = wc.getPage(new URL(url, "protectedRobot/head/eye/3/"));
027        assertTrue(tp.getContent().startsWith("protected eye #3"));
028
029        tp = wc.getPage(new URL(url, "protectedRobot/head/nose"));
030        assertEquals("nose",tp.getContent().trim());
031
032        tp = wc.getPage(new URL(url, "protectedRobot/arm/hand/nail"));
033        assertEquals("/hand/nail",tp.getContent().trim());
034
035        tp = wc.getPage(new URL(url, "protectedRobot/arm"));
036        assertEquals("protected arm",tp.getContent().trim());
037
038        {// list lookup
039            p = wc.getPage(new URL(url, "robot/legs/0/"));
040            assertTrue(p.getWebResponse().getContentAsString().contains("left leg"));
041
042            tp = wc.getPage(new URL(url, "protectedRobot/legs/1/"));
043            assertEquals("protected right leg", tp.getContent().trim());
044        }
045
046        {// map lookup
047            p = wc.getPage(new URL(url, "robot/buttons/red/"));
048            assertTrue(p.getWebResponse().getContentAsString().contains("This is a button"));
049
050            tp = wc.getPage(new URL(url, "protectedRobot/buttons/red/"));
051            assertEquals("protected red button", tp.getContent().trim());
052        }
053    }
054}
