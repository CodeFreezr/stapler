001package org.kohsuke.stapler;
002
003import org.apache.commons.beanutils.Converter;
004
005/**
006 * @author Kohsuke Kawaguchi
007 */
008public class ObjectWithCustomConverter {
009    public final int x,y;
010
011    public ObjectWithCustomConverter(int x, int y) {
012        this.x = x;
013        this.y = y;
014    }
015
016    public static class StaplerConverterImpl implements Converter {
017        public Object convert(Class type, Object value) {
018            String[] tokens = value.toString().split(",");
019            return new ObjectWithCustomConverter(
020                    Integer.parseInt(tokens[0]),
021                    Integer.parseInt(tokens[1]));
022        }
023    }
024}
