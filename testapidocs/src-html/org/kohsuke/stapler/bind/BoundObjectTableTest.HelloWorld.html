001package org.kohsuke.stapler.bind;
002
003import com.gargoylesoftware.htmlunit.TextPage;
004import com.gargoylesoftware.htmlunit.WebClient;
005import org.kohsuke.stapler.HttpResponse;
006import org.kohsuke.stapler.StaplerResponse;
007import org.kohsuke.stapler.test.JettyTestCase;
008
009import java.io.IOException;
010import java.net.URL;
011
012/**
013 * @author Kohsuke Kawaguchi
014 */
015public class BoundObjectTableTest extends JettyTestCase {
016    /**
017     * Exports an object and see if it can be reached.
018     */
019    public void testExport() throws Exception {
020        TextPage page = new WebClient().getPage(new URL(url, "/bind"));
021        assertEquals("hello world",page.getContent());
022    }
023
024    public HttpResponse doBind() throws IOException {
025        Bound h = webApp.boundObjectTable.bind(new HelloWorld("hello world"));
026        System.out.println(h.getURL());
027        return h;
028    }
029
030    public static class HelloWorld {
031        private final String message;
032
033        public HelloWorld(String message) {
034            this.message = message;
035        }
036
037        public void doIndex(StaplerResponse rsp) throws IOException {
038            rsp.setContentType("text/plain;charset=UTF-8");
039            rsp.getWriter().write(message);
040        }
041    }
042}
