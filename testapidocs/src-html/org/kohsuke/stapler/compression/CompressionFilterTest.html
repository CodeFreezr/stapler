001package org.kohsuke.stapler.compression;
002
003import org.apache.commons.httpclient.HttpClient;
004import org.apache.commons.httpclient.HttpMethod;
005import org.apache.commons.httpclient.methods.GetMethod;
006import org.apache.commons.io.IOUtils;
007import org.kohsuke.stapler.StaplerRequest;
008import org.kohsuke.stapler.StaplerResponse;
009import org.kohsuke.stapler.test.JettyTestCase;
010import org.mortbay.jetty.Handler;
011import org.mortbay.jetty.servlet.Context;
012
013import java.io.ByteArrayInputStream;
014import java.io.ByteArrayOutputStream;
015import java.io.IOException;
016import java.net.HttpURLConnection;
017import java.net.URL;
018import java.util.Arrays;
019import java.util.zip.GZIPInputStream;
020import java.util.zip.GZIPOutputStream;
021
022public class CompressionFilterTest extends JettyTestCase {
023
024    @Override
025    protected void configure(Context context) {
026        super.configure(context);
027        context.addFilter(CompressionFilter.class,"/*", Handler.DEFAULT);
028    }
029
030    public void testDoubleCompression() throws Exception {
031        for (String endpoint : Arrays.asList("autoZip","ownZip")) {
032            HttpURLConnection con = (HttpURLConnection) new URL(this.url, endpoint).openConnection();
033            con.setRequestProperty("Accept-Encoding","gzip");
034            byte[] data = IOUtils.toByteArray(con.getInputStream());
035            data = IOUtils.toByteArray(new GZIPInputStream(new ByteArrayInputStream(data)));
036            assertEquals(new String(data), CONTENT);
037        }
038    }
039
040    /**
041     * Turns out {@link #testDoubleCompression()} was insufficient to properly test the case,
042     * as HttpURLConnection appears to ignores the content-length header and read the response to the end.
043     */
044    public void testDoubleCompression2() throws Exception {
045        for (String endpoint : Arrays.asList("autoZip","ownZip")) {
046            HttpClient hc = new HttpClient();
047            HttpMethod m = new GetMethod(this.url + "/"+endpoint);
048            m.setRequestHeader("Accept-Encoding", "gzip");
049            assertEquals(200, hc.executeMethod(m));
050            byte[] data = m.getResponseBody();
051
052            data = IOUtils.toByteArray(new GZIPInputStream(new ByteArrayInputStream(data)));
053            assertEquals(new String(data), CONTENT);
054        }
055    }
056
057    /**
058     * Simulate servlets that tries to handle Content-Encoding on its own
059     */
060    public void doOwnZip(StaplerResponse rsp) throws IOException {
061        rsp.setHeader("Content-Encoding", "gzip");
062        byte[] content = CONTENT.getBytes();
063        ByteArrayOutputStream baos = new ByteArrayOutputStream();
064        GZIPOutputStream o = new GZIPOutputStream(baos);
065        o.write(content);
066        o.close();
067
068        rsp.setContentLength(baos.size());
069        rsp.getOutputStream().write(baos.toByteArray());
070    }
071
072    public void doAutoZip(StaplerRequest req, StaplerResponse rsp) throws IOException {
073        byte[] content = CONTENT.getBytes();
074        rsp.getCompressedOutputStream(req).write(content);
075    }
076
077
078    private static final String CONTENT = "Hello World";
079}
