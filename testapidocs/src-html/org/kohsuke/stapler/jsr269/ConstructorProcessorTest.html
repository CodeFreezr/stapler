001package org.kohsuke.stapler.jsr269;
002
003import java.util.Collections;
004import net.java.dev.hickory.testing.Compilation;
005import org.junit.Test;
006import static org.junit.Assert.*;
007
008public class ConstructorProcessorTest {
009
010    @Test public void basicOutput() {
011        Compilation compilation = new Compilation();
012        compilation.addSource("some.pkg.Stuff").
013                addLine("package some.pkg;").
014                addLine("import org.kohsuke.stapler.DataBoundConstructor;").
015                addLine("public class Stuff {").
016                addLine("  @DataBoundConstructor public Stuff(int count, String name) {}").
017                addLine("}");
018        compilation.doCompile(null, "-source", "6");
019        assertEquals(Collections.emptyList(), Utils.filterSupportedSourceVersionWarnings(compilation.getDiagnostics()));
020        assertEquals("{constructor=count,name}", Utils.normalizeProperties(Utils.getGeneratedResource(compilation, "some/pkg/Stuff.stapler")));
021    }
022
023    @Test public void preAnnotationCompatibility() {
024        Compilation compilation = new Compilation();
025        compilation.addSource("some.pkg.Stuff").
026                addLine("package some.pkg;").
027                addLine("public class Stuff {").
028                addLine("  /** @stapler-constructor */ public Stuff(String name, int count) {}").
029                addLine("}");
030        compilation.doCompile(null, "-source", "6");
031        assertEquals(Collections.emptyList(), Utils.filterSupportedSourceVersionWarnings(compilation.getDiagnostics()));
032        assertEquals("{constructor=name,count}", Utils.normalizeProperties(Utils.getGeneratedResource(compilation, "some/pkg/Stuff.stapler")));
033    }
034
035    @Test public void JENKINS_11739() {
036        Compilation compilation = new Compilation();
037        compilation.addSource("some.pkg.Stuff").
038                addLine("package some.pkg;").
039                addLine("import org.kohsuke.stapler.DataBoundConstructor;").
040                addLine("public class Stuff {").
041                addLine("  @DataBoundConstructor public Stuff(int count, String name) {}").
042                addLine("}");
043        compilation.addSource("some.pkg.package-info").
044                addLine("package some.pkg;");
045        compilation.doCompile(null, "-source", "6");
046        assertEquals(Collections.emptyList(), Utils.filterSupportedSourceVersionWarnings(compilation.getDiagnostics()));
047        assertEquals("{constructor=count,name}", Utils.normalizeProperties(Utils.getGeneratedResource(compilation, "some/pkg/Stuff.stapler")));
048    }
049
