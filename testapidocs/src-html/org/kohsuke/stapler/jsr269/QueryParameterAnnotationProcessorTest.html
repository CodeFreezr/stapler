001package org.kohsuke.stapler.jsr269;
002
003import java.util.Collections;
004import net.java.dev.hickory.testing.Compilation;
005import org.junit.Test;
006import static org.junit.Assert.*;
007
008public class QueryParameterAnnotationProcessorTest {
009
010    @Test public void basicOutput() {
011        Compilation compilation = new Compilation();
012        compilation.addSource("some.pkg.Stuff").
013                addLine("package some.pkg;").
014                addLine("import org.kohsuke.stapler.QueryParameter;").
015                addLine("public class Stuff {").
016                addLine("  public void doOneThing(@QueryParameter String key) {}").
017                addLine("  public void doAnother(@QueryParameter(\"ignoredHere\") String name, @QueryParameter String address) {}").
018                addLine("}");
019        compilation.doCompile(null, "-source", "6");
020        assertEquals(Collections.emptyList(), Utils.filterSupportedSourceVersionWarnings(compilation.getDiagnostics()));
021        assertEquals("key", Utils.getGeneratedResource(compilation, "some/pkg/Stuff/doOneThing.stapler"));
022        assertEquals("name,address", Utils.getGeneratedResource(compilation, "some/pkg/Stuff/doAnother.stapler"));
023    }
024
025    // TODO nested classes use qualified rather than binary name
026
027}
