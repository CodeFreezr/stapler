027package org.kohsuke.stapler.jsr269;
028
029import java.util.Collections;
030import java.util.List;
031import java.util.Locale;
032import javax.tools.Diagnostic;
033import javax.tools.JavaFileObject;
034import net.java.dev.hickory.testing.Compilation;
035import org.junit.Test;
036import static org.junit.Assert.*;
037
038public class WebMethodAnnotationProcessorTest {
039
040    @Test public void doMethods() throws Exception {
041        Compilation compilation = new Compilation();
042        compilation.addSource("some.pkg.Stuff").
043                addLine("package some.pkg;").
044                addLine("import org.kohsuke.stapler.WebMethod;").
045                addLine("public class Stuff {").
046                addLine("  @WebMethod(name=\"hello\") public void doSomething() {}").
047                addLine("}");
048        compilation.doCompile(null, "-source", "6");
049        assertEquals(Collections.emptyList(), compilation.getDiagnostics());
050    }
051
052    @Test public void nonDoMethods() throws Exception {
053        Compilation compilation = new Compilation();
054        compilation.addSource("some.pkg.Stuff").
055                addLine("package some.pkg;").
056                addLine("import org.kohsuke.stapler.WebMethod;").
057                addLine("public class Stuff {").
058                addLine("  @WebMethod(name=\"hello\") public void something() {}").
059                addLine("}");
060        compilation.doCompile(null, "-source", "6");
061        List<Diagnostic<? extends JavaFileObject>> diagnostics = compilation.getDiagnostics();
062        assertEquals(1, diagnostics.size());
063        String msg = diagnostics.get(0).getMessage(Locale.ENGLISH);
064        assertTrue(msg, msg.contains("something"));
065    }
066
067}
