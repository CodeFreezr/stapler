027package org.kohsuke.stapler.jsr269;
028
029import java.util.Collections;
030import java.util.List;
031import java.util.Locale;
032import javax.tools.Diagnostic;
033import javax.tools.JavaFileObject;
034import net.java.dev.hickory.testing.Compilation;
035import org.junit.Test;
036import static org.junit.Assert.*;
037
038public class RequirePOSTAnnotationProcessorTest {
039
040    @Test public void fine() throws Exception {
041        Compilation compilation = new Compilation();
042        compilation.addSource("some.pkg.Stuff").
043                addLine("package some.pkg;").
044                addLine("import org.kohsuke.stapler.interceptor.RequirePOST;").
045                addLine("public class Stuff {").
046                addLine("  @RequirePOST public void doSomething() {}").
047                addLine("}");
048        compilation.doCompile(null, "-source", "6");
049        assertEquals(Collections.emptyList(), Utils.filterSupportedSourceVersionWarnings(compilation.getDiagnostics()));
050    }
051
052    @Test public void noAbstract() throws Exception {
053        Compilation compilation = new Compilation();
054        compilation.addSource("some.pkg.Stuff").
055                addLine("package some.pkg;").
056                addLine("import org.kohsuke.stapler.interceptor.RequirePOST;").
057                addLine("public abstract class Stuff {").
058                addLine("  @RequirePOST public abstract void doSomething();").
059                addLine("}");
060        compilation.doCompile(null, "-source", "6");
061        List<Diagnostic<? extends JavaFileObject>> diagnostics = Utils.filterSupportedSourceVersionWarnings(compilation.getDiagnostics());
062        assertEquals(1, diagnostics.size());
063        assertEquals(Diagnostic.Kind.ERROR, diagnostics.get(0).getKind());
064        String msg = diagnostics.get(0).getMessage(Locale.ENGLISH);
065        assertTrue(msg, msg.contains("abstract"));
066    }
067
068    @Test public void failsToOverride() throws Exception {
069        Compilation compilation = new Compilation();
070        compilation.addSource("some.pkg.Stuff").
071                addLine("package some.pkg;").
072                addLine("import org.kohsuke.stapler.interceptor.RequirePOST;").
073                addLine("public class Stuff {").
074                addLine("  @RequirePOST public void doSomething() {}").
075                addLine("}");
076        compilation.addSource("some.pkg.SpecialStuff").
077                addLine("package some.pkg;").
078                addLine("public class SpecialStuff extends Stuff {").
079                addLine("  @Override public void doSomething() {}").
080                addLine("}");
081        compilation.doCompile(null, "-source", "6");
082        List<Diagnostic<? extends JavaFileObject>> diagnostics = Utils.filterSupportedSourceVersionWarnings(compilation.getDiagnostics());
083        assertEquals(1, diagnostics.size());
084        assertEquals(Diagnostic.Kind.WARNING, diagnostics.get(0).getKind());
085        assertEquals("some/pkg/SpecialStuff.java", diagnostics.get(0).getSource().toUri().toString());
086        String msg = diagnostics.get(0).getMessage(Locale.ENGLISH);
087        assertTrue(msg, msg.contains("overrid"));
088    }
089
090    @Test public void overridesOK() throws Exception {
091        Compilation compilation = new Compilation();
092        compilation.addSource("some.pkg.Stuff").
093                addLine("package some.pkg;").
094                addLine("import org.kohsuke.stapler.interceptor.RequirePOST;").
095                addLine("public class Stuff {").
096                addLine("  @RequirePOST public void doSomething() {}").
097                addLine("}");
098        compilation.addSource("some.pkg.SpecialStuff").
099                addLine("package some.pkg;").
100                addLine("import org.kohsuke.stapler.interceptor.RequirePOST;").
101                addLine("public class SpecialStuff extends Stuff {").
102                addLine("  @RequirePOST  @Override public void doSomething() {}").
103                addLine("}");
104        compilation.doCompile(null, "-source", "6");
105        assertEquals(Collections.emptyList(), Utils.filterSupportedSourceVersionWarnings(compilation.getDiagnostics()));
106    }
107
108}
