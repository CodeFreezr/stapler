029    public void testExclude() throws Exception {
030        final String[] msg = new String[1];
031        WebClient wc = new WebClient();
032        wc.setAlertHandler(new AlertHandler() {
033            public void handleAlert(Page page, String message) {
034                msg[0] = message;
035            }
036        });
037        HtmlPage page = wc.getPage(new URL(url, "/"));
038
039        page.executeJavaScript("v.getSomeExcludedData(callback);");
040
041        Map json = (Map)JSON.parse(msg[0]);
042        assertTrue(json.containsKey("name"));
043        assertTrue(json.containsKey("description"));
044        assertFalse(json.containsKey("secret"));
045    }
046
047    public void testInclude() throws Exception {
048        final String[] msg = new String[1];
049        WebClient wc = new WebClient();

061        assertFalse(json.containsKey("description"));
062        assertFalse(json.containsKey("secret"));
063    }
064
065    public void testExcludeList() throws Exception {
066        final String[] msg = new String[1];
067        WebClient wc = new WebClient();
068        wc.setAlertHandler(new AlertHandler() {
069            public void handleAlert(Page page, String message) {
070                msg[0] = message;
071            }
072        });
073        HtmlPage page = wc.getPage(new URL(url, "/"));
074
075        page.executeJavaScript("v.getSomeExcludedList(callback);");
076
077        Object[] json = (Object[])JSON.parse(msg[0]);
078        assertEquals(3, json.length);
079        for (Object o : json) {
080            Map map = (Map)o;
081            assertTrue(map.containsKey("name"));
082            assertTrue(map.containsKey("description"));
083            assertFalse(map.containsKey("secret"));
084        }
085    }

087    @JsonOutputFilter(excludes = {"secret"})
088    @JavaScriptMethod
089    public MyData getSomeExcludedData() {
090        return new MyData("Bob", "the builder", "super secret value");
091    }
092
093    @JsonOutputFilter(excludes = {"secret"})
094    @JavaScriptMethod
095    public List<MyData> getSomeExcludedList() {
096        return Arrays.asList(new MyData("Bob", "the builder", "super secret value"),
097                new MyData("Lisa", "the coder", "even more super secret"),
098                new MyData("Jenkins", "the butler", "really secret as well"));
099    }
100
101    @JsonOutputFilter(includes = {"name"})
102    @JavaScriptMethod
103    public MyData getSomeIncludedData() {

107    public void doIndex(StaplerResponse rsp) throws IOException {
108        rsp.setContentType("text/html");
109        PrintWriter w = rsp.getWriter();
110        w.println("<html><body><script src='prototype'></script><script src='script'></script>");
111        w.println("<script>var v = "+ WebApp.getCurrent().boundObjectTable.bind(this).getProxyScript()+";</script>");
112        w.println("<script>var callback = function(t){var x=t.responseObject(); alert(Object.toJSON(x)); };</script>");
113        w.println("</body></html>");
114    }
